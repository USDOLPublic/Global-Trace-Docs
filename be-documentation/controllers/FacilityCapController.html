import {
    Body,
    Controller,
    Get,
    HttpCode,
    HttpStatus,
    Post,
    Put,
    UploadedFiles,
    UseGuards,
    UseInterceptors
} from '@nestjs/common';
import { ApiBearerAuth, ApiConsumes, ApiOperation, ApiParam, ApiTags } from '@nestjs/swagger';
import { BaseController } from '~core/http/controllers/base.controller';
import { JwtAuthGuard } from '~users/http/guards/jwt-auth.guard';
import { AssignCapDto } from '../dto/assign-cap.dto';
import { CapService } from '~corrective-action-plans/services/cap.service';
import { CapEntity } from '~corrective-action-plans/entities/cap.entity';
import { Id } from '@diginexhk/nestjs-base-decorator';
import { CurrentUser } from '~users/decorators/current-user.decorator';
import { UserEntity } from '~users/entities/user.entity';
import { CategoryRiskDetailType } from '~risk-assessments/types/category-risk.type';
import { CategoryRiskDetailResponse } from '~facilities/http/response/category-risk-detail.response';
import { ResponseModel } from '@diginexhk/nestjs-response';
import { DataSource } from 'typeorm';
import { UpdateCapDto } from '../dto/update-cap.dto';
import { PermissionGuard } from '~users/http/guards/permission.guard';
import { PermissionEnum } from '~role-permissions/enums/permission.enum';
import { FilesInterceptor } from '@nestjs/platform-express';
import { RequestExtensionCapDto } from '../dto/request-extension-cap.dto.';
import { ValidateAssignCapDto } from '../dto/validate-assign-cap.dto';

@Controller('facilities/:facilityId/caps')
@ApiTags('Cap')
@ApiBearerAuth()
@UseGuards(JwtAuthGuard)
export class FacilityCapController extends BaseController {
    constructor(
        private dataSource: DataSource,
        private capService: CapService
    ) {
        super();
    }

    @Post()
    @ApiOperation({ description: 'Assign cap' })
    @ApiParam({
        name: 'facilityId',
        description: 'supplier id',
        type: 'string'
    })
    @UseGuards(PermissionGuard(PermissionEnum.ASSIGN_CAP))
    @HttpCode(HttpStatus.CREATED)
    @UseInterceptors(FilesInterceptor('files'))
    @ApiConsumes('multipart/form-data')
    async assignCap(
        @CurrentUser() user: UserEntity,
        @Id('facilityId') facilityId: string,
        @Body() dto: AssignCapDto,
        @UploadedFiles() files: Array<Express.Multer.File>
    ): Promise<CapEntity> {
        return this.dataSource.transaction((manager) => {
            return this.capService.withTransaction(manager).assignCap(user, facilityId, dto, files);
        });
    }

    @Post('validate')
    @ApiOperation({ description: 'Validate assign CAP' })
    @ApiParam({
        name: 'facilityId',
        description: 'Supplier ID',
        type: 'string'
    })
    async validateAssignCap(@Id('facilityId') facilityId: string, @Body() dto: ValidateAssignCapDto): Promise<void> {
        return this.capService.validateAssignCap(facilityId, dto);
    }

    @Put(':id')
    @ApiOperation({ description: 'Assign cap' })
    @UseInterceptors(FilesInterceptor('files'))
    @ApiConsumes('multipart/form-data')
    @ApiParam({
        name: 'facilityId',
        description: 'supplier id',
        type: 'string'
    })
    @ApiParam({
        name: 'id',
        description: 'cap id',
        type: 'string'
    })
    @UseGuards(PermissionGuard(PermissionEnum.VIEW_CAP))
    @HttpCode(HttpStatus.NO_CONTENT)
    async updateCap(
        @CurrentUser() user: UserEntity,
        @Id('facilityId') facilityId: string,
        @Id() id: string,
        @Body() dto: UpdateCapDto,
        @UploadedFiles() files: Array<Express.Multer.File>
    ): Promise<CapEntity> {
        return this.dataSource.transaction((manager) => {
            return this.capService.withTransaction(manager).updateCap(user, facilityId, id, dto, files);
        });
    }

    @Put(':id/request-extension')
    @ApiOperation({ description: 'Request extension' })
    @ApiParam({
        name: 'facilityId',
        description: 'supplier id',
        type: 'string'
    })
    @ApiParam({
        name: 'id',
        description: 'cap id',
        type: 'string'
    })
    @UseGuards(PermissionGuard(PermissionEnum.VIEW_CAP))
    @HttpCode(HttpStatus.NO_CONTENT)
    async requestExtension(
        @CurrentUser() user: UserEntity,
        @Id('facilityId') facilityId: string,
        @Id() id: string,
        @Body() dto: RequestExtensionCapDto
    ): Promise<void> {
        return this.capService.requestExtension(user, facilityId, id, dto);
    }

    @Put(':id/request-extension/approve')
    @ApiOperation({ description: 'Approve request extension' })
    @ApiParam({
        name: 'facilityId',
        description: 'supplier id',
        type: 'string'
    })
    @ApiParam({
        name: 'id',
        description: 'cap id',
        type: 'string'
    })
    @UseGuards(PermissionGuard(PermissionEnum.ASSIGN_CAP))
    @HttpCode(HttpStatus.NO_CONTENT)
    async approveExtensionRequest(@Id('facilityId') facilityId: string, @Id() id: string): Promise<void> {
        return this.capService.approveExtensionRequest(facilityId, id);
    }

    @Put(':id/request-extension/decline')
    @ApiOperation({ description: 'Decline request extension' })
    @ApiParam({
        name: 'facilityId',
        description: 'supplier id',
        type: 'string'
    })
    @ApiParam({
        name: 'id',
        description: 'cap id',
        type: 'string'
    })
    @UseGuards(PermissionGuard(PermissionEnum.ASSIGN_CAP))
    @HttpCode(HttpStatus.NO_CONTENT)
    async declineExtensionRequest(@Id('facilityId') facilityId: string, @Id() id: string): Promise<void> {
        return this.capService.declineExtensionRequest(facilityId, id);
    }

    @Get(':id')
    @ResponseModel(CategoryRiskDetailResponse)
    @UseGuards(PermissionGuard(PermissionEnum.VIEW_CAP))
    @ApiOperation({ description: 'Get detail a cap' })
    @ApiParam({
        name: 'facilityId',
        description: 'supplier id',
        type: 'string'
    })
    @ApiParam({
        name: 'id',
        description: 'cap id',
        type: 'string'
    })
    getDetailCap(
        @CurrentUser() user: UserEntity,
        @Id('facilityId') facilityId: string,
        @Id() id: string
    ): Promise<CategoryRiskDetailType> {
        return this.capService.getDetailCap(user, facilityId, id);
    }
}
