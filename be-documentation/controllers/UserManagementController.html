import { Body, Controller, Delete, Get, HttpCode, HttpStatus, Post, Put, UseGuards } from '@nestjs/common';
import { UserService } from '../../services/user.service';
import { BaseController } from '~core/http/controllers/base.controller';
import { CurrentUser } from '../../decorators/current-user.decorator';
import { UserEntity } from '../../entities/user.entity';
import { ApiBearerAuth, ApiOperation, ApiParam, ApiTags } from '@nestjs/swagger';
import { JwtAuthGuard } from '../guards/jwt-auth.guard';
import { DataSource } from 'typeorm';
import { CreateNewUserDto } from '../dto/create-new-user.dto';
import { ResponseModel } from '@diginexhk/nestjs-response';
import { InviteUserResponse } from '../response/invite-user.response';
import { UserPaginationResponse } from '../response/user-pagination.response';
import { Id, Pagination, PaginationParams, SortMultipleParams, Sorts } from '@diginexhk/nestjs-base-decorator';
import { PaginationCollection } from '@diginexhk/typeorm-helper';
import { UpdateUserDto } from '../dto/update-user.dto';
import { UserManagementGuard } from '../guards/user-management.guard';

@Controller('user-managements')
@ApiTags('User Managements')
@ApiBearerAuth()
@UseGuards(JwtAuthGuard)
export class UserManagementController extends BaseController {
    constructor(
        private dataSource: DataSource,
        private userService: UserService
    ) {
        super();
    }

    @Post()
    @UseGuards(UserManagementGuard)
    @ResponseModel(InviteUserResponse)
    @ApiOperation({ description: 'Create new user for  producer, assessment (ESG) and brand roles' })
    createNewUser(@CurrentUser() user: UserEntity, @Body() dto: CreateNewUserDto): Promise<UserEntity> {
        return this.dataSource.transaction((manager) =>
            this.userService.withTransaction(manager).createNewUser(user, dto)
        );
    }

    @Get()
    @ResponseModel(UserPaginationResponse)
    @ApiOperation({ description: 'Get list users for producer, assessment (ESG) and brand roles' })
    getUsersByCurrentUser(
        @Pagination() page: PaginationParams,
        @Sorts({
            allowedFields: ['userType', 'name', 'status', 'lastLoginAt', 'createdAt'],
            default: { sortField: 'createdAt', sortDirection: 'DESC' }
        })
        sort: SortMultipleParams[],
        @CurrentUser() currentUser: UserEntity
    ): Promise<PaginationCollection<UserEntity>> {
        return this.userService.getUsersByCurrentUser(currentUser, page, sort);
    }

    @Post(':id/resend-invitation')
    @UseGuards(UserManagementGuard)
    @ApiOperation({ description: 'Resend invitation user for producer, assessment (ESG) and brand roles' })
    @ApiParam({
        name: 'id',
        description: 'User id',
        type: 'string'
    })
    @HttpCode(HttpStatus.NO_CONTENT)
    resendInvitation(@CurrentUser() currentUser: UserEntity, @Id() id: string): Promise<void> {
        return this.userService.resendInvitation(currentUser, id);
    }

    @Put(':id')
    @UseGuards(UserManagementGuard)
    @ApiOperation({ description: 'Update info user for producer, assessment (ESG) and brand roles' })
    @ApiParam({
        name: 'id',
        description: 'User id',
        type: 'string'
    })
    @HttpCode(HttpStatus.NO_CONTENT)
    updateUser(@CurrentUser() currentUser: UserEntity, @Id() id: string, @Body() dto: UpdateUserDto): Promise<void> {
        return this.userService.updateUser(currentUser, id, dto);
    }

    @Delete(':id')
    @UseGuards(UserManagementGuard)
    @ApiOperation({ description: 'Delete a user for producer, assessment (ESG) and brand roles' })
    @ApiParam({
        name: 'id',
        description: 'User id',
        type: 'string'
    })
    @HttpCode(HttpStatus.NO_CONTENT)
    deleteUser(@CurrentUser() user: UserEntity, @Id() id: string): Promise<void> {
        return this.dataSource.transaction((manager) => this.userService.withTransaction(manager).deleteUser(user, id));
    }
}
