import { Id } from '@diginexhk/nestjs-base-decorator';
import { Body, Controller, Delete, Get, HttpCode, HttpStatus, Post, Put, Query, UseGuards } from '@nestjs/common';
import { BaseController } from '~core/http/controllers/base.controller';
import { ApiBearerAuth, ApiOperation, ApiParam, ApiTags } from '@nestjs/swagger';
import { SupplyChainService } from '../../services/supply-chain.service';
import { JwtAuthGuard } from '~users/http/guards/jwt-auth.guard';
import { RoleGuard } from '~users/http/guards/role.guard';
import { UserRoleEnum } from '~role-permissions/enums/user-role.enum';
import { UpdateSupplyChainNodeDto } from '../dto/update-supply-chain-node.dto';
import { AddRequestToBody } from '~core/decorators/add-request-to-body.decorator';
import { SupplyChainNodeMapping } from '~supply-chains/types/supply-chain-node-mapping.type';
import { DataSource } from 'typeorm';
import { AddSupplyChainNodesDto } from '../dto/add-supply-chain-nodes.dto';
import { UpdateSupplyChainNodeMetadataDto } from '../dto/update-supply-chain-node-metadata.dto';
import { NodeImpacts } from '~supply-chains/types/node-impacts.dto';
import { DeleteSupplyChainNodesDto } from '../dto/delete-supply-chain-nodes.dto';
import { SupplyChainNodeService } from '~supply-chains/services/supply-chain-node.service';

@Controller('supply-chain-nodes')
@ApiTags('SupplyChain')
@ApiBearerAuth()
@UseGuards(JwtAuthGuard, RoleGuard(UserRoleEnum.SUPER_ADMIN))
export class SupplyChainNodeController extends BaseController {
    constructor(
        private dataSource: DataSource,
        private supplyChainService: SupplyChainService,
        private supplyChainNodeService: SupplyChainNodeService
    ) {
        super();
    }

    @Post()
    @ApiOperation({
        description: 'Add a node'
    })
    @HttpCode(HttpStatus.CREATED)
    addSupplyChainNode(@Body() dto: AddSupplyChainNodesDto): Promise<SupplyChainNodeMapping[]> {
        return this.dataSource.transaction((manager) => {
            return this.supplyChainService.withTransaction(manager).addSupplyChainNodes(dto);
        });
    }

    @Put(':id')
    @ApiParam({ name: 'id', description: 'Supply chain node ID', type: 'string' })
    @ApiOperation({ description: 'Update a node in supply chain' })
    @HttpCode(HttpStatus.NO_CONTENT)
    @AddRequestToBody()
    updateSupplyChainNode(@Body() dto: UpdateSupplyChainNodeDto, @Id() id: string): Promise<void> {
        return this.dataSource.transaction((manager) => {
            return this.supplyChainService.withTransaction(manager).updateSupplyChainNode(id, dto);
        });
    }

    @Get('deletion-impacts')
    @ApiOperation({ description: 'Get impacts after deleting supply chain nodes' })
    getDeletionImpacts(@Query() dto: DeleteSupplyChainNodesDto): Promise<NodeImpacts> {
        return this.supplyChainNodeService.getDeletionNodesImpact(dto);
    }

    @Delete('multiple')
    @ApiOperation({ description: 'Delete supply chain nodes' })
    @HttpCode(HttpStatus.NO_CONTENT)
    deleteSupplyChainNodes(@Query() dto: DeleteSupplyChainNodesDto): Promise<void> {
        return this.supplyChainNodeService.deleteSupplyChainNodes(dto);
    }

    @Post('metadata/:metadataId/update-impacts')
    @ApiParam({ name: 'metadataId', description: 'Supply chain metadata ID', type: 'string' })
    @ApiOperation({ description: 'Get impacts after updating supply chain' })
    getUpdateImpacts(
        @Id('metadataId') metadataId: string,
        @Body() dto: UpdateSupplyChainNodeMetadataDto
    ): Promise<NodeImpacts> {
        return this.supplyChainNodeService.getUpdateMetadataImpacts(metadataId, dto);
    }

    @Put('metadata/:metadataId')
    @ApiParam({ name: 'metadataId', description: 'Supply chain node metadata ID', type: 'string' })
    @ApiOperation({ description: 'Update a supply chain node metadata' })
    @HttpCode(HttpStatus.NO_CONTENT)
    @AddRequestToBody()
    updateSupplyChainMetadata(
        @Id('metadataId') metadataId: string,
        @Body() dto: UpdateSupplyChainNodeMetadataDto
    ): Promise<void> {
        return this.dataSource.transaction((manager) => {
            return this.supplyChainNodeService.withTransaction(manager).updateSupplyChainMetadata(metadataId, dto);
        });
    }

    @Delete('metadata/:id')
    @ApiParam({ name: 'id', description: 'Supply chain node metadata ID', type: 'string' })
    @ApiOperation({ description: 'Delete a node in supply chain node metadata' })
    @HttpCode(HttpStatus.NO_CONTENT)
    deleteSupplyChainNode(@Id() id: string): Promise<void> {
        return this.dataSource.transaction(async (manager) => {
            await this.supplyChainNodeService.withTransaction(manager).deleteSupplyChainMetadata(id);
        });
    }
}
