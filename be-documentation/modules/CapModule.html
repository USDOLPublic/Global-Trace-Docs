import { TypeOrmHelperModule } from '@diginexhk/typeorm-helper';
import { Module, forwardRef } from '@nestjs/common';
import { CapRepository } from './repositories/cap.repository';
import { CapEntity } from './entities/cap.entity';
import { CapCommentRepository } from './repositories/cap-comment.repository';
import { CapCommentEntity } from './entities/cap-comment.entity';
import { CapCommentService } from './services/cap-comment.service';
import { RolePermissionModule } from '~role-permissions/role-permission.module';
import { FacilityModule } from '~facilities/facility.module';
import { CapController } from './http/controllers/cap.controller';
import { CapService } from './services/cap.service';
import { CapCommentController } from './http/controllers/cap-comment.controller';
import { FacilityCapController } from './http/controllers/facility-cap.controller';
import { UploadModule } from '~uploads/upload.module';
import { UpdateCapStatusOnTargetDateElapsedTask } from './tasks/update-caps-status-on-target-date-elapsed.task';
import { RequestCapExtensionEntity } from './entities/request-extension.entity';
import { RequestExtensionRepository } from './repositories/request-extension.repository';
import { DeclineExpiredRequestsTask } from './tasks/decline-expired-requests.task';

@Module({
    providers: [CapService, CapCommentService, UpdateCapStatusOnTargetDateElapsedTask, DeclineExpiredRequestsTask],
    exports: [CapService],
    controllers: [FacilityCapController, CapCommentController, CapController],
    imports: [
        TypeOrmHelperModule.forCustomRepository([
            CapRepository,
            CapEntity,
            CapCommentRepository,
            CapCommentEntity,
            RequestExtensionRepository,
            RequestCapExtensionEntity
        ]),
        RolePermissionModule,
        forwardRef(() => FacilityModule),
        UploadModule
    ]
})
export class CapModule {}
