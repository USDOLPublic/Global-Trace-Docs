import { forwardRef, Module } from '@nestjs/common';
import { LaborRiskModule } from '~labor-risks/labor-risk.module';
import { CapModule } from '~corrective-action-plans/cap.module';
import { CommodityRiskModule } from '~commodity-risks/commodity-risk.module';
import { SelfAssessmentModule } from '~self-assessments/self-assessment.module';
import { CategoryUtilService } from './services/category-util.service';
import { CategoryModule } from '~categories/category.module';
import { IndicatorController } from './http/controllers/indicator.controller';
import { ExternalRiskIndexController } from './http/controllers/external-risk-index.controller';
import { RolePermissionModule } from '~role-permissions/role-permission.module';
import { UserApiLimitModule } from '~user-api-limits/user-api-limit.module';

@Module({
    controllers: [IndicatorController, ExternalRiskIndexController],
    providers: [CategoryUtilService],
    imports: [
        forwardRef(() => LaborRiskModule),
        CapModule,
        CommodityRiskModule,
        SelfAssessmentModule,
        CategoryModule,
        RolePermissionModule,
        forwardRef(() => UserApiLimitModule)
    ],
    exports: [CategoryUtilService]
})
export class CategoryUtilModule {}
