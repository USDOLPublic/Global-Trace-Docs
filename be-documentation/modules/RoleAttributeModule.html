import { TypeOrmHelperModule } from '@diginexhk/typeorm-helper';
import { Module, forwardRef } from '@nestjs/common';
import { RolePermissionModule } from '~role-permissions/role-permission.module';
import { RoleAttributeController } from './http/controllers/role-attribute.controller';
import { RoleAttributeService } from './services/role-attribute.service';
import { AttributeEntity } from '~product-definitions/entities/attribute.entity';
import { AttributeRepository } from '~product-definitions/repositories/attribute.repository';
import { FacilityRoleAttributeRepository } from './repositories/facility-role-attribute.repository';
import { FacilityRoleAttributeEntity } from './entities/facility-role-attribute.entity';
import { RoleAttributeRepository } from './repositories/role-attribute.repository';
import { RoleAttributeEntity } from './entities/role-attribute.entity';
import { FacilityRoleAttributeService } from './services/facility-role-attribute.service';
import { ValidateRoleAttributeService } from './services/validate-role-attribute.service';
import { LocationModule } from '~locations/location.module';
import { IdentifierSystemAttributeController } from './http/controllers/identifier-system-attribute.controller';
import { HttpClientModule } from '~http-client/http-client.module';
import { FacilityModule } from '~facilities/facility.module';

@Module({
    providers: [RoleAttributeService, FacilityRoleAttributeService, ValidateRoleAttributeService],
    exports: [RoleAttributeService, FacilityRoleAttributeService, ValidateRoleAttributeService],
    controllers: [RoleAttributeController, IdentifierSystemAttributeController],
    imports: [
        TypeOrmHelperModule.forCustomRepository([
            FacilityRoleAttributeRepository,
            FacilityRoleAttributeEntity,
            RoleAttributeRepository,
            RoleAttributeEntity,
            AttributeRepository,
            AttributeEntity
        ]),
        forwardRef(() => RolePermissionModule),
        LocationModule,
        forwardRef(() => HttpClientModule),
        forwardRef(() => FacilityModule)
    ]
})
export class RoleAttributeModule {}
