import { UserEntity } from '../entities/user.entity';
import { BaseRepository } from '~core/repositories/base.repository';
import { FindOneOptions } from 'typeorm/find-options/FindOneOptions';
import { RoleEntity } from '~role-permissions/entities/role.entity';
import { CustomRepository } from '@diginexhk/typeorm-helper';
import { UserTypeEnum } from '~users/enums/user-type.enum';
import { In } from 'typeorm';
import { RoleTypeEnum } from '~role-permissions/enums/role-type.enum';
import { FindUsersInFacilityParam } from '~users/types/find-users-in-facility-param.type';

@CustomRepository(UserEntity)
export class UserRepository extends BaseRepository<UserEntity> {
    findByEmail(email: string, options?: FindOneOptions<UserEntity>) {
        return this.findOne({ where: { email }, ...options });
    }

    addRole(user: UserEntity, role: RoleEntity) {
        return this.createQueryBuilder().relation(UserEntity, 'role').of(user).add(role);
    }

    findUserById(id: string, options: FindOneOptions<UserEntity>) {
        return this.findById(id, {
            ...options,
            relations: options?.relations || [
                'role',
                'role.permissions',
                'role.hasPermissions',
                'permissions',
                'role.selfAssessmentUploadFiles'
            ]
        });
    }

    countUsersInFacility(facilityId: string): Promise<number> {
        return this.createQueryBuilder('user')
            .innerJoin(`user.facilities`, 'facilities', `'${facilityId}' = facilities.id`)
            .getCount();
    }

    findOneValidUser(userId: string, facilityId: string): Promise<UserEntity> {
        return this.createQueryBuilder('user')
            .where({ id: userId, userType: In([UserTypeEnum.ADMIN, UserTypeEnum.STANDARD]) })
            .innerJoin('user.facilities', 'facilities', `'${facilityId}' = facilities.id`)
            .innerJoin(
                'user.role',
                'role',
                `role.type IN ('${RoleTypeEnum.BRAND}','${RoleTypeEnum.LABOR}','${RoleTypeEnum.PRODUCT}')`
            )
            .getOne();
    }

    getFirstUserInFacility(facilityId: string): Promise<UserEntity> {
        return this.createQueryBuilder('user')
            .where({ userType: UserTypeEnum.ADMIN })
            .innerJoin(`user.facilities`, 'facilities', `'${facilityId}' = facilities.id`)
            .orderBy('user.createdAt', 'ASC')
            .getOne();
    }

    findUsersInFacility({ facilityId, userType, withDeleted }: FindUsersInFacilityParam): Promise<UserEntity[]> {
        const query = this.createQueryBuilder('user').innerJoin(
            `user.facilities`,
            'facilities',
            `'${facilityId}' = facilities.id`
        );

        if (userType) {
            query.where({ userType });
        }

        if (withDeleted) {
            query.withDeleted();
        }

        return query.getMany();
    }
}
