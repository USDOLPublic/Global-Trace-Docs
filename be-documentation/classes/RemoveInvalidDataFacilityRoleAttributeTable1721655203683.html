import { In, QueryRunner } from 'typeorm';
import { BaseMigration } from '@diginexhk/typeorm-helper';
import { FacilityRoleAttributeEntity } from '~role-attributes/entities/facility-role-attribute.entity';
import { RoleAttributeTypeEnum } from '~role-attributes/enums/role-attribute-type.enum';
import { RoleTypeEnum } from '~role-permissions/enums/role-type.enum';
import { RoleEntity } from '~role-permissions/entities/role.entity';
import { RoleAttributeEntity } from '~role-attributes/entities/role-attribute.entity';
import { difference, uniq } from 'lodash';
import { FacilityEntity } from '~facilities/entities/facility.entity';

export class RemoveInvalidDataFacilityRoleAttributeTable1721655203683 extends BaseMigration {
    async run(queryRunner: QueryRunner): Promise<void> {
        await this.updateFacilityRoleAttributes(queryRunner);
        const productRoleIds = await this.getProductRoleIds(queryRunner);

        if (!productRoleIds.length) {
            return;
        }

        const validProductRoleIds = await this.getValidProductRoleIds(queryRunner, productRoleIds);
        const invalidProductRoleIds = difference(productRoleIds, validProductRoleIds);

        const productFacilities = await this.getProductFacilities(queryRunner, productRoleIds);

        const [facilityIdsHasNotRoleAttributes, facilityIdsHasRoleAttributes] = this.getFacilityIdsHasNotRoleAttributes(
            productFacilities,
            validProductRoleIds,
            invalidProductRoleIds
        );

        const facilityIdsHasInvalidAttributes = await this.checkAndGetFacilityIdsHasInvalidAttributes(
            queryRunner,
            facilityIdsHasRoleAttributes
        );

        await queryRunner.manager.softDelete(FacilityRoleAttributeEntity, {
            facilityId: In(facilityIdsHasNotRoleAttributes.concat(facilityIdsHasInvalidAttributes)),
            roleAttributeType: RoleAttributeTypeEnum.OPEN_SUPPLY_HUB_ID
        });
    }

    private async checkAndGetFacilityIdsHasInvalidAttributes(
        queryRunner: QueryRunner,
        facilityIdsHasRoleAttributes: string[]
    ): Promise<string[]> {
        const openSupplyHubIdAttributes = await queryRunner.manager.find(FacilityRoleAttributeEntity, {
            select: ['id', 'facilityId', 'options'],
            where: {
                facilityId: In(facilityIdsHasRoleAttributes),
                roleAttributeType: RoleAttributeTypeEnum.OPEN_SUPPLY_HUB_ID
            }
        });

        return openSupplyHubIdAttributes.reduce((acc, { facilityId, options }) => {
            if (options.length < 5) {
                acc.push(facilityId);
            }
            return acc;
        }, [] as string[]);
    }

    private getFacilityIdsHasNotRoleAttributes(
        productFacilities: FacilityEntity[],
        validProductRoleIds: string[],
        invalidProductRoleIds: string[]
    ): string[][] {
        return productFacilities.reduce(
            (acc, facility) => {
                if (facility?.firstAdmin?.updatedProfileAt) {
                    return acc;
                }

                if (invalidProductRoleIds.includes(facility.typeId)) {
                    acc[0].push(facility.id);
                }

                if (validProductRoleIds.includes(facility.typeId)) {
                    acc[1].push(facility.id);
                }

                return acc;
            },
            [[], []] as string[][]
        );
    }

    private async getProductFacilities(queryRunner: QueryRunner, productRoleIds: string[]): Promise<FacilityEntity[]> {
        return queryRunner.manager.find(FacilityEntity, {
            select: ['id', 'typeId'],
            where: { typeId: In(productRoleIds) },
            relations: ['users']
        });
    }

    private async updateFacilityRoleAttributes(queryRunner: QueryRunner): Promise<void> {
        await queryRunner.manager.update(
            FacilityRoleAttributeEntity,
            {
                roleAttributeType: RoleAttributeTypeEnum.OPEN_SUPPLY_HUB_ID
            },
            { userId: null }
        );
    }

    private async getProductRoleIds(queryRunner: QueryRunner): Promise<string[]> {
        const productRoles = await queryRunner.manager.find(RoleEntity, {
            select: ['id'],
            where: { type: RoleTypeEnum.PRODUCT }
        });

        return productRoles.map(({ id }) => id);
    }

    private async getValidProductRoleIds(queryRunner: QueryRunner, productRoleIds: string[]) {
        const existedRoleAttributes = await queryRunner.manager.find(RoleAttributeEntity, {
            select: ['roleId'],
            where: { roleId: In(productRoleIds), roleAttributeType: RoleAttributeTypeEnum.OPEN_SUPPLY_HUB_ID }
        });

        return uniq(existedRoleAttributes.map(({ roleId }) => roleId));
    }
}
