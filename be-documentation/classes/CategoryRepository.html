import { CustomRepository } from '@diginexhk/typeorm-helper';
import { BaseRepository } from '~core/repositories/base.repository';
import { CategoryEntity } from '../entities/category.entity';
import { CategoryTypeEnum } from '~categories/enums/category-type.enum';

@CustomRepository(CategoryEntity)
export class CategoryRepository extends BaseRepository<CategoryEntity> {
    async getIgnoreCategoryIds(type: CategoryTypeEnum): Promise<string[]> {
        if (type === CategoryTypeEnum.CATEGORY) {
            const records = await this.createQueryBuilder('category')
                .select('category.id', 'id')
                .leftJoin('Category', 'children', 'children.categoryId = category.id AND children.type = :childType', {
                    childType: CategoryTypeEnum.SUB_INDICATOR
                })
                .where('category.type = :parentType', { parentType: CategoryTypeEnum.CATEGORY })
                .groupBy('category.id')
                .having('COUNT(children.id) = 0')
                .getRawMany();
            return records.map((record) => record.id);
        }
        if (type === CategoryTypeEnum.INDICATOR) {
            const records = await this.createQueryBuilder('category')
                .select('category.id', 'id')
                .leftJoin('Category', 'children', 'children.parentId = category.id AND children.type = :childType', {
                    childType: CategoryTypeEnum.SUB_INDICATOR
                })
                .where('category.type = :parentType', { parentType: CategoryTypeEnum.INDICATOR })
                .groupBy('category.id')
                .having('COUNT(children.id) = 0')
                .getRawMany();
            return records.map((record) => record.id);
        }

        return [];
    }

    getAllNestedCategories(): Promise<CategoryEntity[]> {
        return this.createQueryBuilder('category')
            .where('category.type = :type', {
                type: CategoryTypeEnum.CATEGORY
            })
            .leftJoinAndSelect(
                'category.indicators',
                'indicators',
                'indicators.categoryId = category.id AND indicators.type = :indicatorType',
                {
                    indicatorType: CategoryTypeEnum.INDICATOR
                }
            )
            .leftJoinAndSelect(
                'indicators.subIndicators',
                'subIndicators',
                'subIndicators.parentId = indicators.id AND subIndicators.type = :subIndicatorType',
                {
                    subIndicatorType: CategoryTypeEnum.SUB_INDICATOR
                }
            )
            .getMany();
    }

    existsSubAndIndicatorPair(indicatorName: string, subIndicatorNames: string[]): Promise<boolean> {
        return this.createQueryBuilder('indicator')
            .where({ type: CategoryTypeEnum.INDICATOR })
            .andWhere({ name: indicatorName })
            .leftJoinAndSelect('indicator.subIndicators', 'subIndicators')
            .andWhere('subIndicators.type = :type', { type: CategoryTypeEnum.SUB_INDICATOR })
            .andWhere('subIndicators.name In (:...subIndicatorNames)', { subIndicatorNames })
            .getExists();
    }

    existsSubAndIndicatorPairById(indicatorId: string, subIndicatorId: string): Promise<boolean> {
        return this.createQueryBuilder('indicator')
            .where({ type: CategoryTypeEnum.INDICATOR })
            .andWhere({ id: indicatorId })
            .leftJoinAndSelect('indicator.subIndicators', 'subIndicators')
            .andWhere('subIndicators.type = :type', { type: CategoryTypeEnum.SUB_INDICATOR })
            .andWhere('subIndicators.id = :subIndicatorId', { subIndicatorId })
            .getExists();
    }

    findIndicatorsByCategoryId(categoryId: string): Promise<CategoryEntity[]> {
        return this.createQueryBuilder('indicator')
            .where({ type: CategoryTypeEnum.INDICATOR })
            .andWhere({ categoryId })
            .innerJoinAndSelect('indicator.subIndicators', 'subIndicators')
            .andWhere('subIndicators.type = :type', { type: CategoryTypeEnum.SUB_INDICATOR })
            .getMany();
    }
}
