import { QueryRunner } from 'typeorm';
import { BaseMigration } from '@diginexhk/typeorm-helper';
import { RoleAttributeTypeEnum } from '~role-attributes/enums/role-attribute-type.enum';
import format from 'pg-format';
import { LOCATION } from '~role-attributes/constants/location.constant';
import fs from 'fs';
import path from 'path';
import { env } from '~config/env.config';
import { RMI_CID_XML_PATH } from '~http-client/constants/rmi-cid-xml-path.constant';
import { parseStringPromise } from 'xml2js';
import { CELL_XML_DATA } from '~http-client/open-apparel-registry/constants/cell-xml-data.constant';

export class AddFullAddressOfCid1719989895842 extends BaseMigration {
    async run(queryRunner: QueryRunner) {
        const provinceCidAttribute = await this.getCidAttribute(queryRunner, LOCATION.STATE_PROVINCE_REGION);
        const roleAttributes = await this.getRoleAttributes(queryRunner, provinceCidAttribute[0].id);
        for (const roleAttribute of roleAttributes) {
            await this.insertRoleAttribute(queryRunner, [
                roleAttribute.roleId,
                provinceCidAttribute[0].id,
                RoleAttributeTypeEnum.RMI_CID,
                Number(roleAttribute.order) + 1,
                true
            ]);
        }

        const facilityIdCidAttribute = await this.getCidAttribute(queryRunner, 'Facility ID');
        const facilityRoleAttributes = await this.getFacilityRoleAttributes(queryRunner, provinceCidAttribute[0].id);
        for (const facilityRoleAttribute of facilityRoleAttributes) {
            const facilityId = facilityRoleAttribute.options.find(
                (item) => item.attributeId === facilityIdCidAttribute[0].id
            ).value;
            const fullAddressCid = await this.getFullAddressRmiCid(facilityId);
            const addressOption = {
                value: fullAddressCid ?? null,
                attributeId: provinceCidAttribute[0].id,
                roleAttributeType: RoleAttributeTypeEnum.RMI_CID
            };
            facilityRoleAttribute.options.push(addressOption);
            await this.updateFacilityRoleAttribute(
                queryRunner,
                facilityRoleAttribute.id,
                facilityRoleAttribute.options
            );
        }
    }

    async rollback(queryRunner: QueryRunner) {}

    private async getCidAttribute(queryRunner: QueryRunner, name: string) {
        return queryRunner.query(
            `
            SELECT "id"
            FROM "Attribute"
            WHERE "type" = $1
            AND name = $2;
        `,
            [RoleAttributeTypeEnum.RMI_CID, name]
        );
    }

    private async getRoleAttributes(queryRunner: QueryRunner, attributeId: string) {
        return queryRunner.query(
            `
            SELECT "roleId", MAX("order") AS order
            FROM "RoleAttribute"
            WHERE "roleAttributeType" = $1
            AND "roleId" NOT IN (
                SELECT "roleId"
                FROM "RoleAttribute"
                WHERE "attributeId" = $2
            )
            GROUP BY "roleId";
        `,
            [RoleAttributeTypeEnum.RMI_CID, attributeId]
        );
    }

    private async getFacilityRoleAttributes(queryRunner: QueryRunner, provinceCidAttribute: string) {
        return queryRunner.query(
            `
            SELECT *
            FROM "FacilityRoleAttribute"
            WHERE "roleAttributeType" = $1
            AND NOT options @> '[{"attributeId": "${provinceCidAttribute}"}]';
        `,
            [RoleAttributeTypeEnum.RMI_CID]
        );
    }

    private async insertRoleAttribute(queryRunner: QueryRunner, insertData: any[]): Promise<void> {
        const sql = format(
            'INSERT INTO "RoleAttribute" ("roleId", "attributeId", "roleAttributeType", "order", "isOptional") VALUES (%L)',
            insertData
        );
        await queryRunner.query(sql);
    }

    private async updateFacilityRoleAttribute(queryRunner: QueryRunner, id: string, updateOptions: any[]) {
        await queryRunner.query(
            `
            UPDATE "FacilityRoleAttribute"
            SET "options" = $1
            WHERE "id" = $2;
        `,
            [JSON.stringify(updateOptions), id]
        );
    }

    private async getFullAddressRmiCid(facilityId: string) {
        const xmlData = fs.readFileSync(path.join(env.ROOT_PATH, RMI_CID_XML_PATH), 'utf8');
        return await parseStringPromise(xmlData)
            .then((res) => {
                return this.convertRmiCidXml(facilityId, res);
            })
            .catch((err) => {
                console.error(err);
            });
    }

    private async convertRmiCidXml(facilityId: string, data: any) {
        const convertXmlData = JSON.parse(JSON.stringify(data));
        const rowXmlData = convertXmlData.Workbook.Worksheet[0].Table[0].Row;
        const cellXmlData = rowXmlData.map((item) => item.Cell);
        for (const cell of cellXmlData) {
            if (facilityId === cell[CELL_XML_DATA.CELL_1].Data[0]._) {
                return cell[CELL_XML_DATA.CELL_4].Data[0]._;
            }
        }
    }
}
