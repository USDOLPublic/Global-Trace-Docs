import { Column, DeleteDateColumn, Entity, Index, JoinColumn, ManyToOne, OneToMany, OneToOne } from 'typeorm';
import { BaseEntity } from '~core/entities/base.entity';
import { TimestampTransformer } from '~core/transformers/timestamp.transformer';
import { FacilityEntity } from '~facilities/entities/facility.entity';
import { CategoryEntity } from '~categories/entities/category.entity';
import { CapStatusEnum } from '~corrective-action-plans/enums/cap-status.enum';
import { CapCommentEntity } from './cap-comment.entity';
import { RiskScoreLevelEnum } from '~self-assessments/enums/risk-score-level.enum';
import { FileUploadResponse } from '~uploads/types/upload-file-response.type';
import { UserEntity } from '~users/entities/user.entity';
import { RequestCapExtensionEntity } from './request-extension.entity';
import { convertToFilesUploadUrl } from '~core/helpers/file.helper';
import { TransformTo } from '@diginexhk/typeorm-helper';

@Entity('Cap')
export class CapEntity extends BaseEntity {
    @Index()
    @Column()
    facilityId: string;

    @Index()
    @Column()
    subIndicatorId: string;

    @Column({ nullable: true })
    action: string;

    @Column({
        type: 'timestamp',
        transformer: new TimestampTransformer(),
        nullable: true
    })
    targetCompletionAt: Date | number;

    @Column({ type: 'jsonb' })
    @TransformTo(convertToFilesUploadUrl)
    files: FileUploadResponse[];

    @Column({ default: CapStatusEnum.NEW })
    status: CapStatusEnum;

    @Column({ default: RiskScoreLevelEnum.NO_WEIGHT })
    riskScoreLevel: RiskScoreLevelEnum;

    @Column({ nullable: true })
    @Index()
    parentId: string | null;

    @Column({ nullable: true })
    @Index()
    creatorId: string;

    @Column({ nullable: true })
    @Index()
    createdFacilityId: string;

    @DeleteDateColumn({
        type: 'timestamp',
        transformer: new TimestampTransformer()
    })
    deletedAt: Date;

    isOwner?: boolean;

    @ManyToOne(() => FacilityEntity)
    @JoinColumn({ name: 'facilityId' })
    facility: FacilityEntity;

    @OneToOne(() => CategoryEntity)
    @JoinColumn({ name: 'subIndicatorId' })
    subIndicator: CategoryEntity;

    @OneToMany(() => CapCommentEntity, (comment) => comment.cap)
    comments: CapCommentEntity[];

    @ManyToOne(() => UserEntity)
    @JoinColumn({ name: 'creatorId' })
    creator: UserEntity;

    @ManyToOne(() => CapEntity)
    @JoinColumn({ name: 'parentId' })
    parentCap: CapEntity;

    @ManyToOne(() => FacilityEntity)
    @JoinColumn({ name: 'createdFacilityId' })
    createdFacility: FacilityEntity;

    @OneToMany(() => RequestCapExtensionEntity, (requestExtension) => requestExtension.cap)
    requestExtensions: RequestCapExtensionEntity[];
}
