import { Injectable } from '@nestjs/common';
import { Dictionary, difference, groupBy, orderBy, pick, reverse, union } from 'lodash';
import { FacilityEntity } from '~facilities/entities/facility.entity';
import { getFacilityTypeName } from '~facilities/helpers/get-facility-type-name.helper';
import { FacilityRepository } from '~facilities/repositories/facility.repository';
import { Line } from '~facilities/types/line.type';
import { SupplierMapElementType } from '~facilities/types/supplier-map-element.type';
import { TracingSupplierMapsType } from '~facilities/types/tracing-supplier-maps.type';
import { TypeMapElementType } from '~facilities/types/type-map-element.type';
import { SupplyChainService } from '~supply-chains/services/supply-chain.service';
import { UserEntity } from '~users/entities/user.entity';

@Injectable()
export class SupplierMappingService {
    constructor(
        private facilityRepo: FacilityRepository,
        private supplyChainService: SupplyChainService
    ) {}

    async getMappingList(user: UserEntity): Promise<TypeMapElementType[][]> {
        const allSuppliers = await this.facilityRepo.findBrandSuppliers(user.currentFacility);
        const allSuppliersMap = new Map(
            allSuppliers.map((supplier) => {
                supplier.users = orderBy(supplier.users, 'createdAt');
                return [supplier.id, supplier];
            })
        );

        const tracedSupplierMap = new Map();
        const rolesOrder = await this.supplyChainService.getRolesOrder();
        const brandTierIds = await this.supplyChainService.getBrandTierIds(user.role);

        return this.traceFromSuppliers({ allSuppliersMap, tracedSupplierMap }, rolesOrder, brandTierIds);
    }

    private traceFromSuppliers(
        maps: TracingSupplierMapsType,
        rolesOrder: { [key: string]: number },
        brandTierIds: string[]
    ): TypeMapElementType[][] {
        const ids = Array.from(maps.allSuppliersMap.keys());

        const result: TypeMapElementType[][] = [];
        while (true) {
            const remainingIds = ids.filter((id) => !maps.tracedSupplierMap.has(id));
            if (!remainingIds.length) {
                break;
            }

            const supplier = maps.allSuppliersMap.get(remainingIds[0]);
            const mapElementType = this.traceGroup(supplier, maps, rolesOrder, brandTierIds);
            mapElementType && result.push(mapElementType);
        }
        return result;
    }

    private traceGroup(
        supplier: FacilityEntity,
        maps: TracingSupplierMapsType,
        rolesOrder: { [key: string]: number },
        brandTierIds: string[]
    ): TypeMapElementType[] {
        const suppliers = this.traceGroupSuppliers(supplier, maps).filter((item) => brandTierIds.includes(item.typeId));
        if (suppliers.length) {
            const lines = this.getLines(suppliers);
            const groupedLineByTargetId = groupBy(lines, 'targetId');

            const groupSuppliers = groupBy(suppliers, ({ type }) => type.id);
            const groups = this.mapTraceGroup(groupSuppliers, rolesOrder, groupedLineByTargetId);

            const appearedSupplierIds = [];
            for (const group of reverse([...groups])) {
                for (const supplier of group.suppliers) {
                    supplier.targets = difference(supplier.targets, appearedSupplierIds);
                }
                appearedSupplierIds.push(...group.suppliers.map((supplier) => supplier.id));
            }

            return groups;
        }
    }

    private mapTraceGroup(
        groupSuppliers: Dictionary<FacilityEntity[]>,
        rolesOrder: { [key: string]: number },
        groupedLineByTargetId: Dictionary<Line[]>
    ): TypeMapElementType[] {
        const groupSuppliersWithRoleId = Object.entries(groupSuppliers);
        return groupSuppliersWithRoleId
            .sort((a, b) => rolesOrder[a[0]] - rolesOrder[b[0]])
            .map(([roleId, groupSupplier], index) => {
                return {
                    roleId,
                    type: groupSupplier[0].type.name,
                    suppliers: groupSupplier.map((facility) => {
                        let afterSupplierIds: string[] = [];

                        if (index < groupSuppliersWithRoleId.length - 1) {
                            const [, afterSuppliers] = groupSuppliersWithRoleId[index + 1];
                            afterSupplierIds = afterSuppliers.map(({ id }) => id);
                        }

                        const targets = union(
                            (groupedLineByTargetId[facility.id] || []).map(({ id }) => id),
                            facility.facilityPartners.map(({ partnerId }) => partnerId)
                        );

                        return {
                            ...pick(facility, [
                                'id',
                                'name',
                                'selfAssessment',
                                'users',
                                'riskLevel',
                                'typeId',
                                'businessRegisterNumber',
                                'oarId',
                                'facilityPartners',
                                'riskData'
                            ]),
                            label: getFacilityTypeName(facility.typeName, facility.type),
                            targets: targets.filter((id) => !afterSupplierIds.includes(id)),
                            type: facility.typeName,
                            icon: facility.type.icon
                        } as SupplierMapElementType;
                    })
                };
            });
    }

    private getLines(suppliers: FacilityEntity[]): Line[] {
        return suppliers.reduce((acc, { id, facilityPartners }) => {
            if (facilityPartners.length) {
                acc.push(...facilityPartners.map(({ partnerId: targetId }) => ({ id, targetId })));
            }
            return acc;
        }, []);
    }

    private traceGroupSuppliers(supplier: FacilityEntity, maps: TracingSupplierMapsType): FacilityEntity[] {
        const supplierMap = new Map();

        this.addSuppliers(supplier, supplierMap, maps);

        return Array.from(supplierMap.values());
    }

    private addSuppliers(
        supplier: FacilityEntity,
        supplierMap: Map<string, FacilityEntity>,
        maps: TracingSupplierMapsType
    ) {
        if (!supplier || supplierMap.has(supplier.id)) {
            return;
        }

        supplierMap.set(supplier.id, supplier);

        if (!maps.tracedSupplierMap.has(supplier.id)) {
            maps.tracedSupplierMap.set(supplier.id, supplier);
        }

        for (const partner of supplier.facilityPartners) {
            this.addSuppliers(maps.allSuppliersMap.get(partner.partnerId), supplierMap, maps);
        }

        for (const partner of supplier.partnerFacilities) {
            this.addSuppliers(maps.allSuppliersMap.get(partner.facilityId), supplierMap, maps);
        }
    }
}
