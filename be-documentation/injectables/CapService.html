import { PaginationParams, SortParams } from '@diginexhk/nestjs-base-decorator';
import { TransactionService } from '@diginexhk/nestjs-transaction';
import { BadRequestException, Inject, Injectable, forwardRef } from '@nestjs/common';
import { PaginationCollection } from '@diginexhk/typeorm-helper';
import { CapEntity } from '~corrective-action-plans/entities/cap.entity';
import { CapStatusEnum } from '~corrective-action-plans/enums/cap-status.enum';
import { AssignCapDto } from '~corrective-action-plans/http/dto/assign-cap.dto';
import { UpdateCapDto } from '~corrective-action-plans/http/dto/update-cap.dto';
import { GetListCapQuery } from '~corrective-action-plans/queries/get-list-cap.query';
import { CapRepository } from '~corrective-action-plans/repositories/cap.repository';
import { FacilityQueueService } from '~facilities/services/facility-queue.service';
import { FacilityService } from '~facilities/services/facility.service';
import { CategoryRiskDetailType } from '~risk-assessments/types/category-risk.type';
import { FacilityRiskType } from '~risk-assessments/types/facility-risk.type';
import { RoleTypeEnum } from '~role-permissions/enums/role-type.enum';
import { RiskScoreLevelEnum } from '~self-assessments/enums/risk-score-level.enum';
import { UserEntity } from '~users/entities/user.entity';
import { UploadService } from '~uploads/services/upload.service';
import { concat, isEmpty, isEqual, omit, orderBy } from 'lodash';
import { generatePaginateCollection } from '~core/helpers/object-helper';
import { FindManyOptions, IsNull, Not } from 'typeorm';
import { RequestExtensionCapDto } from '~corrective-action-plans/http/dto/request-extension-cap.dto.';
import moment from 'moment';
import { RequestExtensionRepository } from '~corrective-action-plans/repositories/request-extension.repository';
import { RequestExtensionStatusEnum } from '~corrective-action-plans/enums/request-extension-status.enum';
import { requestCapExtension } from '~corrective-action-plans/helpers/request-extension.helper';
import { ValidateAssignCapDto } from '~corrective-action-plans/http/dto/validate-assign-cap.dto';

@Injectable()
export class CapService extends TransactionService {
    constructor(
        private capRepository: CapRepository,
        private facilityQueueService: FacilityQueueService,
        @Inject(forwardRef(() => FacilityService)) private facilityService: FacilityService,
        private uploadService: UploadService,
        private requestExtensionRepository: RequestExtensionRepository
    ) {
        super();
    }

    exists(options: FindManyOptions<CapEntity>): Promise<boolean> {
        return this.capRepository.exists(options);
    }

    async assignCap(
        creator: UserEntity,
        facilityId: string,
        data: AssignCapDto,
        files?: Array<Express.Multer.File>
    ): Promise<CapEntity> {
        if (facilityId === creator.currentFacility.id && creator.role.type === RoleTypeEnum.PRODUCT) {
            throw new BadRequestException({ translate: 'error.cannot_assign_to_same_facility' });
        }

        const hasInProgressCap = await this.exists({
            where: {
                subIndicatorId: data.subIndicatorId,
                facilityId,
                status: Not(CapStatusEnum.RESOLVED),
                createdFacilityId: Not(IsNull())
            }
        });

        if (hasInProgressCap) {
            throw new BadRequestException({ translate: 'error.cannot_assign_another_cap' });
        }

        const uploadFiles = await this.uploadService.uploadFiles(files);

        const assignCap = await this.capRepository.createOne({
            ...data,
            creatorId: creator.id,
            createdFacilityId: creator.currentFacility.id,
            facilityId,
            status: CapStatusEnum.NEW,
            riskScoreLevel: RiskScoreLevelEnum.NO_WEIGHT,
            files: uploadFiles
        });
        await this.facilityQueueService.addFacilityRiskCalculation(facilityId);

        return assignCap;
    }

    async validateAssignCap(facilityId: string, { subIndicatorId }: ValidateAssignCapDto): Promise<void> {
        const hasInProgressCap = await this.exists({
            where: {
                subIndicatorId,
                facilityId,
                status: Not(CapStatusEnum.RESOLVED),
                createdFacilityId: Not(IsNull())
            }
        });

        if (hasInProgressCap) {
            throw new BadRequestException({ translate: 'error.cannot_assign_another_cap' });
        }
    }

    async updateCap(
        currentUser: UserEntity,
        facilityId: string,
        id: string,
        data: UpdateCapDto,
        files?: Array<Express.Multer.File>
    ): Promise<CapEntity> {
        const cap = await this.capRepository.findOneOrFail({
            where: { id, facilityId }
        });

        await this.validateUpdateCap(currentUser.role.type, cap, data);
        if (data?.riskScoreLevel === RiskScoreLevelEnum.NO_WEIGHT) {
            throw new BadRequestException({ translate: 'validation.risk_score_should_not_be_no_weight' });
        }
        const payload: Partial<CapEntity> = omit(data, ['files', 'deleteBlobNames']);
        payload.files = cap.files?.filter((file) => !data.deleteBlobNames?.includes(file.blobName));

        if (!isEmpty(files)) {
            const uploadedFiles = await this.uploadService.uploadFiles(files);
            payload.files = concat(payload.files, uploadedFiles);
        }

        await this.capRepository.update({ id, facilityId }, payload);
        await this.facilityQueueService.addFacilityRiskCalculation(facilityId);

        if (data?.riskScoreLevel) {
            const capUpdate = await this.capRepository.findById(cap.id);
            const parentId = capUpdate.id;

            await this.capRepository.createOne({
                ...omit(capUpdate, ['id', 'createdAt', 'updatedAt']),
                parentId
            });
        }

        return cap;
    }

    async requestExtension(
        currentUser: UserEntity,
        facilityId: string,
        id: string,
        data: RequestExtensionCapDto
    ): Promise<void> {
        const cap = await this.capRepository.findOneOrFail({
            where: { id, facilityId }
        });
        const currentDate = moment().unix();

        if (
            cap.status !== CapStatusEnum.OVERDUE ||
            (cap.targetCompletionAt as number) >= currentDate ||
            currentUser.role.type !== RoleTypeEnum.PRODUCT
        ) {
            throw new BadRequestException({ translate: 'validation.can_not_request_extension_target_completion_at' });
        }

        const requestExtension = await this.requestExtensionRepository.findOne({
            where: { capId: cap.id, status: RequestExtensionStatusEnum.PENDING }
        });
        if (requestExtension) {
            throw new BadRequestException({ translate: 'validation.you_need_to_decline_the_current_request' });
        }
        await this.requestExtensionRepository.createOne({
            capId: id,
            timePeriod: data.timePeriod,
            targetCompletionAt: requestCapExtension(data),
            status: RequestExtensionStatusEnum.PENDING
        });
    }

    async approveExtensionRequest(facilityId: string, id: string): Promise<void> {
        const cap = await this.capRepository.findOneOrFail({
            where: { id, facilityId }
        });
        const currentDate = moment().unix();
        const requestExtension = await this.requestExtensionRepository.findOneOrFail({
            where: { capId: cap.id, status: RequestExtensionStatusEnum.PENDING }
        });

        if ((requestExtension.targetCompletionAt as number) <= currentDate) {
            throw new BadRequestException({ translate: 'validation.the_request_extension_has_already_expired' });
        }

        await this.requestExtensionRepository.update(requestExtension.id, {
            status: RequestExtensionStatusEnum.APPROVED
        });
        await this.capRepository.update(cap.id, {
            status: CapStatusEnum.IN_PROGRESS,
            targetCompletionAt: requestExtension.targetCompletionAt
        });

        await this.facilityQueueService.addFacilityRiskCalculation(facilityId);
    }

    async declineExtensionRequest(facilityId: string, id: string): Promise<void> {
        const cap = await this.capRepository.findOneOrFail({
            where: { id, facilityId }
        });
        const requestExtension = await this.requestExtensionRepository.findOneOrFail({
            where: { capId: cap.id, status: RequestExtensionStatusEnum.PENDING }
        });
        await this.requestExtensionRepository.update(requestExtension.id, {
            status: RequestExtensionStatusEnum.DECLINE
        });
    }

    async updateCapStatusOnTargetDateElapsed(): Promise<void> {
        await this.capRepository.updateCapStatusOnTargetDateElapsed();
    }

    async declineExpiredRequests(): Promise<void> {
        await this.requestExtensionRepository.declineExpiredRequests();
    }

    async getDetailCap(user: UserEntity, facilityId: string, id: string): Promise<CategoryRiskDetailType> {
        const facility = await this.facilityService.getDetailSupplierById(facilityId);

        return this.getRiskDataDetail(facility.riskData, id, user);
    }

    async getRiskDataDetail(
        riskData: FacilityRiskType,
        capId: string,
        user: UserEntity
    ): Promise<CategoryRiskDetailType> {
        const cap = await this.capRepository.findOneOrFail({
            relations: [
                'subIndicator',
                'comments',
                'comments.user',
                'parentCap',
                'parentCap.comments',
                'parentCap.comments.user',
                'requestExtensions',
                'facility',
                'createdFacility'
            ],
            where: { id: capId }
        });
        cap.isOwner = cap.createdFacilityId === user.currentFacility.id;
        cap.comments = cap.comments.filter((item) => !isEmpty(item.user));
        cap.comments = orderBy(cap.comments, 'createdAt', 'desc');
        if (cap?.parentCap) {
            cap.parentCap.comments = cap.parentCap.comments.filter((item) => !isEmpty(item.user));
            cap.comments = orderBy(cap.parentCap.comments, 'createdAt', 'desc');
        }
        const { indicatorRiskData, ...categoryRiskDetail } = riskData.data.find(
            (categoryRisk) => categoryRisk.category.id === cap.subIndicator.categoryId
        );
        const { subIndicatorRiskData = [], ...indicatorRiskDataDetail } =
            indicatorRiskData.find((indicatorRisk) => indicatorRisk.indicator.id === cap.subIndicator.parentId) || {};
        const filterSubIndicatorRiskData = subIndicatorRiskData.find(
            ({ subIndicator }) => subIndicator.id === cap.subIndicator.id
        );

        return {
            ...categoryRiskDetail,
            indicatorRiskData: {
                ...indicatorRiskDataDetail,
                subIndicatorRiskData: {
                    ...filterSubIndicatorRiskData,
                    cap
                }
            }
        } as CategoryRiskDetailType;
    }

    async getCapBySubIndicatorIds(facilityId: string, subIndicatorIds: string[]): Promise<CapEntity[]> {
        return this.capRepository.getCapBySubIndicatorIds(facilityId, subIndicatorIds);
    }

    private validateUpdateCap(role: RoleTypeEnum, cap: CapEntity, dto: UpdateCapDto): void {
        const currentStatus = cap.status;
        if (dto.targetCompletionAt && role === RoleTypeEnum.BRAND && currentStatus !== CapStatusEnum.UNDER_REVIEW) {
            throw new BadRequestException({ translate: 'validation.can_not_change_target_completion_at' });
        }

        if (
            (role === RoleTypeEnum.BRAND &&
                (![CapStatusEnum.IN_PROGRESS, CapStatusEnum.UNDER_REVIEW].includes(currentStatus) ||
                    (currentStatus === CapStatusEnum.UNDER_REVIEW &&
                        ![CapStatusEnum.IN_PROGRESS, CapStatusEnum.RESOLVED].includes(dto.status)))) ||
            (role === RoleTypeEnum.PRODUCT &&
                (![CapStatusEnum.NEW, CapStatusEnum.IN_PROGRESS].includes(currentStatus) ||
                    !this.isEqualCapValue(cap, dto) ||
                    (currentStatus === CapStatusEnum.NEW && dto.status !== CapStatusEnum.IN_PROGRESS) ||
                    (currentStatus === CapStatusEnum.IN_PROGRESS && dto.status !== CapStatusEnum.UNDER_REVIEW)))
        ) {
            throw new BadRequestException({ translate: 'validation.can_not_change_status_this_role' });
        }
    }

    async list(
        user: UserEntity,
        pagination: PaginationParams,
        sortParams: SortParams,
        sortWithNewStatus: boolean
    ): Promise<PaginationCollection<CapEntity>> {
        const [caps, count] = await this.capRepository.findAndCount(
            new GetListCapQuery(user, sortParams, sortWithNewStatus, pagination)
        );
        await caps.loadRelation(['createdFacility', 'requestExtensions']);

        return generatePaginateCollection(pagination, count, caps);
    }

    private isEqualCapValue(cap: CapEntity, dto: UpdateCapDto): boolean {
        const currentCapInput = {
            subIndicatorId: cap.subIndicatorId,
            action: cap.action,
            targetCompletionAt: cap.targetCompletionAt
        };
        const currentCapDto = {
            subIndicatorId: dto.subIndicatorId,
            action: dto.action,
            targetCompletionAt: dto.targetCompletionAt
        };

        return !!isEqual(currentCapInput, currentCapDto);
    }
}
