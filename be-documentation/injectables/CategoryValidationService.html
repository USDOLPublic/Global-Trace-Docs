import { ValidateException, ValidationError } from '@diginexhk/nestjs-exception';
import { Injectable } from '@nestjs/common';
import { groupBy, keyBy, keys } from 'lodash';
import { In } from 'typeorm';
import { CategoryTypeEnum } from '~categories/enums/category-type.enum';
import { CategoryRepository } from '~categories/repositories/category.repository';
import { CreateNestedCategory } from '~category-utils/http/dto/create-nested-categories.dto';
import { createValidationError } from '~core/helpers/create-validation-error.helper';

@Injectable()
export class CategoryValidationService {
    public constructor(private categoryRepo: CategoryRepository) {}

    addValidateNameAlreadyExists(property: string): ValidationError {
        return createValidationError({
            property: `The ${property} name is already exists.`,
            message: 'The $property name is already exists.',
            detail: { property }
        });
    }

    private addValidateDuplicatedName(property: string): ValidationError {
        return createValidationError({
            property: `The ${property} has name is duplicated`,
            message: 'The $property has name is duplicated.',
            detail: { property }
        });
    }

    async validateUniqueSubAndIndicatorPairs(
        categories: CreateNestedCategory[],
        errors: ValidationError[]
    ): Promise<void> {
        const indicators = categories.flatMap(({ indicators }) => indicators);
        const preparedExistedSubAndIndicatorPairs = [];

        for (const { subIndicators, name: indicatorName } of indicators) {
            preparedExistedSubAndIndicatorPairs.push(
                this.categoryRepo.existsSubAndIndicatorPair(
                    indicatorName,
                    subIndicators.map(({ name }) => name)
                )
            );
        }

        const existedSubAndIndicatorPairs = await Promise.all(preparedExistedSubAndIndicatorPairs);

        return this.addValidateUniqueSubAndIndicatorPairs(existedSubAndIndicatorPairs, errors);
    }

    private addValidateUniqueSubAndIndicatorPairs(
        existedSubAndIndicatorPairs: boolean[],
        errors: ValidationError[]
    ): void {
        const hasExistedSubAndIndicatorPairs = existedSubAndIndicatorPairs.some(
            (existedSubAndIndicatorPair) => existedSubAndIndicatorPair
        );

        if (!hasExistedSubAndIndicatorPairs) {
            return;
        }

        errors.push(
            createValidationError({
                property: 'Sub Indicator And Indicator Pair Already Exists',
                message: 'indicator_and_sub_indicator_pair_already_exists'
            })
        );
    }

    validateDuplicatedNames(categoryNames: string[], indicatorNames: string[], errors: ValidationError[]): void {
        const duplicatedCategoryNames = this.getDuplicatedNames(categoryNames);
        const duplicatedIndicatorNames = this.getDuplicatedNames(indicatorNames);

        if (duplicatedCategoryNames.length) {
            errors.push(this.addValidateDuplicatedName('Category'));
        }

        if (duplicatedIndicatorNames.length) {
            errors.push(this.addValidateDuplicatedName('Indicator'));
        }
    }

    private getDuplicatedNames(names: string[]): string[] {
        const grouped = groupBy(names);
        return keys(grouped).filter((key) => grouped[key].length > 1);
    }

    validateDuplicatedSubIndicatorNames(categories: CreateNestedCategory[], errors: ValidationError[]): void {
        const indicators = categories.flatMap(({ indicators }) => indicators);

        for (const indicator of indicators) {
            const subIndicatorNames = indicator.subIndicators.map(({ name }) => name);
            const duplicatedSubIndicatorNames = this.getDuplicatedNames(subIndicatorNames);

            if (!duplicatedSubIndicatorNames.length) {
                continue;
            }

            errors.push(this.addValidateDuplicatedName('Sub Indicator'));

            break;
        }

        if (errors.length) {
            throw new ValidateException(errors);
        }
    }

    async validateIndicatorBelongToOtherCategories(
        indicatorNames: string[],
        categories: CreateNestedCategory[],
        errors: ValidationError[]
    ): Promise<void> {
        const existedIndicators = await this.categoryRepo.find({
            where: { name: In(indicatorNames), type: CategoryTypeEnum.INDICATOR },
            relations: ['category']
        });

        const indicatorAndCategoryPairs = categories.flatMap(({ name: categoryName, indicators }) =>
            indicators.map(({ name: indicatorName }) => ({ categoryName, indicatorName }))
        );
        const mappedIndicatorAndCategoryPairs = keyBy(indicatorAndCategoryPairs, 'categoryName');

        for (const existedIndicator of existedIndicators) {
            if (!indicatorNames.includes(existedIndicator.name)) {
                continue;
            }

            const indicatorAndCategoryPair = mappedIndicatorAndCategoryPairs[existedIndicator.category?.name];
            if (!indicatorAndCategoryPair) {
                errors.push(
                    createValidationError({
                        property: `The same indicator with different category`,
                        message: 'the_same_indicator_with_different_category'
                    })
                );

                break;
            }
        }
    }
}
