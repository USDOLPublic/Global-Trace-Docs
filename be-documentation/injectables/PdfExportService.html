import { HttpService } from '@nestjs/axios';
import { Inject, Injectable, NotFoundException } from '@nestjs/common';
import { Cache } from 'cache-manager';
import { lastValueFrom } from 'rxjs';
import { env } from '~config/env.config';
import { I18nHelper } from '~core/helpers/i18n.helper';
import { makeAToken } from '~core/helpers/string.helper';
import { FacilityService } from '~facilities/services/facility.service';
import { OrderTraceService } from '~order/services/order-trace.service';
import { OrderService } from '~order/services/order.service';
import {
    DEFAULT_TOKEN_PDF,
    DEFAULT_TOKEN_SUPPLIER_DETAIL_PDF
} from '~pdf-exports/constants/default-token-pdf.constant';
import { TTL } from '~pdf-exports/constants/pdf-export.constant';
import { PREVIEW_PDF, SUPPLIER_DETAIL_PREVIEW_PDF } from '~pdf-exports/constants/pdf-preview.constant';
import { SupplierMappingData } from '~pdf-exports/types/supplier-mapping-data.type';
import { UserEntity } from '~users/entities/user.entity';
import { PdfExportTracingObjectService } from './pdf-export-tracing-object.service';
import { FacilityRiskFilterDto } from '~risk-assessments/http/dto/facility-risk-filter.dto';
import { FacilityRiskFilerService } from '~facilities/services/facility-risk-filter.service';
import { HomePageData } from '~pdf-exports/types/homepage-data.type';
import { TracingSupplierType } from '~order/types/tracing-supplier.type';
import { Dictionary, first, groupBy } from 'lodash';
import { FacilityEntity } from '~facilities/entities/facility.entity';
import { RiskItemType } from '~risk-assessments/types/risk-item.type';
import { RiskSourceEnum } from '~risk-assessments/enums/risk-source.enum';
import { CategoryService } from '~categories/services/category.service';
import { RiskAssessmentService } from '~risk-assessments/services/risk-assessment.service';
import { FacilityRiskService } from '~risk-assessments/services/facility-risk.service';
import { CategoryTypeEnum } from '~categories/enums/category-type.enum';
import { ExternalRiskIndicatorGroup } from '~pdf-exports/types/external-risk-indicator-group.type';
import { CommodityRiskService } from '~commodity-risks/services/commodity-risk.service';
import { In } from 'typeorm';
import { CACHE_MANAGER } from '@nestjs/cache-manager';
import { SupplyChainTierService } from '~supply-chains/services/supply-chain-tier.service';
import { FacilityOperationService } from '~facilities/services/facility-operation.service';
import { BusinessDetailService } from '~site-details/services/business-detail.service';

@Injectable()
export class PdfExportService {
    constructor(
        private httpService: HttpService,
        @Inject(CACHE_MANAGER) private cacheManager: Cache,
        private orderService: OrderService,
        private orderTraceService: OrderTraceService,
        private pdfExportTracingObjectService: PdfExportTracingObjectService,
        private facilityService: FacilityService,
        private facilityRiskFilerService: FacilityRiskFilerService,
        private categoryService: CategoryService,
        private riskAssessmentService: RiskAssessmentService,
        private facilityRiskService: FacilityRiskService,
        private commodityRiskService: CommodityRiskService,
        private supplyChainTierService: SupplyChainTierService,
        private facilityOperationService: FacilityOperationService,
        private businessDetailService: BusinessDetailService
    ) {}

    async getPdfData(token: string) {
        const data: string = await this.cacheManager.get(token);
        if (!data) {
            throw new NotFoundException({ translate: 'error.not_found.Token' });
        }

        return JSON.parse(data);
    }

    async generatePdf(orderId: string, timezone: string, user: UserEntity, language?: string): Promise<Buffer> {
        const token = env.APP_ENV === 'development' ? DEFAULT_TOKEN_PDF : makeAToken();
        const [homePage, order, supplierMapping, tracingObjects, tiers] = await Promise.all([
            this.generateHomePageData(user),
            this.orderService.getOrderById(user, orderId),
            this.generateSupplierMappingData(user, orderId),
            this.pdfExportTracingObjectService.generateTracingData(user, orderId),
            this.supplyChainTierService.getTiers(user.role, true)
        ]);

        const countByRiskLevel = this.pdfExportTracingObjectService.countFacilitiesByRiskLevel(tracingObjects);
        const topFourIssues = await this.pdfExportTracingObjectService.getTopFourIssues(tracingObjects);

        const externalRiskIndicators = await this.getExternalRiskIndicators(tracingObjects);
        const tracedSupplierIds = this.getTracedSupplierIds(tracingObjects);
        const dataForMapExternalInfo =
            await this.facilityOperationService.findDataForMapExternalInfo(tracedSupplierIds);

        for (const tracingObject of tracingObjects) {
            if (tracingObject.supplier) {
                I18nHelper.translateFacilityLocation(tracingObject.supplier);
                this.facilityOperationService.mapWithAllExternalInfo(tracingObject.supplier, dataForMapExternalInfo);
            }
        }

        const pdfData = {
            homePage,
            assessment: { order, countByRiskLevel, topFourIssues, externalRiskIndicators },
            supplierMapping,
            tracingObjects,
            language,
            tiers
        };

        await this.cacheManager.set(token, JSON.stringify({ pdfData }), TTL);
        return this.getBufferFileWeb2Pdf(token, timezone, PREVIEW_PDF);
    }

    private getTracedSupplierIds(tracingObjects: TracingSupplierType[]): string[] {
        return tracingObjects.reduce((acc, { supplier }) => {
            if (supplier) {
                acc.push(supplier.id);
            }

            return acc;
        }, []);
    }

    async generateSupplierDetailPdf(
        supplierId: string,
        filters: FacilityRiskFilterDto,
        timezone: string,
        language: string
    ): Promise<Buffer> {
        const token = env.APP_ENV === 'development' ? DEFAULT_TOKEN_SUPPLIER_DETAIL_PDF : makeAToken();
        const supplierDetail = await this.facilityService.findInformationSupplierById(supplierId, filters);
        const filterParams = await this.facilityRiskFilerService.getFilterParams(filters);

        const pdfData = {
            supplierDetail,
            filterParams,
            language
        };

        await this.cacheManager.set(token, JSON.stringify({ pdfData }), TTL);
        return this.getBufferFileWeb2Pdf(token, timezone, SUPPLIER_DETAIL_PREVIEW_PDF);
    }

    private async getBufferFileWeb2Pdf(token: string, timezone: string, pdfPreview: string) {
        try {
            const { data } = await this.convertWeb2Pdf(token, timezone, pdfPreview);
            return data;
        } finally {
            if (env.APP_ENV !== 'development') {
                await this.cacheManager.del(token);
            }
        }
    }

    private async convertWeb2Pdf(token: string, timezone: string, pdfPreview: string, attemptCount: number = 1) {
        try {
            let convertedPageUrl = `${env.WEB_URL}/${pdfPreview}?token=${token}`;

            return await lastValueFrom(
                this.httpService.post(
                    `${env.WEB_2_PDF_API_URL}/pdf/converts`,
                    {
                        url: convertedPageUrl,
                        waitFunction: 'window.isLoadPdfCompleted',
                        timezone
                    },
                    //eslint-disable-next-line @typescript-eslint/naming-convention
                    { headers: { 'api-key': env.WEB_2_PDF_API_KEY }, responseType: 'arraybuffer' }
                )
            );
        } catch (error) {
            console.error(error.response || error);
            if (attemptCount > 3) {
                throw error;
            }

            await this.convertWeb2Pdf(token, timezone, pdfPreview, attemptCount + 1);
        }
    }

    private generateHomePageData(requester: UserEntity): HomePageData {
        return { ...requester.currentFacility, phoneNumber: requester.phoneNumber };
    }

    private async generateSupplierMappingData(user: UserEntity, orderId: string): Promise<SupplierMappingData> {
        const [traceMappingSuppliers, traceResultList, orderSuppliers] = await Promise.all([
            this.orderTraceService.getTraceResultMapping(user, orderId),
            this.orderTraceService.getTraceResultList(user, orderId),
            this.orderTraceService.getSupplierList(user, orderId)
        ]);

        return { traceMappingSuppliers, traceResultList, orderSuppliers };
    }

    async getExternalRiskIndicators(tracingObjects: TracingSupplierType[]): Promise<ExternalRiskIndicatorGroup[]> {
        const suppliers = tracingObjects.reduce((acc, { supplier }) => {
            if (supplier) {
                acc.push(supplier);
            }
            return acc;
        }, []);
        const groupedSuppliers = groupBy(suppliers, 'country.country');
        const [riskMethodology, goodRiskItems, externalRiskIndicatorCategories] = await Promise.all([
            this.riskAssessmentService.getRiskAssessment(),
            this.getProductRiskListingsItems(groupedSuppliers),
            this.categoryService.all({ type: CategoryTypeEnum.EXTERNAL_RISK_INDICATOR })
        ]);
        const groupedRiskItemsByCountry = groupBy(goodRiskItems, ({ country }) => country);
        const externalRiskIndicatorGroup: ExternalRiskIndicatorGroup[] = [];

        for (const [country, suppliers] of Object.entries(groupedSuppliers)) {
            const riskItemsByCountry = groupedRiskItemsByCountry[country];
            const groupedRiskItemsByGood = groupBy(riskItemsByCountry, ({ additionData }) => additionData.good);

            const riskIndicatorGroup = Object.entries(groupedRiskItemsByGood).flatMap(([good, riskItemsByGood]) => {
                const groupedRiskItemsByCategory = groupBy(riskItemsByGood, ({ indicator }) => indicator?.categoryId);
                const riskIndicators = this.facilityRiskService.getAndCalculateExternalRiskIndicators(
                    groupedRiskItemsByCategory,
                    riskMethodology,
                    first(suppliers)
                );
                return {
                    good,
                    riskIndicators: this.facilityService.mapExternalRiskIndicators(
                        riskIndicators,
                        externalRiskIndicatorCategories
                    )
                };
            });

            const externalRiskIndicators = {
                countryRisk: {
                    country: first(suppliers)?.country,
                    risk: await this.facilityRiskService.getCountryRisk(first(suppliers)),
                    category: externalRiskIndicatorCategories.find(({ name }) => name === 'Country based risk')
                },
                riskIndicatorGroup
            };

            externalRiskIndicatorGroup.push(externalRiskIndicators);
        }

        return externalRiskIndicatorGroup;
    }

    private async getProductRiskListingsItems(groupedSuppliers: Dictionary<FacilityEntity[]>): Promise<RiskItemType[]> {
        const promisedGoodRiskItemGroup = [];

        for (const suppliers of Object.values(groupedSuppliers)) {
            let goods: string[] = [];
            for (const supplier of suppliers) {
                if (supplier.type?.isRawMaterialExtractor) {
                    const supplierGoods = await this.businessDetailService.getSelectedGoods(supplier.typeId);
                    if (supplierGoods?.length) {
                        goods = goods.concat(supplierGoods);
                    }
                } else if (supplier.goods?.length) {
                    goods = goods.concat(supplier.goods);
                }
            }

            const countryId = first(suppliers)?.country?.id;
            if (!countryId || !goods.length) {
                continue;
            }

            promisedGoodRiskItemGroup.push(
                this.commodityRiskService.find({
                    where: { countryId, commodity: In(goods) },
                    relations: ['country', 'subIndicator', 'subIndicator.indicator', 'subIndicator.indicator.category']
                })
            );
        }

        const goodRiskItemGroup = await Promise.all(promisedGoodRiskItemGroup);

        return goodRiskItemGroup
            .flatMap((goodRiskItems) =>
                goodRiskItems.map((goodRiskItem) => ({
                    ...goodRiskItem,
                    country: goodRiskItem.country.country,
                    indicator: goodRiskItem.subIndicator?.indicator,
                    source: RiskSourceEnum.PRODUCT_RISK_LISTINGS,
                    additionData: { good: goodRiskItem.commodity }
                }))
            )
            .filter(({ indicator, subIndicator, severity }) => severity && indicator && subIndicator);
    }
}
