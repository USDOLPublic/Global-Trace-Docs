import { TransactionService } from '@diginexhk/nestjs-transaction';
import { Injectable } from '@nestjs/common';
import { FindManyOptions } from 'typeorm';
import { SupplyChainNodeMetadataEntity } from '~supply-chains/entities/supply-chain-node-metadata.entity';
import { SupplyChainNodeEntity } from '~supply-chains/entities/supply-chain-node.entity';
import { SupplyChainNodeMetadataRepository } from '~supply-chains/repositories/supply-chain-node-metadata.repository';
import { SupplyChainNodeRepository } from '~supply-chains/repositories/supply-chain-node.repository';
import { SupplyPositionType } from '~supply-chains/types/supply-position.type';

@Injectable()
export class SupplyChainMetaDataService extends TransactionService {
    constructor(
        private supplyChainNodeMetadataRepo: SupplyChainNodeMetadataRepository,
        private supplyChainNodeRepo: SupplyChainNodeRepository
    ) {
        super();
    }

    find(options: FindManyOptions<SupplyChainNodeMetadataEntity>) {
        return this.supplyChainNodeMetadataRepo.find(options);
    }

    findByProductDefinitionId(productDefinitionId: string, roleId?: string): Promise<SupplyChainNodeEntity[]> {
        return this.supplyChainNodeRepo.findByProductDefinitionId(productDefinitionId, roleId);
    }

    existByProductDefinitionId(productDefinitionId: string, roleId: string, chainId?: string): Promise<boolean> {
        return this.supplyChainNodeRepo.existByProductDefinitionId(productDefinitionId, roleId, chainId);
    }

    existByProductDefinitionIds(productDefinitionIds: string[], roleId: string, chainId?: string): Promise<boolean> {
        return this.supplyChainNodeRepo.existByProductDefinitionIds(productDefinitionIds, roleId, chainId);
    }

    async upsertNodeMetadata(roleId: string, position: SupplyPositionType, chainId: string): Promise<void> {
        const metadata = await this.supplyChainNodeMetadataRepo.findOneBy({ roleId });
        if (!metadata) {
            const partialEntity: Partial<SupplyChainNodeMetadataEntity> = {
                position,
                roleId,
                chainId
            };
            await this.supplyChainNodeMetadataRepo.createOne(partialEntity);
        }
    }
}
