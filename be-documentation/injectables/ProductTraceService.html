import { Injectable } from '@nestjs/common';
import { flatten, uniq } from 'lodash';
import { TransactionEntity } from '~events/entities/transaction.entity';
import { TransformationEntity } from '~events/entities/transformation.entity';
import { TransactionHistoryService } from '~events/services/transaction/transaction-history.service';
import { TransformationHistoryService } from '~events/services/transformation/transformation-history.service';
import { SupplyChainService } from '~supply-chains/services/supply-chain.service';
import { FacilityEntity } from '~facilities/entities/facility.entity';
import { ProductEntity } from '~products/entities/product.entity';
import { TracingProductType } from '~order/types/tracing-product.type';
import { Queue } from 'typescript-collections';

@Injectable()
export class ProductTraceService {
    constructor(
        private transactionHistoryService: TransactionHistoryService,
        private transformationHistoryService: TransformationHistoryService,
        private supplyChainService: SupplyChainService
    ) {}

    async getCountriesFromProduct(supplier: FacilityEntity, product: ProductEntity): Promise<string[]> {
        const transactionIds: string[] = [];
        const tracingObject: TracingProductType = {
            supplier,
            productIds: [product.id],
            products: [product],
            transactionId: null
        };
        let tracingObjects: TracingProductType[] = [tracingObject];

        const queue: Queue<TracingProductType> = new Queue();
        queue.enqueue(tracingObject);

        while (!queue.isEmpty()) {
            const tracingObject: TracingProductType = queue.dequeue();
            const traceResults: TracingProductType[] = await this.trace(tracingObject);

            if (!traceResults?.length) {
                continue;
            }

            for (const item of traceResults) {
                if (item.transactionId) {
                    if (!transactionIds.includes(item.transactionId)) {
                        transactionIds.push(item.transactionId);
                        tracingObjects.push(item);
                        queue.enqueue(item);
                    }
                } else {
                    tracingObjects.push(item);
                    queue.enqueue(item);
                }
            }
        }

        return this.extractCountries(tracingObjects);
    }

    private async extractCountries(tracingObjects: TracingProductType[]): Promise<string[]> {
        const rawMaterialExtractorRoles = await this.supplyChainService.getRawMaterialExtractorRoles();
        const rawMaterialExtractorRoleIds = rawMaterialExtractorRoles.map(({ id }) => id);

        const countryIds: string[] = [];
        for (const item of tracingObjects) {
            if (rawMaterialExtractorRoleIds.includes(item.supplier.typeId)) {
                countryIds.push(item.supplier.countryId);
            }

            for (const product of item.products) {
                if (!product.isManualAdded) {
                    continue;
                }

                for (const additionalAttribute of product.additionalAttributes) {
                    if (additionalAttribute.value?.countryId) {
                        countryIds.push(additionalAttribute.value.countryId);
                    }
                }
            }
        }

        return uniq(countryIds.filter((countryId) => countryId));
    }

    private async trace(tracingObject: TracingProductType): Promise<TracingProductType[]> {
        if (!tracingObject.supplier) {
            return [];
        }

        const transformations = await this.transformationHistoryService.getTransformationFromOutput(
            tracingObject.supplier,
            tracingObject.productIds
        );

        if (!transformations.length) {
            return [];
        }

        const isOutputOfSecondNode = await this.supplyChainService.isOutputOfSecondNode(
            tracingObject.products[0].productDefinitionId
        );
        if (isOutputOfSecondNode) {
            return this.traceManuallyPurchasedProduct(transformations, tracingObject);
        }

        const transformationIds = transformations.map(({ id }) => id);
        const inputProducts = await this.transformationHistoryService.getInputProducts(transformationIds);

        if (!inputProducts.length) {
            return [];
        }

        const inputProductIds = inputProducts.map(({ entityId }) => entityId);
        const transactions = await this.transactionHistoryService.findTransactionsTransactProducts(
            tracingObject.supplier,
            inputProductIds
        );

        if (!transactions.length) {
            return [];
        }

        return this.convertTransactionToTracingObject(transactions);
    }

    private async traceManuallyPurchasedProduct(
        transformations: TransformationEntity[],
        fromTracingObject: TracingProductType
    ): Promise<TracingProductType[]> {
        const result = await Promise.all(
            transformations.map(async (transformation) => {
                const transactions: TransactionEntity[] =
                    await this.transactionHistoryService.findManuallyPurchasedProductFromStartDateOfSeason(
                        fromTracingObject.supplier,
                        transformation.createdAt
                    );

                return this.convertTransactionToTracingObject(transactions);
            })
        );
        return flatten(result);
    }

    private convertTransactionToTracingObject(transactions: TransactionEntity[]): TracingProductType[] {
        const objects: TracingProductType[] = [];
        for (const transaction of transactions) {
            if (!transaction.fromFacility) {
                continue;
            }

            const products = transaction.transactionItems.map(({ product }) => product);
            const productIds = transaction.transactionItems.map(({ entityId }) => entityId);

            objects.push({
                supplier: transaction.fromFacility,
                productIds,
                products,
                transactionId: transaction.id
            });
        }

        return objects;
    }
}
