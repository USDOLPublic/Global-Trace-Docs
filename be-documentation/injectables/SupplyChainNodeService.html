import { TransactionService } from '@diginexhk/nestjs-transaction';
import { BadRequestException, Injectable } from '@nestjs/common';
import { concat, difference, groupBy, isEmpty, isNil, keyBy, uniq } from 'lodash';
import { CalculateFieldService } from '~calculate-field/services/calculate-field.service';
import { RoleTypeEnum } from '~role-permissions/enums/role-type.enum';
import { RoleService } from '~role-permissions/services/role.service';
import { SupplyChainNodeEntity } from '~supply-chains/entities/supply-chain-node.entity';
import { DeleteSupplyChainNodesDto } from '~supply-chains/http/dto/delete-supply-chain-nodes.dto';
import { UpdateSupplyChainNodeMetadataDto } from '~supply-chains/http/dto/update-supply-chain-node-metadata.dto';
import { SupplyChainNodeMetadataRepository } from '~supply-chains/repositories/supply-chain-node-metadata.repository';
import { SupplyChainNodeRepository } from '~supply-chains/repositories/supply-chain-node.repository';
import { GetNodeImpactsParams, NodeImpactsParams } from '~supply-chains/types/get-node-impacts-params.dto';
import { NodeImpacts } from '~supply-chains/types/node-impacts.dto';

@Injectable()
export class SupplyChainNodeService extends TransactionService {
    constructor(
        private supplyChainNodeRepo: SupplyChainNodeRepository,
        private roleService: RoleService,
        private supplyChainNodeMetadataRepo: SupplyChainNodeMetadataRepository,
        private calculateFieldService: CalculateFieldService
    ) {
        super();
    }

    async checkForCycle(roleId: string, fromRoleId: string, chainId: string): Promise<void> {
        const traverseAncestors = async (currentRoleId: string): Promise<void> => {
            if (currentRoleId === roleId) {
                throw new BadRequestException({ translate: 'validation.cycle_detected' });
            }

            const parentNodes = await this.supplyChainNodeRepo.find({
                where: { roleId: currentRoleId, chainId }
            });

            for (const parentNode of parentNodes) {
                if (parentNode?.fromRoleId) {
                    await traverseAncestors(parentNode.fromRoleId);
                }
            }
        };

        await traverseAncestors(fromRoleId);
    }

    async validateRoleForSupplyChainNodeAddition(roleId: string, fromRoleId: string | null): Promise<void> {
        const role = await this.roleService.findRoleById(roleId);
        const node = await this.supplyChainNodeRepo.findOneBy({ roleId });

        if (role.type !== RoleTypeEnum.PRODUCT) {
            throw new BadRequestException({ translate: 'validation.supplier_not_exist_in_supply_chain_map' });
        }
        if (isNil(fromRoleId) && !role.isRawMaterialExtractor) {
            throw new BadRequestException({ translate: 'validation.this_role_cannot_initiate_a_chain' });
        }
        if (!isNil(fromRoleId) && role.isRawMaterialExtractor) {
            throw new BadRequestException({ translate: 'validation.this_role_cannot_be_added_to_downstream' });
        }
        if (isNil(fromRoleId) && node) {
            throw new BadRequestException({ translate: 'validation.this_role_cannot_initiate_a_chain' });
        }
    }

    async getUpdateMetadataImpacts(metadataId: string, dto: UpdateSupplyChainNodeMetadataDto): Promise<NodeImpacts> {
        const metadata = await this.supplyChainNodeMetadataRepo.findById(metadataId);
        const nodes = await this.supplyChainNodeRepo.findBy({ chainId: metadata.chainId, roleId: metadata.roleId });
        const nodeIds = nodes.map((node) => node.id);
        const outputProductDefinitionIds = uniq(nodes.flatMap((node) => node.outputProductDefinitionIds));
        const hasImpact = difference(outputProductDefinitionIds, dto.outputProductDefinitionIds).length > 0;

        if (isEmpty(dto.outputProductDefinitionIds)) {
            return this.getNodeImpacts({ chainId: metadata.chainId, deleteLineIds: nodeIds });
        }

        if (!hasImpact) {
            return { deletedLineIds: [], deletedNodeIds: [], impactNodeIds: [] };
        }

        return this.getNodeImpacts({ chainId: metadata.chainId, impactNodeIds: [metadata.id] });
    }

    async getDeletionNodesImpact(dto: DeleteSupplyChainNodesDto): Promise<NodeImpacts> {
        return this.getNodeImpacts({ chainId: dto.chainId, deleteLineIds: dto.nodeIds });
    }

    // eslint-disable-next-line sonarjs/cognitive-complexity
    async getNodeImpacts(params: GetNodeImpactsParams): Promise<NodeImpacts> {
        const { metadataMap, nodesGroup, parentNodesGroup, rootNodes } = await this.prepareNodeImpactParams(params);

        const deletedLineIds = new Set(params.deleteLineIds);
        const impactNodeIds = new Set(params.impactNodeIds);
        const deletedNodeIds = new Set<string>();
        const visitedNodeIds = new Set<string>();
        const traversedNodeIds = new Set<string>();

        const traverse = (node: SupplyChainNodeEntity): void => {
            const metadata = metadataMap[node.roleId];
            const parentNodes = nodesGroup[node.roleId].filter((node) => !deletedLineIds.has(node.id));
            const nextNodes = parentNodesGroup[node.roleId] || [];
            const isImpact = nodesGroup[node.roleId].some(
                (node) => deletedLineIds.has(node.id) || impactNodeIds.has(metadata.id)
            );
            if (traversedNodeIds.has(metadata.id)) {
                return;
            }
            if (visitedNodeIds.has(metadata.id)) {
                throw new BadRequestException({ translate: 'validation.cycle_detected' });
            }
            if (isEmpty(parentNodes)) {
                deletedNodeIds.add(metadata.id);
                impactNodeIds.delete(metadata.id);
                for (const nextNode of nextNodes) {
                    const metadataId = metadataMap[nextNode.roleId].id;
                    visitedNodeIds.delete(metadataId);
                    traversedNodeIds.delete(metadataId);
                    deletedLineIds.add(nextNode.id);
                }
            } else if (isImpact) {
                impactNodeIds.add(metadata.id);
                for (const nextNode of nextNodes) {
                    impactNodeIds.add(metadataMap[nextNode.roleId].id);
                }
            }
            visitedNodeIds.add(metadata.id);
            for (const nextNode of nextNodes) traverse(nextNode);
            traversedNodeIds.add(metadata.id);
        };

        for (const rootNode of rootNodes) traverse(rootNode);

        return {
            deletedLineIds: Array.from(deletedLineIds),
            deletedNodeIds: Array.from(deletedNodeIds),
            impactNodeIds: Array.from(impactNodeIds)
        };
    }

    private async prepareNodeImpactParams(params: GetNodeImpactsParams): Promise<NodeImpactsParams> {
        const [metadataList, nodes] = await Promise.all([
            this.supplyChainNodeMetadataRepo.findBy({ chainId: params.chainId }),
            this.supplyChainNodeRepo.findBy({ chainId: params.chainId })
        ]);

        const metadataMap = keyBy(metadataList, (metadata) => metadata.roleId);
        const nodesGroup = groupBy(nodes, (node) => node.roleId);
        const parentNodesGroup = groupBy(nodes, (node) => node.fromRoleId);
        const rootNodes = nodes.filter((node) => isNil(node.fromRoleId));

        return {
            metadataMap,
            nodesGroup,
            parentNodesGroup,
            rootNodes
        };
    }

    async processImpactNodes(impacts: NodeImpacts): Promise<void> {
        const { deletedLineIds, deletedNodeIds, impactNodeIds } = impacts;
        if (!isEmpty(deletedLineIds)) {
            await this.supplyChainNodeRepo.delete(deletedLineIds);
        }

        await this.calculateFieldService.deleteCalculateFields(concat(impactNodeIds, deletedNodeIds));
        if (!isEmpty(deletedNodeIds)) {
            await this.supplyChainNodeMetadataRepo.delete(deletedNodeIds);
        }
    }

    async deleteSupplyChainNodes(dto: DeleteSupplyChainNodesDto): Promise<void> {
        const nodeImpacts = await this.getNodeImpacts({ chainId: dto.chainId, deleteLineIds: dto.nodeIds });
        await this.processImpactNodes(nodeImpacts);
    }

    async deleteSupplyChainMetadata(metadataId: string): Promise<void> {
        const { roleId, chainId } = await this.supplyChainNodeMetadataRepo.findById(metadataId);
        const nodes = await this.supplyChainNodeRepo.findBy({ roleId, chainId });
        const nodeIds = nodes.map((node) => node.id);
        const nodeImpacts = await this.getNodeImpacts({ chainId, deleteLineIds: nodeIds });
        await this.processImpactNodes(nodeImpacts);
    }

    async updateSupplyChainMetadata(metadataId: string, dto: UpdateSupplyChainNodeMetadataDto): Promise<void> {
        const metadata = await this.supplyChainNodeMetadataRepo.findById(metadataId);
        const nodeImpacts = await this.getUpdateMetadataImpacts(metadataId, dto);
        await this.processImpactNodes(nodeImpacts);

        if (!nodeImpacts.deletedNodeIds.includes(metadataId)) {
            await this.supplyChainNodeMetadataRepo.update({ id: metadataId }, { position: dto.position });
            await this.supplyChainNodeRepo.update(
                { roleId: metadata.roleId },
                { outputProductDefinitionIds: dto.outputProductDefinitionIds }
            );
        }
    }
}
