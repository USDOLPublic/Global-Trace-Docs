import { BadRequestException, Injectable } from '@nestjs/common';
import { CategoryEntity } from '~categories/entities/category.entity';
import { CategoryFilterDto } from '~categories/http/dto/category-filter.dto';
import { CategoryRepository } from '../repositories/category.repository';
import { FindOptionsWhere } from 'typeorm/find-options/FindOptionsWhere';
import { CategoryTypeEnum } from '~categories/enums/category-type.enum';
import { FindManyOptions, FindOneOptions, ILike, In, Not } from 'typeorm';
import { DNA_RISK_INDICATOR, DNA_RISK_SUB_INDICATOR } from '~risk-assessments/constants/dna-risk.constant';
import { SettingIconCategoriesDto } from '~categories/http/dto/setting-icon-categories.dto';
import { Dictionary, groupBy, isEmpty, keyBy, omit } from 'lodash';
import { TransactionService } from '@diginexhk/nestjs-transaction';
import { ValidationError } from '@diginexhk/nestjs-exception';
import { CategoryWithIndicators } from '~categories/types/category-with-indicators.type';
import { IndicatorWithSubIndicators } from '~categories/types/indicator-with-sub-indicators.type';
import { MappedCategoriesAndIndicators } from '~categories/types/mapped-categories-and-subindicator.type';
import { CategoryValidationService } from './category-validation.service';
import { CreateNestedCategories, CreateNestedCategory } from '~category-utils/http/dto/create-nested-categories.dto';

@Injectable()
export class CategoryService extends TransactionService {
    public constructor(
        private categoryRepo: CategoryRepository,
        private categoryValidationService: CategoryValidationService
    ) {
        super();
    }

    async all(dto: CategoryFilterDto): Promise<CategoryEntity[]> {
        const whereConditions: FindOptionsWhere<CategoryEntity> = {};

        if (dto.type) {
            whereConditions['type'] = dto.type;

            const ignoreIds = await this.categoryRepo.getIgnoreCategoryIds(dto.type);
            if (!isEmpty(ignoreIds)) {
                whereConditions['id'] = Not(In(ignoreIds));
            }
        }

        if (!isEmpty(dto.parentIds)) {
            whereConditions['parentId'] = In(dto?.parentIds);
        }

        return this.categoryRepo.find({
            relations: ['category'],
            where: whereConditions,
            order: { name: 'ASC' }
        });
    }

    find(options: FindManyOptions<CategoryEntity>): Promise<CategoryEntity[]> {
        return this.categoryRepo.find(options);
    }

    findOneOrFail(options: FindOneOptions<CategoryEntity>): Promise<CategoryEntity> {
        return this.categoryRepo.findOneOrFail(options);
    }

    saveOne(data: Partial<CategoryEntity>): Promise<CategoryEntity> {
        return this.categoryRepo.save(data);
    }

    findOneBy(options: FindOptionsWhere<CategoryEntity>): Promise<CategoryEntity> {
        return this.categoryRepo.findOneBy(options);
    }

    findByIds(ids: string[]): Promise<CategoryEntity[]> {
        return this.categoryRepo.findByIds(ids);
    }

    count(conditions: FindManyOptions<CategoryEntity>): Promise<number> {
        return this.categoryRepo.count(conditions);
    }

    async updateOrFail(conditions: FindOptionsWhere<CategoryEntity>, data: Partial<CategoryEntity>): Promise<void> {
        await this.categoryRepo.updateOrFail(conditions, data);
    }

    async update(id: string, data: Partial<CategoryEntity>): Promise<void> {
        await this.categoryRepo.update(id, data);
    }

    async updateBy(options: FindOptionsWhere<CategoryEntity>, data: Partial<CategoryEntity>): Promise<void> {
        await this.categoryRepo.update(options, data);
    }

    createOne(data: Partial<CategoryEntity>): Promise<CategoryEntity> {
        return this.categoryRepo.createOne(data);
    }

    getDnaRiskIndicator(): Promise<{ indicator: CategoryEntity; subIndicator: CategoryEntity }> {
        return this.getComboIndicatorAndSubIndicator(DNA_RISK_INDICATOR, DNA_RISK_SUB_INDICATOR);
    }

    async softDelete(options: FindOptionsWhere<CategoryEntity>): Promise<void> {
        await this.categoryRepo.softDelete(options);
    }

    exist(options: FindManyOptions<CategoryEntity>): Promise<boolean> {
        return this.categoryRepo.exist(options);
    }

    async getComboIndicatorAndSubIndicator(
        indicatorName: string,
        subIndicatorName: string
    ): Promise<{ indicator: CategoryEntity; subIndicator: CategoryEntity }> {
        const indicator = await this.categoryRepo.findOne({
            where: {
                name: ILike(indicatorName),
                type: CategoryTypeEnum.INDICATOR
            },
            relations: ['category']
        });
        let subIndicator: CategoryEntity;

        if (indicator) {
            subIndicator = await this.categoryRepo.findOneBy({
                name: ILike(subIndicatorName),
                type: CategoryTypeEnum.SUB_INDICATOR,
                parentId: indicator.id
            });
        }
        return { indicator, subIndicator };
    }

    async getSettingCategories(): Promise<CategoryEntity[]> {
        const conditions = { type: In([CategoryTypeEnum.CATEGORY, CategoryTypeEnum.EXTERNAL_RISK_INDICATOR]) };

        const ignoreIds = await this.categoryRepo.getIgnoreCategoryIds(CategoryTypeEnum.CATEGORY);
        if (!isEmpty(ignoreIds)) {
            conditions['id'] = Not(In(ignoreIds));
        }

        return this.categoryRepo.find({ where: conditions });
    }

    async settingIconCategory({ settingIconCategories }: SettingIconCategoriesDto): Promise<void> {
        const categoryIds = settingIconCategories.map(({ categoryId }) => categoryId);
        const existedCategories = await this.categoryRepo.find({
            where: {
                id: In(categoryIds),
                type: In([CategoryTypeEnum.CATEGORY, CategoryTypeEnum.EXTERNAL_RISK_INDICATOR])
            }
        });

        if (categoryIds.length !== existedCategories.length) {
            throw new BadRequestException({ translate: 'error.invalid_icon_category' });
        }

        const mappedCategories = keyBy(settingIconCategories, 'categoryId');
        const preparedCategories = existedCategories.map((category) => ({
            ...category,
            icon: mappedCategories[category.id]?.icon || null
        }));

        await this.categoryRepo.save(preparedCategories);
    }

    getAllNestedCategories(): Promise<CategoryEntity[]> {
        return this.categoryRepo.getAllNestedCategories();
    }

    async createNestedCategories({ categories }: CreateNestedCategories): Promise<CategoryEntity[]> {
        const errors: ValidationError[] = [];
        await this.categoryValidationService.validateUniqueSubAndIndicatorPairs(categories, errors);
        const [categoryNames, indicatorNames] = this.getCategoryAndIndicatorNames(categories);

        await this.categoryValidationService.validateIndicatorBelongToOtherCategories(
            indicatorNames,
            categories,
            errors
        );
        this.categoryValidationService.validateDuplicatedNames(categoryNames, indicatorNames, errors);
        this.categoryValidationService.validateDuplicatedSubIndicatorNames(categories, errors);

        const { mappedExistedCategories, mappedExistedIndicators } = await this.findAndMapCategoriesAndIndicators(
            categoryNames,
            indicatorNames
        );

        const createdCategories = await this.createCategories(categories, mappedExistedCategories);
        const createdIndicators = await this.createIndicators(createdCategories, mappedExistedIndicators);

        const preparedSubIndicators = createdIndicators.flatMap(({ id: parentId, categoryId, subIndicators }) => {
            return subIndicators.map((subIndicator) => ({
                ...subIndicator,
                parentId,
                categoryId,
                type: CategoryTypeEnum.SUB_INDICATOR
            }));
        });

        const createdSubIndicators = await this.categoryRepo.save(preparedSubIndicators);
        const groupedIndicators = groupBy(createdIndicators, 'categoryId');
        const groupedSubIndicators = groupBy(createdSubIndicators, 'parentId');

        return createdCategories.map(
            (category) =>
                ({
                    ...omit(category, 'indicators'),
                    indicators: groupedIndicators[category.id].map((indicator) => ({
                        ...omit(indicator, 'subIndicators'),
                        subIndicators: groupedSubIndicators[indicator.id]
                    }))
                }) as CategoryEntity
        );
    }

    private async createCategories(
        categories: CreateNestedCategory[],
        mappedExistedCategories: Dictionary<CategoryEntity>
    ): Promise<CategoryWithIndicators[]> {
        const [existedCategories, preparedCategories] = categories.reduce(
            (acc, category) => {
                const existedCategory = mappedExistedCategories[category.name];
                if (existedCategory) {
                    acc[0].push({ ...existedCategory, indicators: category.indicators });
                } else {
                    acc[1].push({
                        ...category,
                        type: CategoryTypeEnum.CATEGORY
                    });
                }
                return acc;
            },
            [[], []]
        );

        const createdCategories = await this.categoryRepo.save(preparedCategories);

        return existedCategories.concat(createdCategories);
    }

    private async createIndicators(
        createdCategories: CategoryWithIndicators[],
        mappedExistedIndicators: Dictionary<CategoryEntity>
    ): Promise<IndicatorWithSubIndicators[]> {
        const preparedIndicators = [];
        const existedIndicators = [];

        for (const { id: categoryId, indicators } of createdCategories) {
            for (const indicator of indicators) {
                const existedIndicator = mappedExistedIndicators[indicator.name];
                if (existedIndicator) {
                    existedIndicators.push({ ...existedIndicator, subIndicators: indicator.subIndicators });
                } else {
                    preparedIndicators.push({
                        ...indicator,
                        categoryId,
                        id: mappedExistedIndicators[indicator.name]?.id,
                        type: CategoryTypeEnum.INDICATOR
                    });
                }
            }
        }
        const createdIndicators = await this.categoryRepo.save(preparedIndicators);

        return existedIndicators.concat(createdIndicators);
    }

    private async findAndMapCategoriesAndIndicators(
        categoryNames: string[],
        indicatorNames: string[]
    ): Promise<MappedCategoriesAndIndicators> {
        const [existedCategories, existedIndicators] = await Promise.all([
            this.categoryRepo.find({
                where: { type: CategoryTypeEnum.CATEGORY, name: In(categoryNames) }
            }),
            this.categoryRepo.find({
                where: { type: CategoryTypeEnum.INDICATOR, name: In(indicatorNames) }
            })
        ]);

        return {
            mappedExistedCategories: keyBy(existedCategories, 'name'),
            mappedExistedIndicators: keyBy(existedIndicators, 'name')
        };
    }

    private getCategoryAndIndicatorNames(categories: CreateNestedCategory[]): string[][] {
        return categories.reduce(
            (acc, category) => {
                acc[0].push(category.name);
                acc[1].push(...category.indicators.map(({ name }) => name));
                return acc;
            },
            [[] as string[], [] as string[]]
        );
    }

    existsSubAndIndicatorPairById(indicatorId: string, subIndicatorId: string): Promise<boolean> {
        return this.categoryRepo.existsSubAndIndicatorPairById(indicatorId, subIndicatorId);
    }

    findIndicatorsByCategoryId(categoryId: string): Promise<CategoryEntity[]> {
        return this.categoryRepo.findIndicatorsByCategoryId(categoryId);
    }
}
