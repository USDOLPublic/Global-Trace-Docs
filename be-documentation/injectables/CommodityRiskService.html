import { Injectable } from '@nestjs/common';
import { FindManyOptions, FindOneOptions } from 'typeorm';
import { CommodityRiskEntity } from '~commodity-risks/entities/commodity-risk.entity';
import { CommodityEntity } from '~commodity-risks/entities/commodity.entity';
import { CreateCommodityRiskDto } from '~commodity-risks/http/dto/create-commodity-risk.dto';
import { SearchCommodityRiskDto } from '~commodity-risks/http/dto/search-commodity-risk.dto';
import { CommodityRiskRepository } from '~commodity-risks/repositories/commodity-risk.repository';
import { CommodityRepository } from '~commodity-risks/repositories/commodity.repository';
import { ILike } from '~core/operators/ilike.operator';

@Injectable()
export class CommodityRiskService {
    public constructor(
        private commodityRiskRepo: CommodityRiskRepository,
        private commodityRepo: CommodityRepository
    ) {}

    async createCommodityRisk(dto: CreateCommodityRiskDto): Promise<CommodityRiskEntity> {
        const existedCommodity = await this.commodityRepo.findOne({ where: { commodity: ILike(dto.commodity) } });
        if (!existedCommodity) {
            await this.commodityRepo.createOne({ commodity: dto.commodity });
        }

        const existedCommodityRisk = await this.commodityRiskRepo.findOne({
            where: {
                countryId: dto.countryId,
                commodity: ILike(dto.commodity),
                subIndicatorId: dto.subIndicatorId
            }
        });

        return this.commodityRiskRepo.save({ ...existedCommodityRisk, ...dto });
    }

    async searchCommodityRisks(query: SearchCommodityRiskDto): Promise<CommodityRiskEntity[]> {
        return this.commodityRiskRepo.searchCommodityRisks(query);
    }

    find(options: FindOneOptions<CommodityRiskEntity>): Promise<CommodityRiskEntity[]> {
        return this.commodityRiskRepo.find(options);
    }

    exists(options: FindManyOptions<CommodityRiskEntity>): Promise<boolean> {
        return this.commodityRiskRepo.exists(options);
    }

    findCommodities(options: FindOneOptions<CommodityEntity>): Promise<CommodityEntity[]> {
        return this.commodityRepo.find(options);
    }
}
