import { TransactionService } from '@diginexhk/nestjs-transaction';
import { Injectable } from '@nestjs/common';
import { isEmpty, pick } from 'lodash';
import { FindManyOptions } from 'typeorm';
import { RoleEntity } from '~role-permissions/entities/role.entity';
import { RoleTypeEnum } from '~role-permissions/enums/role-type.enum';
import { SupplyChainNodeEntity } from '~supply-chains/entities/supply-chain-node.entity';
import { SupplyChainNodeRepository } from '~supply-chains/repositories/supply-chain-node.repository';
import { Tier } from '~supply-chains/types/tier.type';

@Injectable()
export class SupplyChainTierService extends TransactionService {
    constructor(private supplyChainNodeRepo: SupplyChainNodeRepository) {
        super();
    }

    async getTiers(role: RoleEntity, doesAddBrand: boolean): Promise<Tier[][]> {
        const nodes = await this.getNodes();
        const getAllNestedChildren = this.getAllNestedChildrenTiers(nodes, null, null, 0);

        if (role.type === RoleTypeEnum.BRAND) {
            return this.getBrandTiers(role, doesAddBrand, getAllNestedChildren);
        }

        const chainsContainingRoleId = this.filterChainsContainingRoleId(getAllNestedChildren, role.id);
        const orderNestedChildren = chainsContainingRoleId.sort((a, b) => this.getMaxLevel(b) - this.getMaxLevel(a));
        return this.groupTiersByLevel(orderNestedChildren);
    }

    private getBrandTiers(role: RoleEntity, doesAddBrand: boolean, getAllNestedChildren: Tier[]) {
        const orderNestedChildren = getAllNestedChildren.sort((a, b) => this.getMaxLevel(b) - this.getMaxLevel(a));
        const groupedTiers = this.groupTiersByLevel(orderNestedChildren);
        if (doesAddBrand) {
            return [[pick(role, ['id', 'name'])], ...groupedTiers];
        }
        return groupedTiers;
    }

    private async getNodes(options: FindManyOptions<SupplyChainNodeEntity> = {}): Promise<SupplyChainNodeEntity[]> {
        const supplyChainNodes = await this.supplyChainNodeRepo.find({
            ...options,
            relations: ['role', 'fromRole']
        });
        return supplyChainNodes.filter(({ role, fromRole, fromRoleId }) => role && (!fromRoleId || fromRole));
    }

    private getAllNestedChildrenTiers(
        nodes: SupplyChainNodeEntity[],
        nodeFromRoleId: string | null,
        nodeChainId: string | null,
        level: number = 1
    ): Tier[] {
        let currentNodes: SupplyChainNodeEntity[] = [];
        if (!nodeChainId) {
            currentNodes = nodes.filter(({ fromRoleId }) => fromRoleId === nodeFromRoleId);
        } else {
            currentNodes = nodes.filter(
                ({ roleId, fromRoleId, chainId }) =>
                    fromRoleId === nodeFromRoleId && chainId === nodeChainId && roleId !== fromRoleId
            );
        }

        return currentNodes.map((node) => ({
            name: node.role.name,
            id: node.role.id,
            children: this.getAllNestedChildrenTiers(nodes, node.role.id, node.chainId, level + 1),
            level
        }));
    }

    private filterChainsContainingRoleId(nodes: Tier[], roleId: string): Tier[] {
        return nodes
            .map((item) => {
                if (item.id === roleId) {
                    const newChildren = item.children.map((child) => ({ ...child, children: [] }));
                    return { ...item, children: newChildren };
                }

                const filteredChildren = this.filterChainsContainingRoleId(item.children || [], roleId);
                if (filteredChildren.length > 0) {
                    return {
                        ...item,
                        children: filteredChildren
                    };
                }
                return null;
            })
            .filter(Boolean);
    }

    private groupTiersByLevel(nestedChildrenTiers: Tier[]): Tier[][] {
        const groupedLevels: Record<number, Tier[]> = {};

        const traverse = (node: Tier) => {
            if (node.children) {
                node.children = node.children.sort((a, b) => this.getMaxLevel(b) - this.getMaxLevel(a));
            }

            if (!groupedLevels[node.level]) {
                groupedLevels[node.level] = [];
            }

            if (node.level && isEmpty(node.children)) {
                const keys = Object.keys(groupedLevels);
                const lastKey = keys[keys.length - 1];
                if (!groupedLevels[lastKey].some(({ id }) => id === node.id)) {
                    groupedLevels[lastKey].push({ id: node.id, name: node.name });
                }
            } else {
                const nodeExists = Object.values(groupedLevels).some((level) =>
                    level.some((item) => item.id === node.id)
                );
                if (!nodeExists && !groupedLevels[node.level].some(({ id }) => id === node.id)) {
                    groupedLevels[node.level].push({ id: node.id, name: node.name });
                }
            }
            node.children?.forEach(traverse);
        };
        nestedChildrenTiers.forEach(traverse);

        return Object.values(groupedLevels).reverse();
    }

    private getMaxLevel(node: Tier): number {
        if (!node.children || node.children.length === 0) {
            return node.level || 0;
        }
        return Math.max(node.level || 0, ...node.children.map((child) => this.getMaxLevel(child)));
    }
}
