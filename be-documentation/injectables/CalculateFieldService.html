import { TransactionService } from '@diginexhk/nestjs-transaction';
import { BadRequestException, forwardRef, Inject, Injectable } from '@nestjs/common';
import { isEmpty, isString } from 'lodash';
import { FindOptionsWhere, In } from 'typeorm';
import { CalculateFieldEntity } from '~calculate-field/entities/calculate-field.entity';
import { ExpressionItemEnum } from '~calculate-field/enums/expression-item.enum';
import { validateExpressions } from '~calculate-field/helpers/expressions.helper';
import { AddCalculateFieldDto } from '~calculate-field/http/dto/add-calculate-field.dto';
import { CalculateFieldInputDto } from '~calculate-field/http/dto/calculate-field-input.dto';
import { ValidateCalculateFieldDto } from '~calculate-field/http/dto/validate-calculate-field.dto';
import { CalculateFieldRepository } from '~calculate-field/repositories/calculate-field.repository';
import { ExpressionItem } from '~calculate-field/types/expression-item.type';
import { FieldCategoryEnum } from '~product-definitions/enums/field-category.enum';
import { ProductDefinitionService } from '~product-definitions/services/product-definition.service';

@Injectable()
export class CalculateFieldService extends TransactionService {
    constructor(
        private calculateFieldRepo: CalculateFieldRepository,
        @Inject(forwardRef(() => ProductDefinitionService)) private productDefinitionService: ProductDefinitionService
    ) {
        super();
    }

    async findById(id: string): Promise<CalculateFieldEntity> {
        return this.calculateFieldRepo.findById(id);
    }

    findOneBy(options: FindOptionsWhere<CalculateFieldEntity>): Promise<CalculateFieldEntity> {
        return this.calculateFieldRepo.findOneBy(options);
    }

    async addCalculateField(dto: AddCalculateFieldDto): Promise<CalculateFieldEntity> {
        const partialEntity = await this.prepareCalculateField(
            dto.inputProductDefinitionIds,
            dto.outputProductDefinitionId,
            dto
        );
        return this.calculateFieldRepo.createOne(partialEntity);
    }

    async editCalculateField(calculateFieldId: string, dto: CalculateFieldInputDto): Promise<void> {
        const calculateField = await this.calculateFieldRepo.findById(calculateFieldId);
        const partialEntity = await this.prepareCalculateField(
            dto.inputProductDefinitionIds,
            calculateField.outputProductDefinitionId,
            dto
        );
        await this.calculateFieldRepo.update(calculateFieldId, partialEntity);
    }

    async deleteCalculateFields(metadataIds: string[]): Promise<void> {
        if (isEmpty(metadataIds)) {
            return;
        }
        await this.calculateFieldRepo.softDelete({ supplyChainNodeMetadataId: In(metadataIds) });
    }

    private async prepareCalculateField(
        inputProductDefinitionIds: string[],
        outputProductDefinitionId: string,
        dto: CalculateFieldInputDto
    ): Promise<Partial<CalculateFieldEntity>> {
        const { expressionItems, outputAttributeId } = dto;
        let variables: ExpressionItem[] = [];

        if (!isEmpty(expressionItems)) {
            if (!validateExpressions(expressionItems)) {
                throw new BadRequestException({ translate: 'validation.the_expression_is_not_valid' });
            }
            variables = expressionItems.filter((expression) => expression.type === ExpressionItemEnum.VARIABLE);
        }

        const outputAttributes = await this.productDefinitionService.getDefinitionValues({
            attributeIds: [outputAttributeId],
            categories: [FieldCategoryEnum.NUMBER, FieldCategoryEnum.NUMBER_UNIT_PAIR, FieldCategoryEnum.PERCENTAGE],
            productDefinitionIds: [outputProductDefinitionId]
        });

        const inputAttributes = !isEmpty(variables)
            ? await this.productDefinitionService.getDefinitionValues({
                  attributeIds: variables.map((item) => item.value).filter(isString),
                  productDefinitionIds: variables.map((item) => item.productDefinitionId).filter(isString),
                  categories: [
                      FieldCategoryEnum.NUMBER,
                      FieldCategoryEnum.NUMBER_UNIT_PAIR,
                      FieldCategoryEnum.PERCENTAGE
                  ]
              })
            : [];

        if (outputAttributes.length !== 1 || inputAttributes.length !== variables.length) {
            throw new BadRequestException({ translate: 'validation.the_expression_is_not_valid' });
        }
        return {
            supplyChainNodeMetadataId: dto.supplyChainNodeMetadataId,
            inputAttributeIds: variables.map((item) => item.value).filter(isString),
            expressionItems,
            outputProductDefinitionId,
            inputProductDefinitionIds,
            outputAttributeId
        };
    }

    async validateCalculateField(dto: ValidateCalculateFieldDto): Promise<void> {
        await this.prepareCalculateField(dto.inputProductDefinitionIds, dto.outputProductDefinitionId, dto);
    }
}
