import { TransactionService } from '@diginexhk/nestjs-transaction';
import { Injectable } from '@nestjs/common';
import { Dictionary, groupBy } from 'lodash';
import { DeleteResult, FindManyOptions, In, UpdateResult } from 'typeorm';
import { RoleAttributeTypeEnum } from '~role-attributes/enums/role-attribute-type.enum';
import { CreateAdditionalAttributeDto } from '~role-attributes/http/dto/create-addtional-attribute.dto';
import { AttributeEntity } from '~product-definitions/entities/attribute.entity';
import { FieldTypeEnum } from '~product-definitions/enums/field-type.enum';
import { AttributeRepository } from '~product-definitions/repositories/attribute.repository';
import { RoleAttributeRepository } from '~role-attributes/repositories/role-attribute.repository';
import { RoleAttributeEntity } from '~role-attributes/entities/role-attribute.entity';
import { ValidateException } from '@diginexhk/nestjs-exception';
import { RoleAttributeResponse } from '~role-attributes/response/role-attribute.response';
import { AttributeResponse } from '~product-definitions/http/response/attribute.response';
import { LOCATION } from '~role-attributes/constants/location.constant';

@Injectable()
export class RoleAttributeService extends TransactionService {
    public constructor(
        private attributeRepo: AttributeRepository,
        private roleAttributeRepo: RoleAttributeRepository
    ) {
        super();
    }

    find(options: FindManyOptions<RoleAttributeEntity>): Promise<RoleAttributeEntity[]> {
        return this.roleAttributeRepo.find(options);
    }

    async getIdentifierSystemAttributes(
        types: RoleAttributeTypeEnum[] = []
    ): Promise<Dictionary<AttributeResponse[]> | []> {
        if (!types.length) {
            return [];
        }
        const identifierSystemAttributes = await this.attributeRepo.find({
            where: {
                type: In(types)
            }
        });

        return groupBy(identifierSystemAttributes, 'type');
    }

    getAdditionalAttributes(): Promise<AttributeEntity[]> {
        return this.attributeRepo.find({
            where: {
                type: FieldTypeEnum.ADDITIONAL_ATTRIBUTE
            }
        });
    }

    async getAttributesByRoleIds(roleIds: string[]): Promise<Dictionary<RoleAttributeResponse[]>> {
        const roleAttributes = await this.roleAttributeRepo.find({
            relations: ['attribute'],
            where: {
                roleId: In(roleIds)
            }
        });

        return groupBy(roleAttributes, 'roleId');
    }

    async mappedAttributeWithType(roleId: string): Promise<Dictionary<RoleAttributeResponse[]>> {
        const roleAttributes = await this.roleAttributeRepo.find({
            relations: ['attribute'],
            where: { roleId },
            order: { order: 'ASC' }
        });

        return groupBy(roleAttributes, 'roleAttributeType');
    }

    createAdditionalAttribute(dto: CreateAdditionalAttributeDto): Promise<AttributeEntity> {
        const attributeData = this.convertToAttributeData(dto);
        return this.attributeRepo.createOne(attributeData);
    }

    async updateAdditionalAttribute(id: string, data: CreateAdditionalAttributeDto): Promise<UpdateResult> {
        const attributeData = this.convertToAttributeData(data);
        return await this.attributeRepo.update(id, attributeData);
    }

    async deleteAdditionalAttribute(id: string) {
        return this.attributeRepo.delete(id);
    }

    private convertToAttributeData({
        options,
        name,
        category
    }: CreateAdditionalAttributeDto): Partial<AttributeEntity> {
        const values = options?.map(({ value }) => value);
        if (values && new Set(values).size !== values.length) {
            throw new ValidateException([
                {
                    property: 'value',
                    constraints: {
                        invalidField: {
                            message: 'the_value_is_duplicate',
                            detail: {}
                        } as any
                    }
                }
            ]);
        }

        return {
            name,
            type: FieldTypeEnum.ADDITIONAL_ATTRIBUTE,
            category,
            nameTranslation: { en: name },
            options: options?.map(({ value }) => ({
                value,
                translation: {
                    en: value
                }
            }))
        };
    }

    createMany(data: Partial<RoleAttributeEntity>[]): Promise<RoleAttributeEntity[]> {
        return this.roleAttributeRepo.save(data);
    }

    async upsertRoleAttributes(roleId: string, data: Partial<RoleAttributeEntity>[]): Promise<void> {
        await this.deleteRoleAttributesByRoleId(roleId);

        const stateProvinceRmi = await this.attributeRepo.findOne({
            select: ['id'],
            where: {
                type: FieldTypeEnum.RMI_CID,
                name: LOCATION.STATE_PROVINCE_REGION
            }
        });
        await this.roleAttributeRepo.save(
            data.map((item) => {
                if (stateProvinceRmi.id === item.attributeId) {
                    item.isOptional = true;
                }
                item.roleId = roleId;
                return item;
            })
        );
    }

    async deleteRoleAttributesByRoleId(roleId: string): Promise<DeleteResult> {
        return this.roleAttributeRepo.delete({ roleId });
    }

    getRoleAttributeOrderKeys(): string[] {
        return [
            RoleAttributeTypeEnum.OPEN_SUPPLY_HUB_ID,
            RoleAttributeTypeEnum.RMI_CID,
            RoleAttributeTypeEnum.INTERNAL_INDENTIFIER_SYSTEM,
            RoleAttributeTypeEnum.ADDITIONAL_ATTRIBUTE
        ];
    }

    async mappedAdditionalAttributes(roleId: string): Promise<Dictionary<RoleAttributeResponse[]>> {
        const roleAttributes = await this.roleAttributeRepo.find({
            relations: ['attribute'],
            where: { roleId, roleAttributeType: RoleAttributeTypeEnum.ADDITIONAL_ATTRIBUTE },
            order: { order: 'ASC' }
        });

        return groupBy(roleAttributes, 'roleAttributeType');
    }
}
