import { Injectable } from '@nestjs/common';
import { FarmLevelRiskExcelData } from '~facilities/types/facility-groups/xlsx-farm-level-risk-data.type';
import { FacilityEntity } from '~facilities/entities/facility.entity';
import { Dictionary, groupBy, keyBy, omit } from 'lodash';
import { FarmCertificationEnum } from '~facilities/enums/farm-certification.enum';
import { LocationService } from '~locations/services/location.service';
import { InstructionExcelType } from '~facilities/types/facility-groups/instruction-excel.type';
import { AdditionalRoleEnum } from '~facilities/enums/additional-role.enum';
import { ImportFacilityGroupDto } from '~facilities/http/dto/import-facility-group.dto';
import { convertDateToTimestamp } from '~core/helpers/time.helper';
import { FacilityRoleAttributeDto } from '~role-attributes/http/dto/facility-role-attribute.dto';
import { RoleAttributeEntity } from '~role-attributes/entities/role-attribute.entity';
import { FacilityInformation } from '~facilities/types/geographic-information.type';
import { GeographicInformationEnum } from '~role-attributes/enums/geographic-information.enum';
import { DataForMapExternalInfo } from '~pdf-exports/types/data-for-map-external-infor.type';
import { ExternalInfo } from '~facilities/types/external-info.type';
import { ValueRoleAttributeResponse } from '~role-attributes/response/value-role-attribute.response';
import { LOCATION } from '~role-attributes/constants/location.constant';
import { RoleAttributeTypeEnum } from '~role-attributes/enums/role-attribute-type.enum';
import { In } from 'typeorm';
import { FieldTypeEnum } from '~product-definitions/enums/field-type.enum';
import { FacilityRoleAttributeService } from '~role-attributes/services/facility-role-attribute.service';
import { AttributeService } from '~product-definitions/services/attribute.service';

@Injectable()
export class FacilityOperationService {
    constructor(
        private locationService: LocationService,
        private facilityRoleAttributeService: FacilityRoleAttributeService,
        private attributeService: AttributeService
    ) {}

    async getDataFacility(dto: ImportFacilityGroupDto, data: InstructionExcelType): Promise<Partial<FacilityEntity>> {
        const country = await this.locationService.findCountry({ country: data.country });
        const province = await this.locationService.findProvince({ province: data.province, countryId: country.id });
        const district = await this.locationService.findDistrict({ district: data.district, provinceId: province.id });

        return {
            typeId: dto.roleId,
            name: data.farmGroupName,
            districtId: district.id,
            provinceId: province.id,
            countryId: country.id,
            additionalRole: AdditionalRoleEnum.FARM_GROUP,
            facilityGroupFileId: dto.fileId,
            farmId: data.farmGroupId,
            additionalInformation: { areas: data.areas }
        };
    }

    async mapDataFarmItems(
        dataArray: FarmLevelRiskExcelData[],
        facility: FacilityEntity
    ): Promise<Partial<FacilityEntity>[]> {
        const chunkSize = 20;
        const farms = [];

        for (let i = 0; i < dataArray.length; i += chunkSize) {
            const chunkData = dataArray.slice(i, i + chunkSize);
            const farmItems: Partial<FacilityEntity>[] = await Promise.all(
                chunkData.map((data: FarmLevelRiskExcelData) => this.prepareFarmItem(facility, data))
            );

            farms.push(...farmItems);
        }
        return farms;
    }

    private async prepareFarmItem(
        facility: FacilityEntity,
        data: FarmLevelRiskExcelData
    ): Promise<Partial<FacilityEntity>> {
        const {
            id,
            certification,
            farmName,
            businessRegisterNumber,
            certificationExpiredDate,
            tehsil,
            firstNameContactor,
            lastNameContactor,
            contactPhoneNumber
        } = data;

        return omit(
            {
                farmGroupId: facility.id,
                name: farmName,
                businessRegisterNumber,
                typeId: facility.typeId,
                countryId: facility.countryId,
                provinceId: facility.provinceId,
                districtId: facility.districtId,
                certification: (certification ? certification : null) as FarmCertificationEnum,
                facilityGroupFileId: facility.facilityGroupFileId,
                farmId: `${facility.farmId}-${id}`,
                certificationExpiredDate: certificationExpiredDate
                    ? convertDateToTimestamp(certificationExpiredDate)
                    : null,
                additionalInformation: {
                    tehsil,
                    firstNameContactor,
                    lastNameContactor,
                    contactPhoneNumber
                }
            },
            id ? [] : ['id']
        );
    }

    getFacilityInformationByRMI(
        rMIAttributes: FacilityRoleAttributeDto[],
        mappedExistedRoleAttributes: Dictionary<RoleAttributeEntity>,
        facilityInformation: FacilityInformation
    ): FacilityInformation {
        for (const rMIAttribute of rMIAttributes) {
            const existedRoleAttribute = mappedExistedRoleAttributes[rMIAttribute.attributeId];
            if (!existedRoleAttribute) {
                continue;
            }

            if (!facilityInformation.name && existedRoleAttribute.attribute.name === 'Facility name') {
                facilityInformation['name'] = rMIAttribute.value;
            }

            if (!facilityInformation.countryId && existedRoleAttribute.attribute.name === 'Country location') {
                facilityInformation['countryId'] = rMIAttribute.value;
            }
        }

        return facilityInformation;
    }

    getFacilityInformationByOSH(
        openSupplyHubAttributes: FacilityRoleAttributeDto[],
        mappedExistedRoleAttributes: Dictionary<RoleAttributeEntity>
    ): FacilityInformation {
        const facilityInformation: FacilityInformation = {};
        for (const openSupplyHubAttribute of openSupplyHubAttributes) {
            const existedRoleAttribute = mappedExistedRoleAttributes[openSupplyHubAttribute.attributeId];
            if (!existedRoleAttribute) {
                continue;
            }

            switch (existedRoleAttribute.attribute.name) {
                case GeographicInformationEnum.COUNTRY:
                    facilityInformation['countryId'] = openSupplyHubAttribute.value;
                    break;
                case GeographicInformationEnum.PROVINCE:
                    facilityInformation['provinceId'] = openSupplyHubAttribute.value;
                    break;
                case GeographicInformationEnum.DISTRICT:
                    facilityInformation['districtId'] = openSupplyHubAttribute.value;
                    break;
                case GeographicInformationEnum.ADDRESS:
                    facilityInformation['address'] = openSupplyHubAttribute.value;
                    break;
                case 'Business Name':
                    facilityInformation['name'] = openSupplyHubAttribute.value;
                    break;
                case 'OS ID':
                    facilityInformation['oarId'] = openSupplyHubAttribute.value;
                    break;
            }
        }

        return facilityInformation;
    }

    async findDataForMapExternalInfo(tracedSupplierIds: string[]): Promise<DataForMapExternalInfo> {
        const [countries, provinces, districts, facilityRoleAttributes, attributes] = await Promise.all([
            this.locationService.listCountries(),
            this.locationService.getListProvinces(),
            this.locationService.getListDistricts(),
            this.facilityRoleAttributeService.find({
                select: ['facilityId', 'options', 'roleAttributeType'],
                where: {
                    facilityId: In(tracedSupplierIds),
                    roleAttributeType: In([RoleAttributeTypeEnum.RMI_CID, RoleAttributeTypeEnum.OPEN_SUPPLY_HUB_ID])
                }
            }),
            this.attributeService.find({
                where: { type: In([FieldTypeEnum.OPEN_SUPPLY_HUB_ID, FieldTypeEnum.RMI_CID]) }
            })
        ]);

        return {
            countryMap: keyBy(countries, (country) => country.id),
            provinceMap: keyBy(provinces, (province) => province.id),
            districtMap: keyBy(districts, (district) => district.id),
            attributesMap: keyBy(attributes, 'id'),
            facilityRoleAttributesGroup: groupBy(facilityRoleAttributes, 'facilityId')
        };
    }

    mapWithAllExternalInfo(supplier: FacilityEntity, dataForMapExternalInfo: DataForMapExternalInfo): void {
        const { facilityRoleAttributesGroup } = dataForMapExternalInfo;
        const facilityRoleAttributes = facilityRoleAttributesGroup[supplier.id] || [];
        if (!facilityRoleAttributes.length) {
            return;
        }

        supplier.identifierSystem = {};
        for (const { options, roleAttributeType } of facilityRoleAttributes) {
            supplier.identifierSystem[roleAttributeType] = this.mapWithEachExternalInfo(
                options,
                dataForMapExternalInfo
            );
        }
    }

    private mapWithEachExternalInfo(
        options: ValueRoleAttributeResponse[],
        dataForMapExternalInfo: DataForMapExternalInfo
    ): ExternalInfo[] {
        const { attributesMap } = dataForMapExternalInfo;
        return options.map(({ attributeId, value }) => {
            const attribute = attributesMap[attributeId];
            return {
                name: attribute.name,
                value: this.getValueByAttributeName(value, attribute.name, dataForMapExternalInfo)
            };
        });
    }

    private getValueByAttributeName(
        value: string,
        attributeName: string,
        dataForMapExternalInfo: DataForMapExternalInfo
    ): string {
        const { countryMap, provinceMap, districtMap } = dataForMapExternalInfo;
        switch (attributeName) {
            case LOCATION.COUNTRY:
                return countryMap[value]?.country;
            case LOCATION.PROVINCE_STATE:
                return provinceMap[value]?.province;
            case LOCATION.DISTRICT:
                return districtMap[value]?.district;
            default:
                return value;
        }
    }
}
