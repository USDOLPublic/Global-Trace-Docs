import { TransactionService } from '@diginexhk/nestjs-transaction';
import { BadRequestException, Inject, Injectable, forwardRef } from '@nestjs/common';
import { isNil } from '@nestjs/common/utils/shared.utils';
import { groupBy, keyBy, uniq } from 'lodash';
import { FindManyOptions, FindOneOptions, FindOptionsWhere, IsNull } from 'typeorm';
import { allSettled } from '~core/helpers/settled.helper';
import { RoleEntity } from '~role-permissions/entities/role.entity';
import { PermissionEnum } from '~role-permissions/enums/permission.enum';
import { RolePermissionService } from '~role-permissions/services/role-permission.service';
import { RoleService } from '~role-permissions/services/role.service';
import { SupplyChainNodeMetadataEntity } from '~supply-chains/entities/supply-chain-node-metadata.entity';
import { SupplyChainNodeEntity } from '~supply-chains/entities/supply-chain-node.entity';
import { AddSupplyChainNodeDto } from '~supply-chains/http/dto/add-supply-chain-node.dto';
import { AddSupplyChainNodesDto } from '~supply-chains/http/dto/add-supply-chain-nodes.dto';
import { UpdateSupplyChainNodeDto } from '~supply-chains/http/dto/update-supply-chain-node.dto';
import { SupplyChainNodeMetadataRepository } from '~supply-chains/repositories/supply-chain-node-metadata.repository';
import { SupplyChainNodeRepository } from '~supply-chains/repositories/supply-chain-node.repository';
import { Line } from '~supply-chains/types/line.type';
import { SupplyChainMapping } from '~supply-chains/types/supply-chain-mapping.type';
import { SupplyChainNodeMapping } from '~supply-chains/types/supply-chain-node-mapping.type';
import { SupplyChainNodeService } from './supply-chain-node.service';
import { SupplyChainMetaDataService } from './supply-chain-meta-data.service';
import { randomUUID } from 'crypto';
import { loadRelations } from '@diginexhk/typeorm-helper';
import { SupplyChainTierService } from './supply-chain-tier.service';

@Injectable()
export class SupplyChainService extends TransactionService {
    constructor(
        private supplyChainNodeRepo: SupplyChainNodeRepository,
        private rolePermissionService: RolePermissionService,
        @Inject(forwardRef(() => RoleService)) private roleService: RoleService,
        private supplyChainNodeMetadataRepo: SupplyChainNodeMetadataRepository,
        @Inject(forwardRef(() => SupplyChainNodeService)) private supplyChainNodeService: SupplyChainNodeService,
        private supplyChainMetaDataService: SupplyChainMetaDataService,
        private supplyChainTierService: SupplyChainTierService
    ) {
        super();
    }

    async addSupplyChainNode(dto: AddSupplyChainNodeDto): Promise<SupplyChainNodeMapping> {
        await this.supplyChainNodeService.validateRoleForSupplyChainNodeAddition(dto.roleId, dto.fromRoleId);
        if (dto.fromRoleId) {
            await this.supplyChainNodeService.checkForCycle(dto.roleId, dto.fromRoleId, dto.chainId);
            await this.validateSupplyChainNode(dto.roleId, dto.fromRoleId);
        }
        const partialEntity: Partial<SupplyChainNodeEntity> = dto;
        let supplyChainNode = await this.supplyChainNodeRepo.createOne(partialEntity);
        const [roleIdsAllowSale, roleIdsAllowPurchase] = await this.getRoleIdsAllowSaleAndAllowPurchase();
        return {
            ...supplyChainNode,
            hasBrokerIcon:
                roleIdsAllowPurchase.includes(supplyChainNode.roleId) ||
                roleIdsAllowSale.includes(supplyChainNode.fromRoleId)
        };
    }

    async addSupplyChainNodes(dto: AddSupplyChainNodesDto): Promise<SupplyChainNodeMapping[]> {
        const supplyChainNodes = [];
        for (const node of dto.supplyChainNodes) {
            if (isNil(node.fromRoleId)) {
                node.chainId = randomUUID();
            }
            await this.supplyChainMetaDataService.upsertNodeMetadata(node.roleId, node.position, node.chainId);
        }
        await allSettled(
            dto.supplyChainNodes.map(async (dto) => {
                const supplyChainNode = await this.addSupplyChainNode(dto);
                const fromMetadata = dto.fromRoleId
                    ? await this.supplyChainNodeMetadataRepo.findOneByOrFail({ roleId: dto.fromRoleId })
                    : null;
                const metadata = await this.supplyChainNodeMetadataRepo.findOne({
                    where: { roleId: supplyChainNode.roleId },
                    relations: ['role']
                });
                if (fromMetadata && fromMetadata.chainId != metadata.chainId) {
                    await this.supplyChainNodeMetadataRepo.update(
                        { chainId: fromMetadata.chainId },
                        { chainId: metadata.chainId }
                    );
                    await this.supplyChainNodeRepo.update(
                        { chainId: fromMetadata.chainId },
                        { chainId: metadata.chainId }
                    );
                }
                supplyChainNodes.push({
                    ...supplyChainNode,
                    position: metadata?.position,
                    outputProductDefinitionIds: dto.outputProductDefinitionIds,
                    role: metadata?.role,
                    supplyChainNodeMetadataId: metadata?.id,
                    calculateFields: []
                });
            })
        );

        return supplyChainNodes;
    }

    async updateSupplyChainNode(id: string, dto: UpdateSupplyChainNodeDto): Promise<void> {
        if (dto.fromRoleId) {
            await this.validateSupplyChainNode(dto.roleId, dto.fromRoleId);
        }
        const supplyChainNode = await this.supplyChainNodeRepo.findOneOrFail({
            select: ['roleId'],
            where: { id }
        });
        const { ...otherFields } = dto;
        const partialEntity: Partial<SupplyChainNodeEntity> = otherFields;
        await allSettled([
            this.supplyChainNodeRepo.update(id, partialEntity),
            this.supplyChainNodeRepo.update({ fromRoleId: supplyChainNode.roleId }, { fromRoleId: dto.roleId })
        ]);
    }

    private async validateSupplyChainNode(roleId: string, fromRoleId: string) {
        await this.validatePurchaseRole(roleId);
        const firstNode = await this.supplyChainNodeRepo.findOneBy({
            roleId: fromRoleId,
            fromRoleId: IsNull()
        });
        if (!firstNode) {
            await this.validateSaleRole(fromRoleId);
        }
    }

    private async validatePurchaseRole(roleId: string) {
        const canRolePurchase = await this.rolePermissionService.findOne({
            where: { roleId, permission: { action: PermissionEnum.LOG_PURCHASE } },
            relations: ['role']
        });
        if (!canRolePurchase) {
            const role = await this.roleService.findRoleById(roleId);
            throw new BadRequestException({
                translate: 'validation.role_can_not_purchase',
                args: { roleName: role.name }
            });
        }
    }

    private async validateSaleRole(fromRoleId: string) {
        const canRoleSell = await this.rolePermissionService.findOne({
            where: { roleId: fromRoleId, permission: { action: PermissionEnum.LOG_SALE } },
            relations: ['role']
        });
        if (!canRoleSell) {
            const role = await this.roleService.findRoleById(fromRoleId);
            throw new BadRequestException({
                translate: 'validation.role_can_not_sell',
                args: { roleName: role.name }
            });
        }
    }

    async deleteByOption(option: FindOneOptions<SupplyChainNodeMetadataEntity>): Promise<void> {
        const metadata = await this.supplyChainNodeMetadataRepo.findOne(option);
        if (!metadata) {
            return;
        }
        await this.supplyChainNodeService.deleteSupplyChainMetadata(metadata.id);
    }

    async getSupplyChain(): Promise<SupplyChainMapping[]> {
        const [metadataList, supplyChainNodes, [roleIdsAllowSale, roleIdsAllowPurchase]] = await Promise.all([
            this.supplyChainNodeMetadataRepo.find({ where: {}, relations: ['role', 'calculateFields'] }),
            this.supplyChainNodeRepo.find({ where: {}, relations: ['role', 'fromRole'] }),
            this.getRoleIdsAllowSaleAndAllowPurchase()
        ]);
        const chainIds = uniq(supplyChainNodes.map((node) => node.chainId));
        return chainIds.map((chainId) => {
            const lines = supplyChainNodes.filter((node) => node.chainId === chainId);
            const nodes: SupplyChainMapping['nodes'] = metadataList.filter((metadata) => metadata.chainId === chainId);
            const linesGroup = groupBy(lines, (node) => node.roleId);
            for (const node of nodes) {
                node.outputProductDefinitionIds = uniq(
                    (linesGroup[node.roleId] || []).flatMap((node) => node.outputProductDefinitionIds)
                );
            }
            return {
                nodes,
                lines: this.getLines(lines, nodes, roleIdsAllowSale, roleIdsAllowPurchase),
                chainId
            };
        });
    }

    private getRoleIdsAllowSaleAndAllowPurchase(): Promise<[string[], string[]]> {
        return Promise.all([
            this.rolePermissionService.findRoleIdsHasPermissionByAction(PermissionEnum.ALLOW_SALE_INTERMEDIARIES),
            this.rolePermissionService.findRoleIdsHasPermissionByAction(PermissionEnum.ALLOW_PURCHASE_INTERMEDIARIES)
        ]);
    }

    private getLines(
        supplyChainNodes: SupplyChainNodeEntity[],
        metadataList: SupplyChainNodeMetadataEntity[],
        roleIdsAllowSale: string[],
        roleIdsAllowPurchase: string[]
    ): Line[] {
        const lines = [];
        const metadataMap = keyBy(metadataList, (metadata) => metadata.roleId);
        for (const node of supplyChainNodes) {
            const { roleId, fromRoleId } = node;
            const toNode = metadataMap[roleId];
            const fromNode = node.fromRoleId ? metadataMap[fromRoleId] : null;
            if (toNode) {
                lines.push({
                    toNodeId: toNode.id,
                    fromNodeId: fromNode?.id,
                    hasBrokerIcon: roleIdsAllowPurchase.includes(roleId) || roleIdsAllowSale.includes(fromRoleId),
                    relation: node
                });
            }
        }
        return lines;
    }

    async getPartnerRoleIds(roleId: string): Promise<string[]> {
        const supplyChainNodes = await this.supplyChainNodeRepo.getPartnerRoleIds(roleId);
        const partnerRoleIds: string[] = [];
        for (const node of supplyChainNodes) {
            if (node.fromRoleId === roleId) {
                partnerRoleIds.push(node.roleId);
            } else if (node.roleId === roleId && node.fromRoleId) {
                partnerRoleIds.push(node.fromRoleId);
            }
        }
        return partnerRoleIds;
    }

    async isOutputOfSecondNode(outputProductDefinitionId: string, roleId?: string): Promise<boolean> {
        const supplyChainNodes = await this.supplyChainNodeRepo.findByProductDefinitionId(
            outputProductDefinitionId,
            roleId
        );
        if (!supplyChainNodes.length) {
            return false;
        }
        await loadRelations(supplyChainNodes, ['fromRole']);
        return supplyChainNodes[0]?.fromRole?.isRawMaterialExtractor;
    }

    async doesBuyFromRawMaterialExtractor(roleId: string): Promise<boolean> {
        const preNodeRoles = await this.getRolesOfPreviousNodes(roleId);
        return preNodeRoles.some(({ isRawMaterialExtractor }) => isRawMaterialExtractor);
    }

    async getRolesOfPreviousNodes(roleId: string): Promise<RoleEntity[]> {
        const supplyChainNodes = await this.supplyChainNodeRepo.find({
            where: { roleId },
            relations: ['fromRole']
        });
        return supplyChainNodes.map((node) => node.fromRole).filter((fromRole) => !!fromRole);
    }

    async getRawMaterialExtractorRole(roleId: string): Promise<RoleEntity> {
        const preNodeRoles = await this.getRolesOfPreviousNodes(roleId);
        for (const preNodeRole of preNodeRoles) {
            if (preNodeRole.isRawMaterialExtractor) {
                return preNodeRole;
            }
        }
        for (const preNodeRole of preNodeRoles) {
            const extractorRole = await this.getRawMaterialExtractorRole(preNodeRole.id);
            if (extractorRole) {
                return extractorRole;
            }
        }
        return null;
    }

    async getRawMaterialExtractorRoles(): Promise<RoleEntity[]> {
        const nodes = await this.supplyChainNodeRepo.find({ where: {}, relations: ['role'] });
        return nodes.filter(({ role }) => role?.isRawMaterialExtractor).map(({ role }) => role);
    }

    findSupplyChainNode(options: FindOneOptions<SupplyChainNodeEntity>): Promise<SupplyChainNodeEntity> {
        return this.supplyChainNodeRepo.findOne(options);
    }

    findSupplyChainMetadata(roleId: string): Promise<SupplyChainNodeMetadataEntity> {
        return this.supplyChainNodeMetadataRepo.findOneByOrFail({ roleId });
    }

    findSupplyChainNodeBy(
        where: FindOptionsWhere<SupplyChainNodeEntity> | FindOptionsWhere<SupplyChainNodeEntity>[]
    ): Promise<SupplyChainNodeEntity> {
        return this.supplyChainNodeRepo.findOneBy(where);
    }

    findSupplyChainNodes(
        where: FindOptionsWhere<SupplyChainNodeEntity> | FindOptionsWhere<SupplyChainNodeEntity>[]
    ): Promise<SupplyChainNodeEntity[]> {
        return this.supplyChainNodeRepo.findBy(where);
    }

    find(options: FindManyOptions<SupplyChainNodeEntity>): Promise<SupplyChainNodeEntity[]> {
        return this.supplyChainNodeRepo.find(options);
    }

    findOneBy(options: FindOptionsWhere<SupplyChainNodeMetadataEntity>): Promise<SupplyChainNodeMetadataEntity> {
        return this.supplyChainNodeMetadataRepo.findOneBy(options);
    }

    async getRolesOrder(): Promise<{ [key: string]: number }> {
        const nodes = await this.supplyChainNodeRepo.find({ order: { createdAt: 'ASC' } });
        return nodes
            .map(({ roleId }) => roleId)
            .reduce((acc, roleId, index) => {
                if (!acc.hasOwnProperty(roleId)) {
                    acc[roleId] = index;
                }
                return acc;
            }, {});
    }

    exists(options: FindManyOptions<SupplyChainNodeEntity>): Promise<boolean> {
        return this.supplyChainNodeRepo.exists(options);
    }

    async getBrandTierIds(role: RoleEntity): Promise<string[]> {
        const brandTiers = await this.supplyChainTierService.getTiers(role, false);
        return brandTiers.flatMap((tiers) => tiers.map((item) => item.id));
    }
}
