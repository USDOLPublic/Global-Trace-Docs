import { ValidateException, ValidateFieldException } from '@diginexhk/nestjs-exception';
import { MailService } from '@diginexhk/nestjs-mailer';
import { TransactionService } from '@diginexhk/nestjs-transaction';
import { BaseQuery, PaginationCollection } from '@diginexhk/typeorm-helper';
import { BadRequestException, Global, Inject, Injectable, forwardRef } from '@nestjs/common';
import bcrypt from 'bcrypt';
import { Dictionary, first, isEmpty } from 'lodash';
import moment from 'moment';
import { FindManyOptions, FindOneOptions, ILike, In, Not } from 'typeorm';
import { env } from '~config/env.config';
import { PRODUCT_NAME } from '~core/constants/product-name.constant';
import { DynamicLinkService } from '~dynamic-link/services/dynamic-link.service';
import { UserRoleEnum } from '~role-permissions/enums/user-role.enum';
import { RoleService } from '~role-permissions/services/role.service';
import { InviteUserMail } from '~users/mails/invite-user.mail';
import { UserEntity } from '../entities/user.entity';
import { UserRepository } from '../repositories/user.repository';
import { OtpService } from './otp.service';
import { UpdateUserProfileDto } from '~users/http/dto/update-user-profile.dto';
import { FacilityService } from '~facilities/services/facility.service';
import { FacilityRoleAttributeService } from '~role-attributes/services/facility-role-attribute.service';
import { CreateNewUserDto } from '~users/http/dto/create-new-user.dto';
import { PaginationParams, SortMultipleParams } from '@diginexhk/nestjs-base-decorator';
import { GetListUserQuery } from '~users/queries/get-list-user.query';
import { UpdateUserDto } from '~users/http/dto/update-user.dto';
import { UserStatusEnum } from '~users/enums/user-status.enum';
import { I18nHelper } from '~core/helpers/i18n.helper';
import { LOCATION_RELATIONS } from '~locations/constants/location-relations.constant';
import { GetMe } from '~users/types/get-me.type';
import { RoleAttributeService } from '~role-attributes/services/role-attribute.service';
import { RoleAttributeResponse } from '~role-attributes/response/role-attribute.response';
import { UserTypeEnum } from '~users/enums/user-type.enum';
import { GrievanceReportService } from '~grievance-report/services/grievance-report.service';
import { FindUsersInFacilityParam } from '~users/types/find-users-in-facility-param.type';

@Global()
@Injectable()
export class UserService extends TransactionService {
    public constructor(
        private userRepo: UserRepository,
        private roleService: RoleService,
        @Inject(forwardRef(() => OtpService)) private otpService: OtpService,
        private dynamicLinkService: DynamicLinkService,
        private mailService: MailService,
        @Inject(forwardRef(() => FacilityService)) private facilityService: FacilityService,
        private facilityRoleAttributeService: FacilityRoleAttributeService,
        private roleAttributeService: RoleAttributeService,
        @Inject(forwardRef(() => GrievanceReportService)) private grievanceReportService: GrievanceReportService
    ) {
        super();
    }

    count(options?: FindManyOptions<UserEntity> | BaseQuery<UserEntity>): Promise<number> {
        return this.userRepo.count(options);
    }

    findUserById(id: string, options?: FindOneOptions<UserEntity>): Promise<UserEntity> {
        return this.userRepo.findUserById(id, options);
    }

    findUserByEmail(email: string, options?: FindOneOptions<UserEntity>) {
        return this.userRepo.findByEmail(email, options);
    }

    findByIdNotFail(id: string) {
        return this.userRepo.findOne({
            where: { id },
            relations: [
                'role',
                'role.permissions',
                'role.hasPermissions',
                'permissions',
                'permissions.roles',
                'facilities',
                'role.selfAssessmentUploadFiles'
            ]
        });
    }

    findOneOrFail(options?: FindOneOptions<UserEntity>): Promise<UserEntity> {
        return this.userRepo.findOneOrFail(options);
    }

    findOneNotFail(options?: FindOneOptions<UserEntity>): Promise<UserEntity> {
        return this.userRepo.findOne(options);
    }

    async changePassword(userId: string, newPassword: string, oldPassword: string) {
        const user = await this.userRepo.findById(userId, { select: ['id', 'password'] });

        if (!bcrypt.compareSync(oldPassword, user.password)) {
            throw new ValidateFieldException('oldPassword', 'incorrect_old_password', 'wrongPass');
        }

        if (oldPassword === newPassword) {
            throw new ValidateFieldException('oldPassword', 'duplicate_old_password', 'wrongPass');
        }

        await this.userRepo.update(user.id, { password: bcrypt.hashSync(newPassword, env.SALT_ROUND) });
    }

    update(id: string, data: Partial<UserEntity>) {
        return this.userRepo.update(id, data);
    }

    findByIds(ids: string[]): Promise<UserEntity[]> {
        return this.userRepo.findBy({ id: In(ids) });
    }

    async generateDynamicLink(link: string): Promise<string> {
        const { url } = await this.dynamicLinkService.createShortDynamicLink(link);
        return url;
    }

    async sendInvitationMail(user: UserEntity, invitedUser: UserEntity | undefined) {
        const isVerified = !isEmpty(invitedUser) && (await this.roleService.canUserLogin(user));

        if (!isVerified) {
            return;
        }

        const otp = await this.otpService.generateNewOtp(invitedUser, env.INVITE_USER_TTL);
        let signUpLink = `${env.WEB_URL}/sign-up?invite-token=${otp.token}`;
        signUpLink = await this.generateDynamicLink(signUpLink);

        return this.mailService.addToQueue(new InviteUserMail(user, invitedUser, signUpLink, PRODUCT_NAME));
    }

    async assignRole(user: UserEntity, role: UserRoleEnum) {
        const roleEntity = await this.roleService.findRoleByName(role);

        return this.userRepo.addRole(user, roleEntity);
    }

    async changeSupplierRole(user: UserEntity, roleId: string) {
        const role = await this.roleService.findRoleById(roleId);
        await this.userRepo.update(user.id, { roleId: role.id });
    }

    updateUserInformation(id: string, userData: Partial<UserEntity> = {}) {
        return this.userRepo.updateOrFail({ id }, userData);
    }

    async finishGuidance(user: UserEntity) {
        await this.userRepo.update(user.id, {
            finishedGuidanceAt: moment().toDate()
        });
    }

    createOne(data: Partial<UserEntity>) {
        return this.userRepo.save(data);
    }

    async validateUniqueEmail(userId: string, email: string) {
        const isExists = await this.userRepo.exists({ where: { id: Not(userId), email: ILike(email) } });
        if (isExists) {
            throw new ValidateException([
                {
                    property: 'email',
                    constraints: {
                        invalidField: { message: 'The email has already been taken.', detail: {} } as any
                    }
                }
            ]);
        }
    }

    async updateProfile(currentUser: UserEntity, dto: UpdateUserProfileDto): Promise<void> {
        const { user, facility, roleAttributes } = dto;
        const hasUpdatedProfile = await this.firstUserHasUpdatedProfile(currentUser.currentFacility.id);

        if (user) {
            await this.updateUserInformation(currentUser.id, user);
        }
        if (facility) {
            await this.facilityService.updateFacilityInformation(currentUser, facility, roleAttributes);
        }
        if (roleAttributes?.length) {
            await this.facilityRoleAttributeService.createFacilityRoleAttributes({
                userId: currentUser.id,
                facilityId: currentUser.currentFacility.id,
                roleId: currentUser.roleId,
                data: roleAttributes,
                hasUpdatedProfile
            });
        }

        await this.userRepo.update(currentUser.id, { updatedProfileAt: moment().toDate() });
    }

    async createNewUser(
        user: UserEntity,
        { email, firstName, lastName, userType }: CreateNewUserDto
    ): Promise<UserEntity> {
        const createdUser = await this.userRepo.save({
            email,
            firstName,
            lastName,
            userType,
            roleId: user.roleId,
            permissions: [],
            facilities: [user.currentFacility]
        });

        await this.sendInvitationMail(user, createdUser);

        return createdUser;
    }

    getUsersByCurrentUser(
        currentUser: UserEntity,
        paginationParams: PaginationParams,
        sortParams: SortMultipleParams[]
    ): Promise<PaginationCollection<UserEntity>> {
        return this.userRepo.pagination(
            new GetListUserQuery(currentUser.id, currentUser.currentFacility.id, sortParams),
            paginationParams
        );
    }

    async updateUser(currentUser: UserEntity, id: string, dto: UpdateUserDto): Promise<void> {
        const user = await this.findAndValidateValidUser(currentUser.currentFacility.id, id);
        if (user.status === UserStatusEnum.INVITED) {
            throw new BadRequestException({ translate: 'error.user_is_not_onboard' });
        }

        await this.userRepo.update({ id }, { ...dto });
    }

    async deleteUser(currentUser: UserEntity, id: string): Promise<void> {
        await this.findAndValidateValidUser(currentUser.currentFacility.id, id);
        if (currentUser.id === id) {
            throw new BadRequestException({ translate: 'error.not_allowed_to_remove_yourself' });
        }

        await this.userRepo.softDelete({ id });
        await this.assignReportToOtherAdmin(id, currentUser.currentFacility?.id);
    }

    async assignReportToOtherAdmin(id: string, facilityId: string): Promise<void> {
        if (!facilityId) {
            return;
        }
        const assigneeIds = await this.getAssigneeIdsInFacility(facilityId, UserTypeEnum.ADMIN);

        if (!assigneeIds.length) {
            return;
        }

        await this.grievanceReportService.update({ assigneeId: id }, { assigneeId: first(assigneeIds) });
    }

    async findAndValidateValidUser(requestorFacilityId: string, userId: string): Promise<UserEntity> {
        const validUser = await this.userRepo.findOneValidUser(userId, requestorFacilityId);
        if (!validUser) {
            throw new BadRequestException({ translate: 'error.invalid_user' });
        }

        return validUser;
    }

    async resendInvitation(currentUser: UserEntity, invitedUserId: string): Promise<void> {
        const invitedUser = await this.findAndValidateValidUser(currentUser.currentFacility.id, invitedUserId);
        if (invitedUser.status !== UserStatusEnum.INVITED) {
            throw new BadRequestException({ translate: 'error.user_signed_up' });
        }

        return this.sendInvitationMail(currentUser, invitedUser);
    }

    async checkIsFirstUser(userId: string, facilityId: string): Promise<boolean> {
        if (!facilityId) {
            return true;
        }

        const firstUserInFacility = await this.userRepo.getFirstUserInFacility(facilityId);
        return userId === firstUserInFacility?.id;
    }

    async getMe(user: UserEntity): Promise<GetMe> {
        if (user.currentFacility) {
            await user.currentFacility.loadRelation(LOCATION_RELATIONS);
            I18nHelper.translateFacilityLocation(user.currentFacility);
        }

        const roleAttributeOrderKeys = this.roleAttributeService.getRoleAttributeOrderKeys();
        const roleAttributes = await this.getRoleAttributes(user.currentFacility?.id, user.id, user.roleId);
        user.role.uploadedSAQ = user.role.selfAssessmentUploadFiles?.length > 0;

        return {
            ...user,
            roleAttributes,
            roleAttributeOrderKeys,
            isFirstUser: await this.checkIsFirstUser(user.id, user.currentFacility?.id)
        };
    }

    private async getRoleAttributes(
        facilityId: string,
        userId: string,
        roleId: string
    ): Promise<Dictionary<RoleAttributeResponse[]>> {
        const [isExistsFacilityRoleAttribute, isExistsUserRoleAttribute] = await Promise.all([
            facilityId
                ? this.facilityRoleAttributeService.isExists({
                      where: { facilityId }
                  })
                : false,
            this.facilityRoleAttributeService.isExists({
                where: { userId }
            })
        ]);

        if (!isExistsFacilityRoleAttribute && !isExistsUserRoleAttribute) {
            return this.roleAttributeService.mappedAttributeWithType(roleId);
        }

        if (!isExistsUserRoleAttribute) {
            const existedFacilityRoleAttribute =
                await this.facilityRoleAttributeService.getFacilityRoleAttributesByFacilityId(
                    facilityId,
                    userId,
                    roleId
                );

            const additionalAttributes = await this.roleAttributeService.mappedAdditionalAttributes(roleId);

            return { ...existedFacilityRoleAttribute, ...additionalAttributes };
        }

        return this.facilityRoleAttributeService.getFacilityRoleAttributesByFacilityId(facilityId, userId, roleId);
    }

    async firstUserHasUpdatedProfile(facilityId: string): Promise<boolean> {
        const firstUser = await this.userRepo.getFirstUserInFacility(facilityId);
        return Boolean(firstUser?.updatedProfileAt);
    }

    findUsersInFacility(param: FindUsersInFacilityParam): Promise<UserEntity[]> {
        return this.userRepo.findUsersInFacility(param);
    }

    async getAssigneeIdsInFacility(facilityId: string, userType?: UserTypeEnum): Promise<string[]> {
        const users = await this.findUsersInFacility({ facilityId, userType });
        return users.map((user) => user.id);
    }
}
