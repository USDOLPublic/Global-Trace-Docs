import { TransactionService } from '@diginexhk/nestjs-transaction';
import { Injectable } from '@nestjs/common';
import { FacilityRoleAttributeRepository } from '~role-attributes/repositories/facility-role-attribute.repository';
import { Dictionary, groupBy, keyBy, map, uniqBy } from 'lodash';
import { AttributeRepository } from '~product-definitions/repositories/attribute.repository';
import { ValueRoleAttributeResponse } from '~role-attributes/response/value-role-attribute.response';
import { FacilityRoleAttributeEntity } from '~role-attributes/entities/facility-role-attribute.entity';
import { FindManyOptions, FindOptionsWhere } from 'typeorm';
import { ValidateRoleAttributeService } from './validate-role-attribute.service';
import { RoleAttributeRepository } from '~role-attributes/repositories/role-attribute.repository';
import { RoleAttributeTypeEnum } from '~role-attributes/enums/role-attribute-type.enum';
import { FieldCategoryEnum } from '~product-definitions/enums/field-category.enum';
import { LocationService } from '~locations/services/location.service';
import { AttributeEntity } from '~product-definitions/entities/attribute.entity';
import { FieldTypeEnum } from '~product-definitions/enums/field-type.enum';
import moment from 'moment';
import { LOCATION } from '~role-attributes/constants/location.constant';
import { CreateFacilityRoleAttributesParam } from '~role-attributes/types/create-facility-role-attributes-param';
import { RoleAttributeEntity } from '~role-attributes/entities/role-attribute.entity';
import { FacilityRoleAttributeDto } from '~role-attributes/http/dto/facility-role-attribute.dto';
import { LocationMapped } from '~role-attributes/types/location-maped.type';

@Injectable()
export class FacilityRoleAttributeService extends TransactionService {
    public constructor(
        private facilityRoleAttributeRepo: FacilityRoleAttributeRepository,
        private attributeRepo: AttributeRepository,
        private roleAttributeRepo: RoleAttributeRepository,
        private validateRoleAttributeService: ValidateRoleAttributeService,
        private locationService: LocationService
    ) {
        super();
    }

    find(options: FindManyOptions<FacilityRoleAttributeEntity>): Promise<FacilityRoleAttributeEntity[]> {
        return this.facilityRoleAttributeRepo.find(options);
    }

    async createFacilityRoleAttributes({
        roleId,
        facilityId,
        userId,
        data,
        hasUpdatedProfile
    }: CreateFacilityRoleAttributesParam): Promise<void> {
        const roleAttributes = await this.findWithHasUpdatedProfile(roleId, hasUpdatedProfile);
        const currentRoleAttributes = await this.facilityRoleAttributeRepo.find({ where: { userId } });
        await this.facilityRoleAttributeRepo.delete({ userId });
        if (hasUpdatedProfile) {
            data = data.filter(
                ({ roleAttributeType }) => roleAttributeType === RoleAttributeTypeEnum.ADDITIONAL_ATTRIBUTE
            );
        }

        this.validateRoleAttributeService.validateDuplicatedAttributes(roleAttributes, data);

        await Promise.all([
            this.validateRoleAttributeService.checkRequiredAttributes(roleAttributes, data),
            this.validateRoleAttributeService.validateAttributesValue(roleAttributes, data)
        ]);

        const mappedFacilityRoleAttributes = groupBy(data, 'roleAttributeType');
        const roleAttributeTypes = Object.keys(mappedFacilityRoleAttributes) as RoleAttributeTypeEnum[];
        const groupedRoleAttributes = groupBy(roleAttributes, 'roleAttributeType');

        await this.facilityRoleAttributeRepo.save(
            roleAttributeTypes.map((roleAttributeType) => {
                const options = mappedFacilityRoleAttributes[roleAttributeType];
                const roleAttributes = groupedRoleAttributes[roleAttributeType];
                return {
                    roleAttributeType,
                    facilityId: roleAttributeType !== RoleAttributeTypeEnum.ADDITIONAL_ATTRIBUTE ? facilityId : null,
                    userId: roleAttributeType === RoleAttributeTypeEnum.ADDITIONAL_ATTRIBUTE ? userId : null,
                    options: this.prepareOptionData(roleAttributes, options, currentRoleAttributes, roleAttributeType)
                };
            })
        );
    }

    private prepareOptionData(
        roleAttributes: RoleAttributeEntity[],
        options: FacilityRoleAttributeDto[],
        currentRoleAttributes: FacilityRoleAttributeEntity[],
        roleAttributeType: RoleAttributeTypeEnum
    ): (FacilityRoleAttributeDto | ValueRoleAttributeResponse)[] {
        const currentOptions =
            currentRoleAttributes.find((item) => item.roleAttributeType === roleAttributeType)?.options || [];
        const uniqueOptions = uniqBy([...options, ...currentOptions], (item) => item.attributeId);
        const mappedUniqueOptions = keyBy(uniqueOptions, 'attributeId');

        const missingRoleAttribues = roleAttributes.reduce((acc, roleAttribute) => {
            if (!mappedUniqueOptions[roleAttribute.attributeId]) {
                acc.push({
                    attributeId: roleAttribute.attributeId,
                    roleAttributeType: roleAttribute.roleAttributeType,
                    value: null
                });
            }
            return acc;
        }, []);

        return uniqueOptions.concat(missingRoleAttribues);
    }

    async getFacilityRoleAttributesByFacilityId(
        facilityId: string,
        userId: string,
        roleId: string
    ): Promise<Dictionary<ValueRoleAttributeResponse[]>> {
        const roleAttributes = await this.roleAttributeRepo.find({
            where: { roleId }
        });
        const roleAttributeMap = keyBy(roleAttributes, (roleAttribute) => roleAttribute.attributeId);

        const facilityRoleAttributes = await this.facilityRoleAttributeRepo.find({
            where: [{ facilityId }, { userId }]
        });

        const [countries, provinces, districts] = await Promise.all([
            this.locationService.listCountries(),
            this.locationService.getListProvinces(),
            this.locationService.getListDistricts()
        ]);

        const countryMap = keyBy(countries, (country) => country.id);
        const provinceMap = keyBy(provinces, (province) => province.id);
        const districtMap = keyBy(districts, (district) => district.id);

        let valueRoleAttributes = facilityRoleAttributes.flatMap(({ options }) =>
            options.map((item) => item)
        ) as ValueRoleAttributeResponse[];

        valueRoleAttributes = await this.getValueRoleAttributes(valueRoleAttributes, roleAttributeMap, {
            countryMap,
            provinceMap,
            districtMap
        });
        return groupBy(valueRoleAttributes, 'roleAttributeType');
    }

    private async getValueRoleAttributes(
        valueRoleAttributes: ValueRoleAttributeResponse[],
        roleAttributeMap: Dictionary<RoleAttributeEntity>,
        locationMapped: LocationMapped
    ): Promise<ValueRoleAttributeResponse[]> {
        const attributes = await this.attributeRepo.findByIds(map(valueRoleAttributes, 'attributeId'));
        const attributesKeyId = keyBy(attributes, 'id');
        let countryAttributeId = '';
        let provinceAttributeId = '';

        return valueRoleAttributes.map((item) => {
            const attribute = attributesKeyId[item.attributeId];
            const dataRoleAttribute: ValueRoleAttributeResponse = {
                ...item,
                isOptional: roleAttributeMap[attribute.id]?.isOptional,
                attribute
            };
            if (item.roleAttributeType === RoleAttributeTypeEnum.OPEN_SUPPLY_HUB_ID) {
                if (attribute.category === FieldCategoryEnum.COUNTRY) {
                    countryAttributeId = attribute.id;
                }
                if (attribute.category === FieldCategoryEnum.PROVINCE) {
                    provinceAttributeId = attribute.id;
                    dataRoleAttribute.countryAttributeId = countryAttributeId;
                }
                if (attribute.category === FieldCategoryEnum.DISTRICT) {
                    dataRoleAttribute.countryAttributeId = countryAttributeId;
                    dataRoleAttribute.provinceAttributeId = provinceAttributeId;
                }
            }
            return {
                ...dataRoleAttribute,
                label: this.getAttributeLabel(attribute, item, locationMapped)
            };
        });
    }

    private getAttributeLabel(
        attribute: AttributeEntity,
        option: ValueRoleAttributeResponse,
        metadata: LocationMapped
    ): any {
        const { value } = option;

        if (attribute.category === FieldCategoryEnum.DATE) {
            return this.getTimeLabel(option);
        }

        if (attribute.category === FieldCategoryEnum.ATTACHMENTS) {
            return this.getAttachmentLabel(option);
        }

        const { countryMap, districtMap, provinceMap } = metadata;

        if (attribute.category === FieldCategoryEnum.COUNTRY_PROVINCE_DISTRICT) {
            if (!value) {
                return '';
            }

            const country = countryMap[value.countryId]?.country;
            const province = provinceMap[value.provinceId]?.province;
            const district = districtMap[value.districtId]?.district;

            return `${country}, ${district}, ${province}`;
        }

        if (attribute.type === FieldTypeEnum.OPEN_SUPPLY_HUB_ID) {
            switch (attribute.name) {
                case LOCATION.COUNTRY:
                    return countryMap[value]?.country;
                case LOCATION.PROVINCE_STATE:
                    return provinceMap[value]?.province;
                case LOCATION.DISTRICT:
                    return districtMap[value]?.district;
            }
        }

        if (attribute.type === FieldTypeEnum.RMI_CID && attribute.category === FieldCategoryEnum.COUNTRY) {
            return countryMap[value]?.country;
        }

        return value;
    }

    private getTimeLabel({ value }: ValueRoleAttributeResponse): string {
        if (!value) {
            return null;
        }

        return moment.unix(value).local().format('DD/MM/YYYY HH:mm');
    }

    private getAttachmentLabel({ value }: ValueRoleAttributeResponse): string {
        if (!value?.length) {
            return '';
        }

        return value.reduce((acc, { fileName }, index) => {
            if (index === value.length - 1) {
                acc = acc + fileName;
            } else {
                acc = acc + fileName + ' \n ';
            }

            return acc;
        }, '') as string;
    }

    isExists(conditions: FindManyOptions<FacilityRoleAttributeEntity>): Promise<boolean> {
        return this.facilityRoleAttributeRepo.exist(conditions);
    }

    private findWithHasUpdatedProfile(
        roleId: string,
        hasUpdatedProfile: boolean = false
    ): Promise<RoleAttributeEntity[]> {
        const findOptionsWhere: FindOptionsWhere<RoleAttributeEntity> = {};
        if (hasUpdatedProfile) {
            findOptionsWhere['roleAttributeType'] = RoleAttributeTypeEnum.ADDITIONAL_ATTRIBUTE;
        }

        return this.roleAttributeRepo.find({
            relations: ['attribute'],
            where: { ...findOptionsWhere, roleId }
        });
    }
}
