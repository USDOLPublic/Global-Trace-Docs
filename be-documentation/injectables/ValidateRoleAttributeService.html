import { ValidateException, ValidationError } from '@diginexhk/nestjs-exception';
import { TransactionService } from '@diginexhk/nestjs-transaction';
import { Injectable } from '@nestjs/common';
import { isUUID } from 'class-validator';
import { Dictionary, first, groupBy, isNull, isUndefined, keyBy } from 'lodash';
import moment from 'moment';
import { createValidationError, extractFirstError } from '~core/helpers/create-validation-error.helper';
import { maxDecimalValue } from '~core/helpers/number.helper';
import { hasAllProperties } from '~core/helpers/type.helper';
import { FileUploadType } from '~core/types/file-upload.type';
import { LocationService } from '~locations/services/location.service';
import { CountryProvinceDistrictAttributeType } from '~events/types/country-province-district-attribute.type';
import { FieldCategoryEnum } from '~product-definitions/enums/field-category.enum';
import { RoleAttributeEntity } from '~role-attributes/entities/role-attribute.entity';
import { FacilityRoleAttributeDto } from '~role-attributes/http/dto/facility-role-attribute.dto';
import { ValidateRoleAttributeOptionType } from '~role-attributes/types/validate-role-attribute-option.type';
import { ValidateRoleAttributeErrorType } from '~role-attributes/types/validate-role-attribute-error.type';
import { RoleAttributeTypeEnum } from '~role-attributes/enums/role-attribute-type.enum';
import { MAX_LENGTH_UNIQUE_ID_ATTRIBUTE } from '~role-attributes/constants/max-length-unique-id-attribute.constants';
import { AttributeRepository } from '~product-definitions/repositories/attribute.repository';
import { FacilityRoleAttributeRepository } from '~role-attributes/repositories/facility-role-attribute.repository';
import { ValueRoleAttributeResponse } from '~role-attributes/response/value-role-attribute.response';

@Injectable()
export class ValidateRoleAttributeService extends TransactionService {
    constructor(
        private locationService: LocationService,
        private facilityRoleAttributeRepo: FacilityRoleAttributeRepository,
        private attributeRepo: AttributeRepository
    ) {
        super();
    }

    checkRequiredAttributes(
        roleAttributes: RoleAttributeEntity[],
        facilityRoleAttributes: FacilityRoleAttributeDto[]
    ): void {
        const requiredAttributes = roleAttributes.filter(
            (roleAttribute) =>
                !roleAttribute.isOptional &&
                Object.values(RoleAttributeTypeEnum).includes(roleAttribute.roleAttributeType)
        );
        const errors: ValidationError[] = [];

        const missingRequiredAttributes: RoleAttributeEntity[] = requiredAttributes.filter(
            (rA) => !!!facilityRoleAttributes.find((attribute) => attribute.attributeId === rA.attributeId)
        );

        if (missingRequiredAttributes.length) {
            const groupedMissingRequiredAttributes = groupBy(missingRequiredAttributes, 'roleAttributeType');
            for (const [key, missingRequiredAttributes] of Object.entries(groupedMissingRequiredAttributes)) {
                const subErrors = this.validateRequiredAttributesEachType(missingRequiredAttributes);
                const { message, detail } = extractFirstError(first(subErrors)) || {};
                errors.push(
                    createValidationError({
                        property: key,
                        message,
                        children: subErrors,
                        detail
                    })
                );
            }
        }

        if (errors.length) throw new ValidateException(errors);
    }

    private validateRequiredAttributesEachType(missingRequiredAttributes: RoleAttributeEntity[]): ValidationError[] {
        return missingRequiredAttributes.map(({ attribute }) => ({
            property: attribute.id,
            constraints: {
                attribute: {
                    message: 'product_attribute_is_required',
                    detail: { attributeName: attribute.name }
                }
            }
        }));
    }

    async validateAttributesValue(
        roleAttributes: RoleAttributeEntity[],
        facilityRoleAttributes: FacilityRoleAttributeDto[]
    ): Promise<void> {
        const errors: ValidationError[] = [];
        const groupedFacilityRoleAttributes = groupBy(facilityRoleAttributes, 'roleAttributeType');
        const groupedExistedUniqueIDAttributes = await this.getGroupedExistedUniqueIDAttributes(roleAttributes);

        for await (const [key, values] of Object.entries(groupedFacilityRoleAttributes)) {
            const childrenValidationErrors = await this.validateAttributesValueEachType(
                roleAttributes,
                values,
                groupedExistedUniqueIDAttributes
            );
            if (childrenValidationErrors.length) {
                const { message, detail } = extractFirstError(first(childrenValidationErrors));
                errors.push(
                    createValidationError({
                        property: key,
                        message,
                        children: childrenValidationErrors,
                        detail
                    })
                );
            }
        }

        if (errors.length) throw new ValidateException(errors);
    }

    async validateAttributesValueEachType(
        roleAttributes: RoleAttributeEntity[],
        facilityRoleAttributesDto: FacilityRoleAttributeDto[],
        groupedExistedUniqueIDAttributes: Dictionary<ValueRoleAttributeResponse[]>
    ): Promise<ValidationError[]> {
        const childrenValidationErrors: ValidationError[] = [];

        for (const { quantityUnit, attributeId, value } of facilityRoleAttributesDto) {
            const roleAttribute = roleAttributes.find((item) => item.attributeId === attributeId);
            if (!roleAttribute) {
                continue;
            }

            const {
                isOptional,
                attribute: { name: attributeName, category, options }
            } = roleAttribute;

            if (isOptional && (isNull(value) || isUndefined(value))) {
                continue;
            }

            if (!value || isNull(value) || isUndefined(value)) {
                childrenValidationErrors.push(
                    this.createAttributeValidationError({
                        attributeName,
                        property: attributeId,
                        message: 'value_should_not_be_empty'
                    })
                );
            }

            await this.validateAttributeValue(
                category,
                {
                    attributeName,
                    options,
                    value,
                    quantityUnit,
                    property: attributeId,
                    groupedExistedUniqueIDAttributes
                },
                childrenValidationErrors
            );
        }

        return childrenValidationErrors;
    }

    private async validateAttributeValue(
        category: FieldCategoryEnum,
        option: ValidateRoleAttributeOptionType,
        errors: ValidationError[]
    ): Promise<void> {
        const { attributeName, options, value, quantityUnit, property, groupedExistedUniqueIDAttributes } = option;

        switch (category) {
            case FieldCategoryEnum.UNIQUE_ID:
                this.validateUniqueId({ property, attributeName, value, groupedExistedUniqueIDAttributes }, errors);
                break;
            case FieldCategoryEnum.TEXT:
                this.validateText({ property, attributeName, value }, errors);
                break;
            case FieldCategoryEnum.NUMBER:
                this.validateNumber({ property, attributeName, value }, errors);
                break;
            case FieldCategoryEnum.PERCENTAGE:
                this.validatePercentage({ property, attributeName, value }, errors);
                break;
            case FieldCategoryEnum.DATE:
                this.validateDate({ property, attributeName, value }, errors);
                break;
            case FieldCategoryEnum.LIST:
                this.validateList({ property, attributeName, options, value }, errors);
                break;
            case FieldCategoryEnum.NUMBER_UNIT_PAIR:
                this.validateNumberUnitPair({ property, attributeName, options, value, quantityUnit }, errors);
                break;
            case FieldCategoryEnum.COUNTRY_PROVINCE_DISTRICT:
                await this.validateCountryProvinceDistrict({ property, attributeName, value }, errors);
                break;
            case FieldCategoryEnum.ATTACHMENTS:
                this.validateAttachments({ property, attributeName, value }, errors);
                break;
        }
    }

    private createAttributeValidationError(error: ValidateRoleAttributeErrorType) {
        const { property, attributeName, attributeType, attributeValue, message } = error;

        return createValidationError({
            property,
            message: message || 'invalid_product_attribute_type',
            detail: {
                attributeName,
                attributeType,
                attributeValue
            }
        });
    }

    private validateUniqueId(option: ValidateRoleAttributeOptionType, errors: ValidationError[]) {
        const { property, attributeName, value, groupedExistedUniqueIDAttributes } = option;
        if (value?.length > MAX_LENGTH_UNIQUE_ID_ATTRIBUTE) {
            return errors.push(
                this.createAttributeValidationError({
                    property,
                    attributeName,
                    message: 'unique_id_should_not_be_exceed_64_characters'
                })
            );
        }

        const existedUniqueIDAttributes = groupedExistedUniqueIDAttributes[property] || [];
        const existedUniqueIDAttributeValues = existedUniqueIDAttributes.map(({ value }) => value);

        if (existedUniqueIDAttributeValues.includes(value)) {
            errors.push(
                this.createAttributeValidationError({
                    property,
                    attributeName,
                    message: 'unique_id_has_been_exist'
                })
            );
        }
    }

    private validateText(option: ValidateRoleAttributeOptionType, errors: ValidationError[]) {
        const { property, attributeName, value } = option;

        if (typeof value !== 'string') {
            errors.push(
                this.createAttributeValidationError({
                    property,
                    attributeName,
                    attributeType: FieldCategoryEnum.TEXT
                })
            );
        }
    }

    private validateNumber(option: ValidateRoleAttributeOptionType, errors: ValidationError[]) {
        const { property, attributeName, value } = option;

        if (typeof value !== 'number') {
            return errors.push(
                this.createAttributeValidationError({
                    property,
                    attributeName,
                    attributeType: FieldCategoryEnum.NUMBER
                })
            );
        }

        if (this.isExceedMaxValue(value)) {
            errors.push(
                this.createAttributeValidationError({
                    property,
                    attributeName,
                    message: 'exceed_maximum_decimal'
                })
            );
        }
    }

    private validatePercentage(option: ValidateRoleAttributeOptionType, errors: ValidationError[]): void {
        const { property, attributeName, value } = option;

        if (typeof value !== 'number' || value < 0 || value > 100) {
            errors.push(
                this.createAttributeValidationError({
                    property,
                    attributeName,
                    attributeType: FieldCategoryEnum.PERCENTAGE,
                    message: 'invalid_product_attribute_percentage'
                })
            );
        }
    }

    private validateDate(option: ValidateRoleAttributeOptionType, errors: ValidationError[]) {
        const { property, attributeName, value } = option;

        if (!moment(value).isValid()) {
            errors.push(
                this.createAttributeValidationError({
                    property,
                    attributeName,
                    attributeType: FieldCategoryEnum.DATE
                })
            );
        }
    }

    private validateList(option: ValidateRoleAttributeOptionType, errors: ValidationError[]) {
        const { property, attributeName, options, value } = option;
        const foundValue = options.find((item) => item.value === value);

        if (!!!foundValue) {
            errors.push(
                this.createAttributeValidationError({
                    property,
                    attributeName,
                    attributeValue: options.map((item) => item.value),
                    message: 'invalid_product_attribute_value'
                })
            );
        }
    }

    private validateNumberUnitPair(option: ValidateRoleAttributeOptionType, errors: ValidationError[]) {
        const { property, attributeName, options, value, quantityUnit } = option;

        if (!quantityUnit || !!!options.find((item) => item.value === quantityUnit)) {
            return errors.push(
                this.createAttributeValidationError({
                    property,
                    attributeName,
                    attributeValue: options.map((item) => item.value),
                    message: 'invalid_quantity_unit'
                })
            );
        }

        if (typeof value !== 'number') {
            return errors.push(
                this.createAttributeValidationError({
                    property,
                    attributeName,
                    attributeType: FieldCategoryEnum.NUMBER_UNIT_PAIR
                })
            );
        }

        if (this.isExceedMaxValue(value)) {
            errors.push(
                this.createAttributeValidationError({
                    property,
                    attributeName,
                    message: 'exceed_maximum_decimal'
                })
            );
        }
    }

    private async validateCountryProvinceDistrict(option: ValidateRoleAttributeOptionType, errors: ValidationError[]) {
        const { property, attributeName, value } = option;
        const sampleCountryProvinceDistrictAttribute: CountryProvinceDistrictAttributeType = {
            countryId: 'string',
            provinceId: 'string',
            districtId: 'string'
        };
        if (!(value instanceof Object) || !hasAllProperties(value, sampleCountryProvinceDistrictAttribute)) {
            return errors.push(
                this.createAttributeValidationError({
                    property,
                    attributeName,
                    attributeType: FieldCategoryEnum.COUNTRY_PROVINCE_DISTRICT
                })
            );
        }

        for (const [key, id] of Object.entries(value)) {
            if (!isUUID(id)) {
                errors.push(
                    this.createAttributeValidationError({
                        property,
                        attributeName: `${attributeName}.${key}`,
                        message: 'product_attribute_must_be_uuid'
                    })
                );

                continue;
            }

            switch (key) {
                case 'countryId':
                    await this.validateCountry(id, option, errors);
                    break;
                case 'provinceId':
                    await this.validateProvince(id, option, errors);
                    break;
                case 'districtId':
                    await this.validateDistrict(id, option, errors);
                    break;
            }
        }
    }

    private async validateCountry(
        countryId: string,
        option: ValidateRoleAttributeOptionType,
        errors: ValidationError[]
    ) {
        const { property, attributeName } = option;

        const country = await this.locationService.findCountryById(countryId);

        if (!country) {
            errors.push(
                this.createAttributeValidationError({
                    property,
                    attributeName,
                    message: 'invalid_country_id'
                })
            );
        }
    }

    private async validateProvince(
        provinceId: string,
        option: ValidateRoleAttributeOptionType,
        errors: ValidationError[]
    ) {
        const { property, attributeName } = option;

        const province = await this.locationService.findProvinceById(provinceId);

        if (!province) {
            errors.push(
                this.createAttributeValidationError({
                    property,
                    attributeName,
                    message: 'invalid_province_id'
                })
            );
        }
    }

    private async validateDistrict(
        districtId: string,
        option: ValidateRoleAttributeOptionType,
        errors: ValidationError[]
    ) {
        const { property, attributeName } = option;

        const district = await this.locationService.findDistrictById(districtId);

        if (!district) {
            errors.push(
                this.createAttributeValidationError({
                    property,
                    attributeName,
                    message: 'invalid_district_id'
                })
            );
        }
    }

    private validateAttachments(option: ValidateRoleAttributeOptionType, errors: ValidationError[]) {
        const { property, attributeName, value } = option;
        const sampleFileUpload: FileUploadType = { fileName: 'string', blobName: 'string' };

        if (!Array.isArray(value)) {
            errors.push(
                this.createAttributeValidationError({
                    property,
                    attributeName,
                    attributeType: FieldCategoryEnum.ATTACHMENTS
                })
            );
        } else if (!value.every((item) => value instanceof Object && hasAllProperties(item, sampleFileUpload))) {
            errors.push(
                this.createAttributeValidationError({
                    property,
                    attributeName,
                    attributeType: 'FileUploadType'
                })
            );
        }
    }

    private isExceedMaxValue(value: number) {
        return value > maxDecimalValue(14, 2);
    }

    private async getGroupedExistedUniqueIDAttributes(
        roleAttributes: RoleAttributeEntity[]
    ): Promise<Dictionary<ValueRoleAttributeResponse[]>> {
        const uniqueIDAttributeIds = roleAttributes.reduce((acc, { attribute }) => {
            if (attribute.category === FieldCategoryEnum.UNIQUE_ID) {
                acc.push(attribute.id);
            }
            return acc;
        }, []);

        const existedFacilityRoleAttributes = await this.facilityRoleAttributeRepo.find();

        const existedUniqueIDAttributes = existedFacilityRoleAttributes.flatMap(({ options }) => {
            return options.reduce((acc, option) => {
                if (uniqueIDAttributeIds.includes(option.attributeId)) {
                    acc.push(option);
                }
                return acc;
            }, []);
        });

        return groupBy(existedUniqueIDAttributes, 'attributeId');
    }

    validateDuplicatedAttributes(
        roleAttributes: RoleAttributeEntity[],
        facilityRoleAttributes: FacilityRoleAttributeDto[]
    ): void {
        const duplicatedAttributes = this.getDuplicatedAttributes(roleAttributes, facilityRoleAttributes);
        if (!duplicatedAttributes.length) {
            return;
        }

        const errors: ValidationError[] = [];
        const groupedDuplicatedAttributes = groupBy(duplicatedAttributes, 'roleAttributeType');
        for (const [key, duplicatedAttributes] of Object.entries(groupedDuplicatedAttributes)) {
            errors.push(
                createValidationError({
                    property: key,
                    message: 'The $property is invalid.',
                    children: this.validateDuplicatedAttribute(duplicatedAttributes),
                    detail: { property: key }
                })
            );
        }

        throw new ValidateException(errors);
    }

    private getDuplicatedAttributes(
        roleAttributes: RoleAttributeEntity[],
        facilityRoleAttributes: FacilityRoleAttributeDto[]
    ): RoleAttributeEntity[] {
        if (!facilityRoleAttributes.length) {
            return [];
        }

        const groupedFacilityRoleAttributes = groupBy(facilityRoleAttributes, 'attributeId');
        const mappedRoleAttributes = keyBy(roleAttributes, 'attributeId');

        return Object.values(groupedFacilityRoleAttributes).reduce(
            (acc: RoleAttributeEntity[], facilityRoleAttributes) => {
                const roleAttribute = mappedRoleAttributes[first(facilityRoleAttributes).attributeId];
                if (facilityRoleAttributes.length > 1 && roleAttribute) {
                    acc.push(roleAttribute);
                }
                return acc;
            },
            []
        );
    }

    private validateDuplicatedAttribute(duplicatedAttributes: RoleAttributeEntity[]): ValidationError[] {
        return duplicatedAttributes.map(({ attribute }) => ({
            property: attribute.id,
            constraints: {
                attribute: {
                    message: 'attribute_is_duplicated',
                    detail: { attributeName: attribute.name }
                }
            }
        }));
    }
}
