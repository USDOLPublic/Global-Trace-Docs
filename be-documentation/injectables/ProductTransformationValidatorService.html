import { BadRequestException, Injectable } from '@nestjs/common';
import { ProductEntity } from '~products/entities/product.entity';
import { UserEntity } from '~users/entities/user.entity';
import { ProductActivityService } from '../product-activity.service';
import { HarvestSeasonService } from '~role-permissions/services/harvest-season.service';
import { TimeRangeType } from '~events/types/time-range.type';
import { convertToKg } from '~events/helpers/convert-to-kg.helper';
import { AttributeService } from '~product-definitions/services/attribute.service';
import { concat, keyBy, sumBy, isEmpty } from 'lodash';
import { AttributeEntity } from '~product-definitions/entities/attribute.entity';
import { FieldTypeEnum } from '~product-definitions/enums/field-type.enum';
import { CalculateFieldEntity } from '~calculate-field/entities/calculate-field.entity';
import { FieldCategoryEnum } from '~product-definitions/enums/field-category.enum';
import { SupplyChainService } from '~supply-chains/services/supply-chain.service';

@Injectable()
export class ProductTransformationValidatorService {
    constructor(
        private harvestSeasonService: HarvestSeasonService,
        private productActivityService: ProductActivityService,
        private attributeService: AttributeService,
        private supplyChainService: SupplyChainService
    ) {}

    async validateTransformations(user: UserEntity, outputProducts: ProductEntity[], inputProducts?: ProductEntity[]) {
        const roleId = user.currentFacility.typeId;
        const metadata = await this.supplyChainService.findSupplyChainMetadata(roleId);
        await metadata.loadRelation('calculateFields');
        const productDefinitionId = outputProducts[0].productDefinitionId;

        const calculateField = metadata.calculateFields.find(
            (calculateField) => calculateField.outputProductDefinitionId === productDefinitionId
        );

        if (!calculateField || isEmpty(calculateField.expressionItems)) {
            await this.validateTransformationsByDefault(user, outputProducts, inputProducts);
        } else {
            if (!isEmpty(inputProducts)) {
                await this.checkOutputLessThanInput(calculateField, inputProducts, outputProducts);
            } else {
                await this.checkStockInHand(calculateField, user, outputProducts);
            }
        }
    }

    async checkStockInHand(
        calculateField: CalculateFieldEntity,
        currentUser: UserEntity,
        outputProducts: ProductEntity[]
    ) {
        const currentFacility = currentUser.currentFacility;
        const timeRange = await this.harvestSeasonService.getCurrentReconciliationWindow(
            currentUser.currentFacility,
            currentUser.role
        );

        const [inputProducts, currentOutputProducts] = await this.productActivityService.getProductsByRange(
            currentFacility.id,
            timeRange
        );

        await this.checkOutputLessThanInput(
            calculateField,
            inputProducts,
            concat(currentOutputProducts, outputProducts)
        );
    }

    private async validateTransformationsByDefault(
        user: UserEntity,
        outputProducts: ProductEntity[],
        inputProducts?: ProductEntity[]
    ): Promise<void> {
        if (!isEmpty(inputProducts)) {
            const inputTotalWeight = this.calculateTotalWeight(inputProducts);
            const outputTotalWeight = this.calculateTotalWeight(outputProducts);
            if (outputTotalWeight > inputTotalWeight) {
                throw new BadRequestException({ translate: 'error.assign_product_id_output_product_list' });
            }
        } else {
            await this.checkStockInHandOldLogic(user, outputProducts);
        }
    }

    private async checkOutputLessThanInput(
        calculateField: CalculateFieldEntity,
        inputProducts: ProductEntity[],
        outputProducts: ProductEntity[]
    ): Promise<void> {
        const attributeIds = concat(inputProducts, outputProducts).flatMap((product) =>
            product.additionalAttributes.map((attribute) => attribute.attributeId)
        );
        const attributes = await this.attributeService.findByIds(attributeIds);
        const attributeMap = keyBy(attributes, (attribute) => attribute.id);

        const inputVariables = inputProducts.reduce((variables, product) => {
            for (const attribute of product.additionalAttributes) {
                const key = `${product.productDefinitionId}-${attribute.attributeId}`;
                variables[key] = this.getAttributeValue(attributeMap, attribute);
            }
            return variables;
        }, {});

        const inputTotalWeight = calculateField.evaluate(inputVariables);

        const outputTotalWeight = sumBy(outputProducts, (product) => {
            const outputAttribute = product.additionalAttributes.find(
                ({ attributeId }) => attributeId === calculateField.outputAttributeId
            );
            if (outputAttribute) {
                return this.getAttributeValue(attributeMap, outputAttribute);
            } else {
                return convertToKg(product.quantity, product.quantityUnit);
            }
        });

        if (outputTotalWeight > inputTotalWeight) {
            throw new BadRequestException({ translate: 'error.assign_product_id_output_product_list' });
        }
    }

    private getAttributeValue(
        attributeMap: Record<string, AttributeEntity>,
        attribute: { attributeId: string; value: any; quantityUnit?: string }
    ): number {
        if (attributeMap[attribute.attributeId].type === FieldTypeEnum.PRODUCT_QUANTITY) {
            return convertToKg(attribute.value, attribute.quantityUnit);
        } else if (attributeMap[attribute.attributeId].category == FieldCategoryEnum.PERCENTAGE) {
            return attribute.value * 0.01;
        } else {
            return attribute.value;
        }
    }

    private calculateTotalWeight(products: { quantity: number; quantityUnit?: string }[]) {
        return products.reduce<number>(
            (previousValue, item) => previousValue + convertToKg(item.quantity, item.quantityUnit),
            0
        );
    }

    async checkStockInHandOldLogic(currentUser: UserEntity, outputProducts: ProductEntity[]) {
        const timeRange: TimeRangeType<Date> = await this.harvestSeasonService.getCurrentReconciliationWindow(
            currentUser.currentFacility,
            currentUser.role
        );

        const totalPurchases = await this.productActivityService.getTotalPurchases(
            currentUser.currentFacility,
            timeRange
        );
        const totalByProducts = await this.productActivityService.getTotalByProducts(
            currentUser.currentFacility,
            timeRange
        );
        const totalOutputs = await this.productActivityService.getTotalOutputs(currentUser.currentFacility, timeRange);
        const totalLoggingOutputs = outputProducts.reduce<number>((previousValue, product) => {
            return previousValue + convertToKg(product.quantity, product.quantityUnit);
        }, 0);

        if (totalPurchases < totalOutputs + totalByProducts + totalLoggingOutputs) {
            throw new BadRequestException({
                translate: 'validation.insufficient_stock_in_hand_please_register_all_purchases'
            });
        }
    }
}
