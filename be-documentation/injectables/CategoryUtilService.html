import { currentLanguage } from '@diginexhk/nestjs-cls-translation';
import { ValidateException, ValidationError } from '@diginexhk/nestjs-exception';
import { TransactionService } from '@diginexhk/nestjs-transaction';
import { forwardRef, Inject, Injectable } from '@nestjs/common';
import { FindManyOptions, In, Not } from 'typeorm';
import { CategoryEntity } from '~categories/entities/category.entity';
import { CategoryTypeEnum } from '~categories/enums/category-type.enum';
import { CategoryValidationService } from '~categories/services/category-validation.service';
import { CategoryService } from '~categories/services/category.service';
import { CreateExternalRiskIndexDto } from '~category-utils/http/dto/create-external-risk-index.dto';
import { CreateNestedCategories } from '~category-utils/http/dto/create-nested-categories.dto';
import { UpdateCategoryDto } from '~category-utils/http/dto/update-category.dto';
import { UpdateExternalRiskIndexDto } from '~category-utils/http/dto/update-external-risk-index.dto';
import { CommodityRiskService } from '~commodity-risks/services/commodity-risk.service';
import { createValidationError } from '~core/helpers/create-validation-error.helper';
import { allSettled } from '~core/helpers/settled.helper';
import { ILike } from '~core/operators/ilike.operator';
import { CapService } from '~corrective-action-plans/services/cap.service';
import { LaborRiskService } from '~labor-risks/services/labor-risk.service';
import { SelfAssessmentQuestionResponseService } from '~self-assessments/services/self-assessment-question-response.service';
import { I18nField } from '~self-assessments/types/i18n-field.type';
import { RiskSeverityEnum } from '~taxonomy-exploitations/enums/risk-severity.enum';

@Injectable()
export class CategoryUtilService extends TransactionService {
    public constructor(
        private capService: CapService,
        @Inject(forwardRef(() => LaborRiskService)) private laborRiskService: LaborRiskService,
        private commodityRiskService: CommodityRiskService,
        private selfAssessmentQuestionResponseService: SelfAssessmentQuestionResponseService,
        private categoryService: CategoryService,
        private categoryValidationService: CategoryValidationService
    ) {
        super();
    }

    async existLaborRiskInputs(subIndicatorIds: string[], indicatorId: string): Promise<boolean> {
        const existedLaborRiskInputs = await Promise.all([
            this.capService.exists({ where: { subIndicatorId: In(subIndicatorIds) } }),
            this.laborRiskService.exists({ where: { subIndicatorId: In(subIndicatorIds), indicatorId } }),
            this.selfAssessmentQuestionResponseService.exists({
                where: { subIndicatorId: In(subIndicatorIds), indicatorId }
            }),
            this.commodityRiskService.exists({ where: { subIndicatorId: In(subIndicatorIds) } })
        ]);

        return existedLaborRiskInputs.some((existedLaborRiskInput) => existedLaborRiskInput);
    }

    getAllNestedCategories(): Promise<CategoryEntity[]> {
        return this.categoryService.getAllNestedCategories();
    }

    createNestedCategories(dto: CreateNestedCategories): Promise<CategoryEntity[]> {
        return this.categoryService.createNestedCategories(dto);
    }

    async updateCategory(id: string, { name, riskSeverity }: UpdateCategoryDto): Promise<void> {
        const translation: I18nField = {};
        translation[`${currentLanguage()}`] = name;

        const existedCategory = await this.categoryService.findOneOrFail({
            where: {
                id,
                type: In([CategoryTypeEnum.CATEGORY, CategoryTypeEnum.SUB_INDICATOR, CategoryTypeEnum.INDICATOR])
            }
        });

        switch (existedCategory.type) {
            case CategoryTypeEnum.CATEGORY:
                await this.validateUpdateCategory(id, name, CategoryTypeEnum.CATEGORY, 'Category');
                await this.categoryService.saveOne({ ...existedCategory, name, translation });
                break;
            case CategoryTypeEnum.SUB_INDICATOR:
                await this.validateUpdateSubIndicator(id, name, existedCategory.parentId, riskSeverity);
                await this.categoryService.saveOne({ ...existedCategory, name, riskSeverity, translation });
                break;
            default:
                await this.validateUpdateIndicator(id, name, CategoryTypeEnum.INDICATOR);
                await this.categoryService.saveOne({ ...existedCategory, name, translation });
                break;
        }
    }

    private async validateUpdateCategory(
        id: string,
        name: string,
        type: CategoryTypeEnum,
        property: string
    ): Promise<void> {
        const errors: ValidationError[] = [];
        const [existedCategory, existedLaborRiskInputs] = await Promise.all([
            this.categoryService.exist({
                where: { id: Not(id), name, type }
            }),
            this.existsLaborRiskInputsWithCategory(id)
        ]);

        if (existedCategory) {
            errors.push(this.categoryValidationService.addValidateNameAlreadyExists(property));
        }

        if (existedLaborRiskInputs) {
            errors.push(this.addValidateExistedLaborRiskInputs(property));
        }

        if (!errors.length) {
            return;
        }

        throw new ValidateException(errors);
    }

    async deleteCategoryById(id: string): Promise<void> {
        const existedCategory = await this.categoryService.findOneOrFail({
            where: {
                id,
                type: In([CategoryTypeEnum.CATEGORY, CategoryTypeEnum.SUB_INDICATOR, CategoryTypeEnum.INDICATOR])
            }
        });

        switch (existedCategory.type) {
            case CategoryTypeEnum.CATEGORY:
                await this.deleteCategory(id);
                break;
            case CategoryTypeEnum.SUB_INDICATOR:
                await this.validateExistedLaborRiskInputs([id], existedCategory.parentId, 'Sub Indicator');
                await this.categoryService.softDelete({ id });
                break;
            default:
                await this.deleteIndicator(id);
                break;
        }
    }

    async existsLaborRiskInputsWithCategory(categoryId: string): Promise<boolean> {
        const indicators = await this.categoryService.findIndicatorsByCategoryId(categoryId);

        const existedLaborRiskInputsWithCategory = await Promise.all(
            indicators.map(({ id: indicatorId, subIndicators }) =>
                this.existLaborRiskInputs(
                    subIndicators.map(({ id }) => id),
                    indicatorId
                )
            )
        );

        return existedLaborRiskInputsWithCategory.some((existedLaborRiskInput) => existedLaborRiskInput);
    }

    private async deleteCategory(id: string): Promise<void> {
        const existedLaborRiskInputs = await this.existsLaborRiskInputsWithCategory(id);

        if (existedLaborRiskInputs) {
            throw new ValidateException([this.addValidateExistedLaborRiskInputs('Category')]);
        }

        await allSettled([
            this.categoryService.softDelete({ id }),
            this.categoryService.softDelete({ categoryId: id })
        ]);
    }

    private async deleteIndicator(id: string): Promise<void> {
        const subIndicators = await this.categoryService.find({ where: { parentId: id } });
        const subIndicatorIds = subIndicators.map(({ id }) => id);

        await this.validateExistedLaborRiskInputs(subIndicatorIds, id, 'Indicator');
        await allSettled([this.categoryService.softDelete({ id }), this.categoryService.softDelete({ parentId: id })]);
    }

    async validateUpdateIndicator(id: string, name: string, type: CategoryTypeEnum): Promise<void> {
        const errors: ValidationError[] = [];
        const existedCategory = await this.categoryService.exist({
            where: { id: Not(id), name, type }
        });

        if (existedCategory) {
            errors.push(this.categoryValidationService.addValidateNameAlreadyExists('Indicator'));
        }

        const subIndicators = await this.categoryService.find({ where: { parentId: id } });
        const subIndicatorIds = subIndicators.map(({ id }) => id);

        const existedLaborRiskInputs = await this.existLaborRiskInputs(subIndicatorIds, id);

        if (existedLaborRiskInputs) {
            errors.push(this.addValidateExistedLaborRiskInputs('Indicator'));
        }

        if (errors.length) {
            throw new ValidateException(errors);
        }
    }

    async validateUpdateSubIndicator(
        id: string,
        name: string,
        parentId: string,
        riskSeverity: RiskSeverityEnum
    ): Promise<void> {
        const errors: ValidationError[] = [];
        if (!riskSeverity) {
            errors.push(
                createValidationError({
                    property: 'The riskSeverity should not be empty',
                    message: '$property should not be empty',
                    detail: { property: 'riskSeverity' }
                })
            );
        }

        const existedSubIndicator = await this.categoryService.exist({
            where: { id: Not(id), type: CategoryTypeEnum.SUB_INDICATOR, parentId, name }
        });

        if (existedSubIndicator) {
            errors.push(this.categoryValidationService.addValidateNameAlreadyExists('Sub Indicator'));
        }

        const existedLaborRiskInputs = await this.existLaborRiskInputs([id], parentId);

        if (existedLaborRiskInputs) {
            errors.push(this.addValidateExistedLaborRiskInputs('Sub Indicator'));
        }

        if (errors.length) {
            throw new ValidateException(errors);
        }
    }

    addValidateExistedLaborRiskInputs(property: string): ValidationError {
        return createValidationError({
            property: 'Labor risk inputs have existed',
            message: 'This $property is linked to existing incident reports and cannot be deleted.',
            detail: { property }
        });
    }

    async validateExistedLaborRiskInputs(
        subIndicatorIds: string[],
        indicatorId: string,
        property: string
    ): Promise<void> {
        const existedLaborRiskInputs = await this.existLaborRiskInputs(subIndicatorIds, indicatorId);
        if (existedLaborRiskInputs) {
            throw new ValidateException([this.addValidateExistedLaborRiskInputs(property)]);
        }
    }

    existsSubAndIndicatorPairById(indicatorId: string, subIndicatorId: string): Promise<boolean> {
        return this.categoryService.existsSubAndIndicatorPairById(indicatorId, subIndicatorId);
    }

    async createExternalRiskIndex(creatorId: string, dto: CreateExternalRiskIndexDto): Promise<CategoryEntity> {
        const existedNameExternalRiskIndex = await this.categoryService.exist({
            where: { name: ILike(dto.name), type: CategoryTypeEnum.EXTERNAL_RISK_INDICATOR }
        });
        this.addValidateExistedExternalRiskIndex(existedNameExternalRiskIndex);

        return this.categoryService.createOne({
            ...dto,
            creatorId,
            type: CategoryTypeEnum.EXTERNAL_RISK_INDICATOR,
            isDefaultExternalRiskIndex: false
        });
    }

    async updateExternalRiskIndex(id: string, dto: UpdateExternalRiskIndexDto): Promise<void> {
        if (dto.name) {
            const existedNameExternalRiskIndex = await this.categoryService.exist({
                where: { id: Not(id), name: ILike(dto.name), type: CategoryTypeEnum.EXTERNAL_RISK_INDICATOR }
            });
            this.addValidateExistedExternalRiskIndex(existedNameExternalRiskIndex);
        }

        await this.categoryService.updateOrFail({ id, type: CategoryTypeEnum.EXTERNAL_RISK_INDICATOR }, dto);
    }

    private addValidateExistedExternalRiskIndex(existedNameExternalRiskIndex: boolean): void {
        if (existedNameExternalRiskIndex) {
            throw new ValidateException([
                {
                    property: 'name',
                    constraints: {
                        invalidField: {
                            message: 'external_risk_index_already_exists',
                            detail: {}
                        } as any
                    }
                }
            ]);
        }
    }

    getExternalRiskIndices(): Promise<CategoryEntity[]> {
        return this.categoryService.find({
            where: { type: CategoryTypeEnum.EXTERNAL_RISK_INDICATOR }
        });
    }

    count(options: FindManyOptions<CategoryEntity>): Promise<number> {
        return this.categoryService.count(options);
    }

    exist(options: FindManyOptions<CategoryEntity>): Promise<boolean> {
        return this.categoryService.exist(options);
    }
}
